# 08-2. JSON 타입과 함수

### JSON 타입 기초

DuckDB는 JSON을 별도의 타입으로 관리한다. 문자열처럼 보이지만 내부적으로는 구조를 인식하고, JSON 전용 함수와 연산자를 통해 필드 탐색·수정·집계가 가능하다.

```sql
-- JSON 타입 선언과 리터럴
SELECT '{"name": "Alice", "age": 30}'::JSON;
SELECT '[1, 2, 3, "four", true, null]'::JSON;

-- JSON 유효성 검사
SELECT JSON_VALID('{"key": "value"}');   -- true
SELECT JSON_VALID('{broken json}');       -- false
SELECT JSON_VALID('null');               -- true
SELECT JSON_VALID('"string"');           -- true

-- JSON 타입 확인
SELECT JSON_TYPE('{"a":1}');            -- 'OBJECT'
SELECT JSON_TYPE('[1,2,3]');            -- 'ARRAY'
SELECT JSON_TYPE('"hello"');            -- 'VARCHAR'
SELECT JSON_TYPE('42');                 -- 'UBIGINT'
SELECT JSON_TYPE('3.14');              -- 'DOUBLE'
SELECT JSON_TYPE('true');              -- 'BOOLEAN'
SELECT JSON_TYPE('null');              -- 'NULL'
```

### 필드 접근 연산자

JSON 데이터에서 특정 필드를 꺼낼 때 가장 자주 쓰이는 것이 `->` 와 `->>` 연산자다. 두 연산자는 생김새가 비슷하지만 반환 타입이 다르므로 혼동하지 않는 것이 중요하다. `->` 는 JSON 타입 그대로 반환하므로 체이닝(중간 탐색)에 쓰고, `->>` 는 최종 값을 VARCHAR로 꺼낼 때 사용한다. 산술 연산이나 비교를 위해 숫자·날짜 타입이 필요하다면 `->>` 로 꺼낸 뒤 `::INTEGER` 같은 캐스트를 추가로 적용한다.

> **주의:** `->`로 꺼낸 값은 JSON 타입이므로 `WHERE data->'age' > 30` 처럼 직접 비교하면 타입 오류가 발생할 수 있다. 숫자 비교가 필요할 때는 반드시 `(data->>'age')::INTEGER > 30` 형태로 변환한다.

```sql
-- -> : JSON 반환 (중간 탐색, 체이닝 가능)
-- ->> : VARCHAR 반환 (최종 값 추출)

SELECT
    data->'name'              AS name_json,    -- JSON 타입
    data->>'name'             AS name_str,     -- VARCHAR 타입
    data->'address'->>'city'  AS city,         -- 중첩 탐색
    data->>'scores'           AS scores_str    -- 배열을 문자열로
FROM (
    SELECT '{
        "name": "Alice",
        "address": {"city": "Seoul", "zip": "04523"},
        "scores": [92, 88, 95]
    }'::JSON AS data
);

-- 배열 인덱스 (0-based)
SELECT
    data->0    AS first,   -- JSON
    data->>1   AS second   -- VARCHAR
FROM (SELECT '[10, 20, 30]'::JSON AS data);
```

### JSON_EXTRACT 함수

```sql
-- JSON_EXTRACT: JSONPath 표현식 사용
SELECT JSON_EXTRACT('{"a":{"b":{"c":42}}}', '$.a.b.c');        -- 42
SELECT JSON_EXTRACT('{"scores":[9,8,10]}', '$.scores[1]');     -- 8
SELECT JSON_EXTRACT('{"scores":[9,8,10]}', '$.scores[*]');     -- [9,8,10]
SELECT JSON_EXTRACT('[{"x":1},{"x":2}]', '$[*].x');            -- [1,2]

-- JSON_EXTRACT_STRING: VARCHAR 반환
SELECT JSON_EXTRACT_STRING('{"name":"Alice"}', '$.name');      -- 'Alice'

-- 여러 경로 동시 추출
SELECT JSON_EXTRACT(data, ['$.name', '$.age', '$.city']) AS vals
FROM (SELECT '{"name":"Alice","age":30,"city":"Seoul"}'::JSON AS data);
-- ['Alice', 30, 'Seoul']
```

### JSON 구조 탐색

```sql
-- 키 목록 조회
SELECT JSON_KEYS('{"a":1,"b":2,"c":3}');       -- ['a','b','c']

-- 배열 길이
SELECT JSON_ARRAY_LENGTH('[1,2,3,4,5]');        -- 5
SELECT JSON_ARRAY_LENGTH('{"key":[1,2,3]}', '$.key'); -- 3

-- 구조 추론
SELECT JSON_STRUCTURE('{"name":"Alice","age":30}');
-- {"name":"VARCHAR","age":"UBIGINT"}

SELECT JSON_STRUCTURE('[{"id":1,"val":1.5},{"id":2,"val":2.5}]');
-- [{"id":"UBIGINT","val":"DOUBLE"}]

-- 중첩 깊이 측정
SELECT JSON_DEPTH('{"a":{"b":{"c":1}}}');   -- 3
SELECT JSON_DEPTH('[1,[2,[3]]]');            -- 3
```

### JSON 생성 함수

```sql
-- JSON 객체 생성
SELECT JSON_OBJECT('name', 'Alice', 'age', 30, 'city', 'Seoul');
-- {"name":"Alice","age":30,"city":"Seoul"}

-- 키-값 리스트에서 생성
SELECT JSON_OBJECT_AGG(k, v)
FROM (VALUES ('a', 1), ('b', 2), ('c', 3)) t(k, v);
-- {"a":1,"b":2,"c":3}

-- JSON 배열 생성
SELECT JSON_ARRAY(1, 'two', true, NULL, [3, 4]);
-- [1,"two",true,null,[3,4]]

-- 값 목록에서 배열 생성
SELECT JSON_GROUP_ARRAY(name ORDER BY salary DESC)
FROM employees
GROUP BY department;
-- ["Carol","Bob","Alice"]

-- 키-값 쌍을 JSON 객체로 집계
SELECT JSON_GROUP_OBJECT(name, salary)
FROM employees
WHERE department = 'Engineering';
-- {"Alice":95000,"Bob":82000,"Carol":105000}

-- 행을 JSON 객체로 변환
SELECT TO_JSON({'name': 'Alice', 'age': 30});
-- {"name":"Alice","age":30}

SELECT ROW_TO_JSON(employees) FROM employees LIMIT 1;
-- {"id":1,"name":"Alice","department":"Engineering","salary":95000,...}
```

### JSON 수정 함수

```sql
-- 값 설정
SELECT JSON_SET(
    '{"name":"Alice","age":30}'::JSON,
    '$.age', 31,          -- 기존 값 덮어쓰기
    '$.city', 'Seoul'     -- 새 키 추가
);
-- {"name":"Alice","age":31,"city":"Seoul"}

-- 키 삭제
SELECT JSON_REMOVE('{"a":1,"b":2,"c":3}'::JSON, '$.b');
-- {"a":1,"c":3}

-- 배열 원소 삭제
SELECT JSON_REMOVE('[1,2,3,4,5]'::JSON, '$[2]');
-- [1,2,4,5]

-- 값 삽입 (기존 값 덮어쓰지 않음)
SELECT JSON_INSERT('{"a":1}'::JSON, '$.b', 2, '$.a', 99);
-- {"a":1,"b":2}   ← a는 기존 값 유지

-- JSON 병합 (깊은 병합)
SELECT JSON_MERGE_PATCH(
    '{"a":1,"b":{"x":10}}'::JSON,
    '{"b":{"y":20},"c":3}'::JSON
);
-- {"a":1,"b":{"y":20},"c":3}  (b가 교체됨)
```

### JSON 포맷팅

```sql
-- 압축 (공백 제거)
SELECT JSON('{"name":  "Alice",  "age" :  30}');
-- {"name":"Alice","age":30}

-- 정렬된 키
SELECT JSON_NORMALIZE('{"b":2,"a":1,"c":3}');
-- {"a":1,"b":2,"c":3}

-- 예쁜 출력 (Python에서)
```
