# 08-3. 중첩 JSON 펼치기

실무 JSON은 대부분 중첩 구조를 갖는다. 배열과 객체가 얽힌 데이터를 관계형 테이블로 변환하는 패턴을 체계적으로 익혀야 한다.

### UNNEST: 배열을 행으로 펼치기

```sql
-- JSON 배열 → 행
SELECT UNNEST(['Seoul', 'Busan', 'Incheon']) AS city;
-- city
-- ──────
-- Seoul
-- Busan
-- Incheon

-- JSON 파일의 배열 열 펼치기
CREATE TABLE orders (
    id    INTEGER,
    items JSON   -- [{"product":"apple","qty":3}, {"product":"banana","qty":1}]
);
INSERT INTO orders VALUES
    (1, '[{"product":"apple","qty":3},{"product":"banana","qty":1}]'),
    (2, '[{"product":"cherry","qty":5}]');

-- 배열 펼치기
SELECT
    o.id,
    item->>'product' AS product,
    (item->>'qty')::INTEGER AS qty
FROM orders o,
     UNNEST(o.items::JSON[]) AS t(item);
-- id | product | qty
-- ───┼─────────┼────
-- 1  | apple   | 3
-- 1  | banana  | 1
-- 2  | cherry  | 5
```

### JSON_EACH: 객체를 키-값 행으로 펼치기

```sql
-- JSON 객체의 각 키-값 쌍을 행으로
SELECT key, value
FROM JSON_EACH('{"apple":120,"banana":80,"cherry":250}');
-- key    | value
-- ───────┼──────
-- apple  | 120
-- banana | 80
-- cherry | 250

-- 테이블 열에 적용
SELECT
    product_id,
    attr.key   AS attribute,
    attr.value AS val
FROM products,
     JSON_EACH(attributes) AS attr;

-- 특정 타입으로 캐스팅
SELECT
    key   AS month,
    value::INTEGER AS revenue
FROM JSON_EACH('{"Jan":1200,"Feb":980,"Mar":1540}');
```

### JSON_ARRAY_ELEMENTS: 배열 원소를 행으로

```sql
-- 배열 원소 하나씩 행으로
SELECT value
FROM JSON_ARRAY_ELEMENTS('[10, 20, 30, 40, 50]');
-- value
-- ─────
-- 10
-- 20
-- ...

-- 인덱스와 함께
SELECT ordinality AS idx, value
FROM JSON_ARRAY_ELEMENTS('[10,20,30]') WITH ORDINALITY;
-- idx | value
-- ────┼──────
-- 1   | 10
-- 2   | 20
-- 3   | 30

-- 중첩 배열 원소 접근
SELECT
    id,
    elem->>'name'  AS tag_name,
    elem->>'score' AS tag_score
FROM articles,
     JSON_ARRAY_ELEMENTS(tags) AS elem
WHERE (elem->>'score')::DOUBLE > 0.8;
```

### 다단계 중첩 펼치기

실무에서 자주 만나는 다단계 중첩 구조를 단계별로 펼친다.

```sql
-- 예시 데이터
-- {
--   "order_id": 101,
--   "customer": {"id": 5, "name": "Alice"},
--   "items": [
--     {"product_id": 10, "name": "Apple", "qty": 3,
--      "options": [{"key": "color", "val": "red"}]},
--     {"product_id": 20, "name": "Banana", "qty": 1,
--      "options": []}
--   ]
-- }

CREATE TABLE raw_orders (data JSON);

-- 1단계: 최상위 필드 추출
WITH base AS (
    SELECT
        (data->>'order_id')::INTEGER             AS order_id,
        data->'customer'->>'name'                AS customer_name,
        data->'items'                            AS items_json
    FROM raw_orders
),
-- 2단계: items 배열 펼치기
items_flat AS (
    SELECT
        order_id,
        customer_name,
        (item->>'product_id')::INTEGER           AS product_id,
        item->>'name'                            AS product_name,
        (item->>'qty')::INTEGER                  AS qty,
        item->'options'                          AS options_json
    FROM base,
         UNNEST(items_json::JSON[]) AS t(item)
),
-- 3단계: options 배열 펼치기
options_flat AS (
    SELECT
        order_id,
        customer_name,
        product_id,
        product_name,
        qty,
        opt->>'key'   AS option_key,
        opt->>'val'   AS option_val
    FROM items_flat,
         UNNEST(
             CASE
                 WHEN JSON_ARRAY_LENGTH(options_json) > 0
                 THEN options_json::JSON[]
                 ELSE ARRAY['{"key":null,"val":null}'::JSON]
             END
         ) AS t(opt)
)
SELECT * FROM options_flat;
```

### STRUCT로 중첩 JSON 컬럼 변환

```sql
-- JSON → STRUCT (스키마가 일정할 때)
SELECT
    id,
    (data->>'name')::VARCHAR  AS name,
    -- address 객체를 STRUCT로
    STRUCT_PACK(
        city    := data->'address'->>'city',
        zip     := data->'address'->>'zip',
        street  := data->'address'->>'street'
    ) AS address
FROM (SELECT 1 AS id, '{
    "name":"Alice",
    "address":{"city":"Seoul","zip":"04523","street":"테헤란로 123"}
}'::JSON AS data);

-- STRUCT 접근
SELECT address.city FROM above_query;
```

### JSON 컬럼을 관계형 테이블로 변환

```sql
-- API 로그 테이블 (JSON 구조)
CREATE TABLE api_logs AS
SELECT * FROM read_json_auto('api_logs.ndjson');
-- 컬럼: ts TIMESTAMP, method VARCHAR, path VARCHAR,
--       request JSON, response JSON, meta JSON

-- 완전히 펼친 분석 테이블 생성
CREATE TABLE api_logs_flat AS
SELECT
    ts,
    method,
    path,
    -- request 필드
    (request->>'user_id')::INTEGER       AS user_id,
    request->>'session_id'               AS session_id,
    -- response 필드
    (response->>'status')::INTEGER       AS status_code,
    (response->>'duration_ms')::DOUBLE   AS duration_ms,
    response->>'error'                   AS error_msg,
    -- meta 필드
    meta->>'ip'                          AS client_ip,
    meta->>'user_agent'                  AS user_agent,
    -- 오류 여부 파생 열
    (response->>'status')::INTEGER >= 400 AS is_error
FROM api_logs;

-- 이후 분석
SELECT
    DATE_TRUNC('hour', ts) AS hour,
    COUNT(*) AS requests,
    AVG(duration_ms) AS avg_ms,
    SUM(is_error::INTEGER) AS errors,
    SUM(is_error::INTEGER) * 100.0 / COUNT(*) AS error_rate
FROM api_logs_flat
GROUP BY 1
ORDER BY 1;
```

### 스키마가 다른 JSON 처리

```sql
-- 이벤트 타입마다 payload 구조가 다른 경우
-- {"type":"login",    "payload":{"user_id":1,"ip":"1.2.3.4"}}
-- {"type":"purchase", "payload":{"user_id":1,"product_id":42,"amount":99.9}}
-- {"type":"logout",   "payload":{"user_id":1,"session_duration":1800}}

SELECT
    ts,
    event_type,
    -- 공통 필드
    (payload->>'user_id')::INTEGER           AS user_id,
    -- login 전용
    CASE WHEN event_type = 'login'
         THEN payload->>'ip' END             AS login_ip,
    -- purchase 전용
    CASE WHEN event_type = 'purchase'
         THEN (payload->>'amount')::DECIMAL(10,2) END AS purchase_amount,
    CASE WHEN event_type = 'purchase'
         THEN (payload->>'product_id')::INTEGER END   AS product_id,
    -- logout 전용
    CASE WHEN event_type = 'logout'
         THEN (payload->>'session_duration')::INTEGER END AS session_sec
FROM events;

-- 또는 PIVOT으로 이벤트 타입별 집계
PIVOT (
    SELECT
        DATE_TRUNC('day', ts) AS day,
        event_type,
        COUNT(*) AS cnt
    FROM events
    GROUP BY 1, 2
)
ON event_type
USING SUM(cnt)
GROUP BY day;
```

### JSON 배열 집계와 재구성

```sql
-- 행 → JSON 배열로 집계
SELECT
    department,
    JSON_GROUP_ARRAY(
        JSON_OBJECT('name', name, 'salary', salary)
        ORDER BY salary DESC
    ) AS members
FROM employees
GROUP BY department;
-- department  | members
-- ────────────┼──────────────────────────────────────────
-- Engineering | [{"name":"Carol","salary":105000},...]

-- 결과를 JSON 파일로 저장
COPY (
    SELECT
        department,
        JSON_GROUP_ARRAY(
            JSON_OBJECT('id', id, 'name', name, 'salary', salary)
        ) AS employees
    FROM employees
    GROUP BY department
) TO 'dept_employees.json' (FORMAT JSON, ARRAY true);
```

### 성능 고려사항

```sql
-- 팁 1: 자주 쓰는 JSON 필드는 별도 열로 구체화
CREATE TABLE events_materialized AS
SELECT
    id,
    ts,
    event_type,
    (payload->>'user_id')::INTEGER AS user_id,  -- 구체화
    (payload->>'amount')::DOUBLE   AS amount,   -- 구체화
    payload                                      -- 원본 보존
FROM events;
-- user_id, amount는 이제 네이티브 타입으로 빠른 필터링 가능

-- 팁 2: JSON 경로 탐색은 GENERATED 열로
CREATE TABLE logs (
    raw    JSON,
    level  VARCHAR GENERATED ALWAYS AS (raw->>'level')  VIRTUAL,
    ts     TIMESTAMP GENERATED ALWAYS AS
               (TRY_CAST(raw->>'ts' AS TIMESTAMP))       VIRTUAL
);

-- 팁 3: 대용량 NDJSON은 스트리밍 처리
SELECT COUNT(*), AVG((data->>'amount')::DOUBLE)
FROM read_json(
    'huge.ndjson',
    format    = 'newline_delimited',
    sample_size = 1000  -- 타입 감지 샘플 줄여서 빠르게 시작
);
```
