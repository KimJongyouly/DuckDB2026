# 12-2. Pandas / Polars / Arrow와 연동

DuckDB의 Python 연동에서 가장 강력한 기능은 외부 DataFrame·Table을 복사 없이(Zero-copy) SQL로 쿼리하는 것이다.

### Pandas 연동

```python
import duckdb
import pandas as pd

con = duckdb.connect()

# DataFrame → DuckDB 쿼리 (Zero-copy 참조)
df = pd.read_csv('sales.csv')
con.register('sales_df', df)

result = con.execute("""
    SELECT region, SUM(amount) AS total
    FROM sales_df
    WHERE order_date >= '2024-01-01'
    GROUP BY region
    ORDER BY total DESC
""").df()

# 또는 register 없이 직접 참조 (Python 변수명이 테이블 이름)
result = con.execute("SELECT * FROM df LIMIT 5").df()
# con.execute()는 로컬 네임스페이스의 'df'를 자동으로 찾음

# DuckDB 결과 → DataFrame
df_result = con.execute("""
    SELECT department, AVG(salary) AS avg_sal
    FROM employees
    GROUP BY department
""").df()
print(type(df_result))  # <class 'pandas.core.frame.DataFrame'>

# DuckDB → DataFrame 변환 방법들
result = con.execute("SELECT * FROM employees").df()           # Pandas DataFrame
result = con.execute("SELECT * FROM employees").fetchdf()      # 동일
result = con.execute("SELECT * FROM employees").pl()           # Polars DataFrame
result = con.execute("SELECT * FROM employees").arrow()        # PyArrow Table
result = con.execute("SELECT * FROM employees").fetchnumpy()   # NumPy dict
```

### Pandas 연동 심화

```python
import duckdb
import pandas as pd
import numpy as np

con = duckdb.connect()

# 대용량 DataFrame 처리: DuckDB가 Pandas보다 빠른 집계
large_df = pd.DataFrame({
    'id':     np.arange(10_000_000),
    'val':    np.random.randn(10_000_000),
    'group':  np.random.choice(['A','B','C','D'], 10_000_000)
})

# Pandas 방식 (느림)
# result = large_df.groupby('group')['val'].agg(['mean','std','count'])

# DuckDB 방식 (빠름 — 멀티스레드 벡터화 처리)
result = con.execute("""
    SELECT
        "group",
        AVG(val)   AS mean,
        STDDEV(val) AS std,
        COUNT(*)   AS cnt
    FROM large_df
    GROUP BY "group"
""").df()

# DataFrame을 DuckDB 테이블로 영구 저장
con2 = duckdb.connect('analytics.duckdb')
con2.execute("CREATE TABLE sales AS SELECT * FROM df")

# DuckDB 테이블 → DataFrame → 수정 → 다시 저장
df_mod = con2.execute("SELECT * FROM sales").df()
df_mod['tax'] = df_mod['amount'] * 0.1
con2.execute("DROP TABLE sales")
con2.execute("CREATE TABLE sales AS SELECT * FROM df_mod")
```

### Polars 연동

```python
import duckdb
import polars as pl

con = duckdb.connect()

# Polars DataFrame → DuckDB (Arrow 기반 Zero-copy)
pl_df = pl.read_csv('sales.csv')
con.register('pl_sales', pl_df)

result = con.execute("""
    SELECT region, COUNT(*) AS cnt, SUM(amount) AS total
    FROM pl_sales
    GROUP BY region
""").pl()    # 결과도 Polars DataFrame으로
print(type(result))  # <class 'polars.DataFrame'>

# 직접 참조 패턴
result = con.execute("SELECT * FROM pl_df LIMIT 5").pl()

# Polars LazyFrame은 먼저 collect() 필요
lazy = pl.scan_parquet('data/*.parquet')
eager = lazy.collect()
con.register('pl_lazy_data', eager)

# DuckDB 결과 → Polars
df_result = con.execute("""
    SELECT YEAR(order_date) AS year, SUM(amount) AS revenue
    FROM sales
    GROUP BY year
""").pl()

# Polars 네이티브 연산과 DuckDB 혼합
pl_filtered = pl_df.filter(pl.col('amount') > 1000)
con.register('filtered', pl_filtered)
summary = con.execute("""
    SELECT region, AVG(amount) FROM filtered GROUP BY region
""").pl()
```

### PyArrow 연동

```python
import duckdb
import pyarrow as pa
import pyarrow.parquet as pq

con = duckdb.connect()

# Parquet → Arrow Table → DuckDB (Zero-copy)
arrow_table = pq.read_table('sales.parquet')
con.register('arrow_sales', arrow_table)

result = con.execute("""
    SELECT * FROM arrow_sales WHERE amount > 10000
""").arrow()  # 결과도 Arrow Table로

# Arrow Table → Parquet 저장
pq.write_table(result, 'filtered_sales.parquet')

# Arrow RecordBatch 스트리밍
import pyarrow.dataset as ds

dataset = ds.dataset('data/', format='parquet')
for batch in dataset.to_batches():
    con.register('batch', batch)
    summary = con.execute("""
        SELECT SUM(amount) FROM batch
    """).fetchone()[0]

# DuckDB → Arrow (대용량 결과 스트리밍)
reader = con.execute("SELECT * FROM large_table").fetch_arrow_reader(10000)
while True:
    try:
        batch = reader.read_next_batch()
        process(batch)
    except StopIteration:
        break
```

### 데이터프레임 자동 감지 (Magic Tables)

DuckDB Python 클라이언트는 `con.execute()` 호출 시 SQL에 나타나는 이름을 로컬·전역 네임스페이스에서 자동으로 찾아 테이블로 등록한다.

```python
import duckdb
import pandas as pd

con = duckdb.connect()

employees_df = pd.read_csv('employees.csv')    # 변수명: employees_df
orders_df    = pd.read_parquet('orders.parquet')

# 변수명을 테이블명처럼 직접 사용
result = con.execute("""
    SELECT e.name, COUNT(o.id) AS order_count
    FROM employees_df e
    JOIN orders_df o ON e.id = o.employee_id
    GROUP BY e.name
    ORDER BY order_count DESC
""").df()

# 함수 내부에서도 동작 (로컬 변수 감지)
def analyze(df):
    return duckdb.execute("""
        SELECT region, SUM(amount) AS total
        FROM df                          -- 로컬 변수 'df' 자동 참조
        GROUP BY region
    """).df()
```

### 성능 비교: Pandas vs DuckDB

```python
import duckdb
import pandas as pd
import numpy as np
import time

# 1000만 행 데이터 준비
n = 10_000_000
df = pd.DataFrame({
    'category': np.random.choice(['A','B','C','D','E'], n),
    'value':    np.random.randn(n) * 1000,
    'flag':     np.random.choice([True, False], n)
})

con = duckdb.connect()

# Pandas GroupBy
t0 = time.time()
_ = df[df.flag].groupby('category')['value'].agg(['mean','sum','count'])
print(f"Pandas:  {time.time()-t0:.2f}s")

# DuckDB
t0 = time.time()
_ = con.execute("""
    SELECT category,
           AVG(value)   AS mean,
           SUM(value)   AS sum,
           COUNT(*)     AS count
    FROM df
    WHERE flag = true
    GROUP BY category
""").df()
print(f"DuckDB:  {time.time()-t0:.2f}s")
# 일반적으로 DuckDB가 2~10배 빠름 (멀티코어 활용)
```
