# 03-6. SAMPLE, UNNEST, VALUES

### SAMPLE: 무작위 샘플링

대용량 데이터를 분석할 때 전체 데이터 대신 샘플로 빠른 탐색이 가능하다.

```sql
-- 행 수 기준 샘플링 (RESERVOIR 방식, 재현 가능)
SELECT * FROM large_table USING SAMPLE 1000;

-- 비율 기준 샘플링 (약 10%)
SELECT * FROM large_table USING SAMPLE 10 PERCENT;

-- BERNOULLI: 각 행을 독립적으로 확률에 따라 포함
SELECT * FROM large_table USING SAMPLE 10% (BERNOULLI);

-- RESERVOIR: 정확히 N개, 메모리 효율적
SELECT * FROM large_table USING SAMPLE 500 (RESERVOIR);

-- SYSTEM: 블록 단위 샘플링, 가장 빠름 (근사치)
SELECT * FROM large_table USING SAMPLE 5% (SYSTEM);

-- 재현 가능한 샘플: 시드 고정
SELECT * FROM large_table USING SAMPLE 10% (BERNOULLI, 42);
```

```
샘플링 방식 비교
────────────────────────────────────────────
BERNOULLI  정확도 높음   메모리 보통   재현 가능
RESERVOIR  정확한 N개   메모리 O(N)   재현 가능
SYSTEM     가장 빠름    메모리 낮음   근사치
```

### UNNEST: 배열과 리스트 펼치기

```sql
-- 배열 리터럴 펼치기
SELECT UNNEST([1, 2, 3, 4, 5]) AS num;

-- 열에 담긴 배열 펼치기
CREATE TABLE orders (
    order_id INTEGER,
    items VARCHAR[]
);
INSERT INTO orders VALUES
    (1, ['apple', 'banana', 'cherry']),
    (2, ['mango', 'kiwi']);

-- UNNEST로 배열을 행으로 변환
SELECT order_id, UNNEST(items) AS item
FROM orders;
-- 결과:
-- order_id | item
-- ─────────┼────────
-- 1        | apple
-- 1        | banana
-- 1        | cherry
-- 2        | mango
-- 2        | kiwi

-- 인덱스와 함께 (WITH ORDINALITY 패턴)
SELECT order_id, idx, item
FROM orders,
     UNNEST(items) WITH ORDINALITY AS t(item, idx);

-- 다중 배열 동시 UNNEST (zip 방식)
SELECT UNNEST([1,2,3]) AS num, UNNEST(['a','b','c']) AS letter;
```

### VALUES: 인라인 데이터

```sql
-- 단순 인라인 테이블
SELECT * FROM (VALUES
    (1, 'Alice', 75000),
    (2, 'Bob',   82000),
    (3, 'Carol', 91000)
) AS t(id, name, salary);

-- INSERT에서 활용
INSERT INTO employees (id, name, salary)
VALUES
    (4, 'David', 68000),
    (5, 'Eve',   95000);

-- CTE와 함께
WITH sample_data(id, name, score) AS (
    VALUES (1, 'Alice', 92),
           (2, 'Bob', 85),
           (3, 'Carol', 78)
)
SELECT name, score,
       RANK() OVER (ORDER BY score DESC) AS rank
FROM sample_data;
```
