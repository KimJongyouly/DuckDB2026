# 06-1. 테이블 생성·변경·삭제

데이터를 저장하려면 먼저 어떤 형태로 저장할지 구조를 정해야 한다. CREATE TABLE은 그 구조를 DuckDB에 등록하는 명령이고, ALTER TABLE과 DROP TABLE은 이미 등록된 구조를 수정하거나 폐기하는 명령이다. 요구사항이 바뀔 때 스키마를 손쉽게 변경할 수 있다는 점이 DuckDB의 장점 중 하나다.

### CREATE TABLE

```sql
-- 기본 구조
CREATE TABLE employees (
    id          INTEGER PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    email       VARCHAR UNIQUE,
    department  VARCHAR,
    salary      DECIMAL(12, 2) DEFAULT 0,
    hire_date   DATE DEFAULT CURRENT_DATE,
    active      BOOLEAN DEFAULT true,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- IF NOT EXISTS: 이미 존재하면 무시
CREATE TABLE IF NOT EXISTS employees (
    id INTEGER PRIMARY KEY,
    name VARCHAR NOT NULL
);

-- OR REPLACE: 기존 테이블 교체
CREATE OR REPLACE TABLE employees (
    id INTEGER PRIMARY KEY,
    name VARCHAR NOT NULL
);
```

### 컬럼 제약조건

```sql
CREATE TABLE orders (
    -- 인라인 제약조건 (열 수준)
    id          INTEGER PRIMARY KEY,                        -- 기본키
    product_id  INTEGER NOT NULL,                           -- NULL 불가
    customer_id INTEGER REFERENCES customers(id),           -- 외래키
    email       VARCHAR UNIQUE,                             -- 고유값
    amount      DECIMAL(10,2) CHECK (amount > 0),           -- 값 범위
    status      VARCHAR DEFAULT 'pending',                  -- 기본값
    code        VARCHAR GENERATED ALWAYS AS                 -- 생성 열
                (UPPER(LEFT(status, 3))) VIRTUAL,

    -- 테이블 수준 제약조건
    CONSTRAINT pk_orders PRIMARY KEY (id),
    CONSTRAINT fk_product FOREIGN KEY (product_id)
        REFERENCES products(id) ON DELETE CASCADE,
    CONSTRAINT chk_amount CHECK (amount BETWEEN 0 AND 1000000),
    CONSTRAINT uq_order_email UNIQUE (customer_id, email)
);
```

### GENERATED (생성 열)

값을 직접 저장하는 대신 다른 열로부터 계산하여 제공하는 열을 생성 열(Generated Column)이라 한다. 세금 포함 가격이나 대문자 표기처럼 파생 값을 별도로 관리하지 않아도 되므로 데이터 일관성을 자연스럽게 유지할 수 있다.

```sql
CREATE TABLE products (
    id          INTEGER PRIMARY KEY,
    price       DECIMAL(10, 2),
    tax_rate    DECIMAL(5, 4) DEFAULT 0.1,

    -- VIRTUAL: 저장하지 않고 읽을 때마다 계산
    price_with_tax DECIMAL(10, 2)
        GENERATED ALWAYS AS (price * (1 + tax_rate)) VIRTUAL,

    -- 실제로는 DuckDB가 VIRTUAL만 지원
    -- STORED는 향후 지원 예정
    name        VARCHAR,
    name_upper  VARCHAR GENERATED ALWAYS AS (UPPER(name)) VIRTUAL
);

SELECT id, price, price_with_tax FROM products;
-- price_with_tax는 조회 시점에 계산
```

> **주의:** DuckDB는 현재 VIRTUAL 방식의 생성 열만 지원한다. STORED(디스크에 저장) 방식은 아직 지원되지 않으므로, 생성 열을 WHERE 조건이나 ORDER BY에 자주 사용한다면 인덱스 적용 여부를 별도로 검토해야 한다.

### CREATE TABLE AS SELECT (CTAS)

쿼리 결과를 즉시 새 테이블로 저장하는 패턴이다. 분석 작업 중 중간 결과를 영구 테이블로 굳히거나, 외부 파일에서 데이터를 한 번에 로드할 때 특히 유용하다.

```sql
-- 쿼리 결과로 테이블 생성
CREATE TABLE high_earners AS
SELECT * FROM employees WHERE salary > 100000;

-- 구조만 복사 (데이터 없음)
CREATE TABLE employees_backup AS
SELECT * FROM employees WHERE false;

-- 파일에서 직접 테이블 생성
CREATE TABLE sales AS
SELECT * FROM read_parquet('data/sales_*.parquet');

CREATE TABLE logs AS
SELECT * FROM read_csv('logs/*.csv', header=true);

-- 외부 파일 참조 (데이터 복사 없음)
CREATE VIEW raw_data AS
SELECT * FROM read_parquet('s3://bucket/data/*.parquet');
```

> **팁:** `SELECT * FROM employees WHERE false`처럼 항상 거짓인 조건을 붙이면 행은 하나도 복사하지 않고 열 구조만 그대로 가져온다. 빈 테이블을 빠르게 만들 때 활용할 수 있다.

### ALTER TABLE: 테이블 변경

```sql
-- 열 추가
ALTER TABLE employees ADD COLUMN phone VARCHAR;
ALTER TABLE employees ADD COLUMN score DOUBLE DEFAULT 0.0;

-- 열 삭제
ALTER TABLE employees DROP COLUMN phone;
ALTER TABLE employees DROP COLUMN IF EXISTS phone;

-- 열 이름 변경
ALTER TABLE employees RENAME COLUMN score TO performance_score;

-- 열 타입 변경
ALTER TABLE employees ALTER COLUMN salary TYPE BIGINT;
ALTER TABLE employees ALTER COLUMN hire_date SET DATA TYPE TIMESTAMP;

-- 기본값 설정/제거
ALTER TABLE employees ALTER COLUMN active SET DEFAULT true;
ALTER TABLE employees ALTER COLUMN score DROP DEFAULT;

-- NOT NULL 설정/제거
ALTER TABLE employees ALTER COLUMN email SET NOT NULL;
ALTER TABLE employees ALTER COLUMN email DROP NOT NULL;

-- 테이블 이름 변경
ALTER TABLE employees RENAME TO staff;

-- 제약조건 추가/삭제
ALTER TABLE employees ADD CONSTRAINT uq_email UNIQUE (email);
ALTER TABLE employees DROP CONSTRAINT uq_email;
```

### DROP TABLE: 테이블 삭제

테이블을 완전히 제거할 때는 DROP TABLE을 사용한다. 데이터와 구조가 모두 사라지므로, 실수를 방지하려면 먼저 백업하거나 트랜잭션 안에서 실행하는 것이 안전하다. 데이터는 지우되 구조는 남기고 싶다면 TRUNCATE를 대신 사용한다.

```sql
-- 기본 삭제
DROP TABLE employees;

-- 없어도 오류 없이
DROP TABLE IF EXISTS employees;

-- CASCADE: 의존하는 뷰·제약조건도 함께 삭제
DROP TABLE employees CASCADE;

-- 데이터만 삭제 (구조 유지)
TRUNCATE TABLE employees;
DELETE FROM employees;  -- 동일하지만 WHERE 절 사용 가능
```

> **주의:** DROP TABLE CASCADE는 해당 테이블을 참조하는 뷰와 외래키 제약조건까지 함께 삭제한다. 의도치 않은 연쇄 삭제가 발생할 수 있으므로 반드시 의존 관계를 먼저 확인한다.

### 스키마와 데이터베이스

```sql
-- 스키마 생성
CREATE SCHEMA analytics;
CREATE SCHEMA IF NOT EXISTS reporting;

-- 스키마에 테이블 생성
CREATE TABLE analytics.daily_stats (
    date    DATE,
    visits  INTEGER
);

-- 스키마 삭제
DROP SCHEMA analytics;
DROP SCHEMA analytics CASCADE;  -- 내부 객체 포함

-- 데이터베이스 목록 조회
SHOW DATABASES;
SHOW SCHEMAS;
SHOW TABLES;
SHOW ALL TABLES;  -- 모든 스키마의 테이블
DESCRIBE employees;  -- 테이블 구조
```

테이블 구조가 갖춰졌다면 이제 데이터를 채울 차례다. 다음 절에서는 DML 명령으로 데이터를 삽입·수정·삭제하는 방법과, DuckDB가 제공하는 UPSERT·RETURNING 같은 고급 기능을 살펴본다.
