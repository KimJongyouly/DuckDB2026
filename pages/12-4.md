### DuckDB + ipywidgets: 인터랙티브 분석

```python
import duckdb
import pandas as pd
import ipywidgets as widgets
from IPython.display import display
import matplotlib.pyplot as plt
import matplotlib
matplotlib.use('module://matplotlib_inline.backend_inline')

con = duckdb.connect()
# (sales 테이블 존재 가정)

# 드롭다운으로 지역 선택
region_widget = widgets.Dropdown(
    options=['전체', 'Seoul', 'Busan', 'Incheon', 'Daegu', 'Daejeon'],
    value='전체',
    description='지역:'
)
metric_widget = widgets.RadioButtons(
    options=['매출합계', '주문수', '평균주문금액'],
    value='매출합계',
    description='지표:'
)
output = widgets.Output()

def update_chart(change):
    region = region_widget.value
    metric = metric_widget.value

    where_clause = "" if region == '전체' else f"WHERE region = '{region}'"
    metric_col = {
        '매출합계':   'SUM(amount)',
        '주문수':     'COUNT(*)',
        '평균주문금액': 'AVG(amount)'
    }[metric]

    df = con.execute(f"""
        SELECT DATE_TRUNC('month', order_date) AS month,
               {metric_col} AS value
        FROM sales
        {where_clause}
        GROUP BY month
        ORDER BY month
    """).df()

    with output:
        output.clear_output(wait=True)
        fig, ax = plt.subplots(figsize=(10, 4))
        ax.plot(df['month'].astype(str), df['value'], marker='o')
        ax.set_title(f'{region} - {metric}')
        ax.set_xlabel('월')
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()

region_widget.observe(update_chart, names='value')
metric_widget.observe(update_chart, names='value')

display(widgets.HBox([region_widget, metric_widget]), output)
update_chart(None)  # 초기 렌더링
```

### 대용량 결과 페이지네이션

```python
# 셀: 대용량 쿼리를 청크로 처리
def process_in_chunks(query: str, chunk_size: int = 50_000):
    """대용량 쿼리 결과를 청크 단위로 처리"""
    cursor = con.execute(query)
    total = 0

    while True:
        chunk = cursor.fetchmany(chunk_size)
        if not chunk:
            break
        df_chunk = pd.DataFrame(chunk,
                                columns=[d[0] for d in cursor.description])
        # 청크별 처리
        total += len(df_chunk)
        print(f"  처리: {total:,}행", end='\r')

    print(f"\n완료: {total:,}행")

process_in_chunks("SELECT * FROM sales")

# Arrow 스트리밍 (메모리 효율 최고)
reader = con.execute("SELECT * FROM sales").fetch_arrow_reader(50_000)
for i, batch in enumerate(reader):
    df = batch.to_pandas()
    # 배치 처리...
    print(f"배치 {i}: {len(df):,}행", end='\r')
```

### Notebook 환경 최적화 설정

```python
# 셀: Notebook 시작 시 실행 설정
import duckdb
import pandas as pd

# Pandas 출력 설정
pd.set_option('display.max_columns', 50)
pd.set_option('display.max_rows', 100)
pd.set_option('display.float_format', '{:,.2f}'.format)

# DuckDB 연결 및 설정
con = duckdb.connect(config={
    'threads':        8,
    'memory_limit':   '4GB',
    'enable_progress_bar': True,  # 장시간 쿼리 진행률 표시
})

# 자주 쓰는 헬퍼 함수
def q(sql):
    """SQL → DataFrame (단축 함수)"""
    return con.execute(sql).df()

def qshow(sql, n=10):
    """SQL → 상위 N행 출력"""
    return con.execute(sql).df().head(n)

def qsql(sql):
    """SQL → DuckDB Relation (메서드 체이닝용)"""
    return con.sql(sql)

# 사용 예
q("SELECT * FROM sales LIMIT 5")
qshow("SELECT region, SUM(amount) FROM sales GROUP BY region ORDER BY 2 DESC")
```

### %%duckdb 커스텀 매직 등록

```python
# 셀: 커스텀 IPython 매직 등록
from IPython.core.magic import register_cell_magic
import duckdb

_con = duckdb.connect()

@register_cell_magic
def dql(line, cell):
    """%%dql — DuckDB SQL 셀 매직"""
    var_name = line.strip() or '_dql_result'
    result = _con.execute(cell).df()
    get_ipython().push({var_name: result})
    return result

# 사용법:
# %%dql dept_stats
# SELECT dept, AVG(salary) FROM employees GROUP BY dept
#
# → dept_stats 변수에 DataFrame 저장
```
