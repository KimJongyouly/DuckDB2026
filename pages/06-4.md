# 06-4. 뷰(VIEW)와 시퀀스

### VIEW

뷰는 저장된 SELECT 쿼리에 이름을 붙인 것이다. 데이터를 복제하지 않고 논리적 테이블을 제공한다.

```sql
-- 기본 뷰 생성
CREATE VIEW active_employees AS
SELECT id, name, department, salary, hire_date
FROM employees
WHERE active = true;

-- 뷰 사용 (일반 테이블처럼)
SELECT * FROM active_employees WHERE department = 'Engineering';
SELECT COUNT(*) FROM active_employees;

-- OR REPLACE: 뷰 교체
CREATE OR REPLACE VIEW active_employees AS
SELECT id, name, department, salary, hire_date, email
FROM employees
WHERE active = true AND hire_date IS NOT NULL;

-- IF NOT EXISTS
CREATE VIEW IF NOT EXISTS active_employees AS
SELECT * FROM employees WHERE active = true;

-- 복잡한 쿼리를 뷰로 추상화
CREATE VIEW department_stats AS
SELECT
    department,
    COUNT(*)                           AS headcount,
    AVG(salary)                        AS avg_salary,
    MIN(salary)                        AS min_salary,
    MAX(salary)                        AS max_salary,
    PERCENTILE_CONT(0.5) WITHIN GROUP
        (ORDER BY salary)              AS median_salary,
    SUM(salary)                        AS total_payroll
FROM employees
WHERE active = true
GROUP BY department;

-- 뷰 삭제
DROP VIEW active_employees;
DROP VIEW IF EXISTS active_employees;
DROP VIEW active_employees CASCADE;  -- 의존 뷰 포함

-- 뷰 목록 조회
SHOW VIEWS;
SELECT * FROM duckdb_views();
```

### 뷰의 활용 패턴

```sql
-- 패턴 1: 보안 - 민감 열 숨기기
CREATE VIEW employees_public AS
SELECT id, name, department, hire_date
FROM employees;
-- salary, email 등 민감 정보 제외

-- 패턴 2: 복잡한 조인 단순화
CREATE VIEW sales_summary AS
SELECT
    o.id AS order_id,
    c.name AS customer_name,
    p.name AS product_name,
    o.quantity,
    o.quantity * p.price AS total
FROM orders o
JOIN customers c ON o.customer_id = c.id
JOIN products  p ON o.product_id  = p.id;

-- 이후 단순한 쿼리로 사용
SELECT customer_name, SUM(total) AS total_spent
FROM sales_summary
GROUP BY customer_name
ORDER BY total_spent DESC;

-- 패턴 3: 파일 기반 뷰 (외부 데이터 추상화)
CREATE VIEW raw_events AS
SELECT * FROM read_parquet('s3://data-lake/events/*.parquet');

-- 패턴 4: 체이닝 뷰
CREATE VIEW monthly_sales AS
SELECT DATE_TRUNC('month', order_date) AS month, SUM(total) AS revenue
FROM sales_summary
GROUP BY 1;

CREATE VIEW yoy_comparison AS
SELECT
    month,
    revenue,
    LAG(revenue, 12) OVER (ORDER BY month) AS prev_year,
    revenue / LAG(revenue, 12) OVER (ORDER BY month) - 1 AS yoy
FROM monthly_sales;
```

### SEQUENCE: 시퀀스

시퀀스는 순차적인 고유 정수를 생성한다. 기본키나 주문 번호 같은 자동 증가 값에 사용한다.

```sql
-- 시퀀스 생성
CREATE SEQUENCE seq_employee_id;
CREATE SEQUENCE seq_order_id START WITH 1000 INCREMENT BY 1;
CREATE SEQUENCE seq_even START 2 INCREMENT 2;
CREATE SEQUENCE seq_countdown START 100 INCREMENT -1 MINVALUE 1;

-- 옵션
CREATE SEQUENCE seq_user_id
    START WITH 1
    INCREMENT BY 1
    MINVALUE 1
    MAXVALUE 9999999
    CYCLE;               -- 최댓값 도달 시 처음으로 순환

-- 다음 값 가져오기
SELECT NEXTVAL('seq_employee_id');    -- 1, 2, 3, ...
SELECT NEXTVAL('seq_order_id');       -- 1000, 1001, ...

-- 현재 값 확인 (NEXTVAL 호출 후)
SELECT CURRVAL('seq_employee_id');

-- 시퀀스를 기본값으로 사용
CREATE TABLE employees (
    id   INTEGER DEFAULT NEXTVAL('seq_employee_id') PRIMARY KEY,
    name VARCHAR NOT NULL
);

INSERT INTO employees (name) VALUES ('Alice');  -- id 자동 할당
INSERT INTO employees (name) VALUES ('Bob');

-- 시퀀스 초기화/변경
ALTER SEQUENCE seq_employee_id RESTART WITH 1;
ALTER SEQUENCE seq_order_id INCREMENT BY 5;

-- 시퀀스 삭제
DROP SEQUENCE seq_employee_id;
DROP SEQUENCE IF EXISTS seq_employee_id;

-- 시퀀스 목록 조회
SELECT * FROM duckdb_sequences();
```

### MACRO: 재사용 가능한 함수 정의

```sql
-- 스칼라 매크로 (인라인 함수처럼 동작)
CREATE MACRO discount_price(price, pct) AS
    ROUND(price * (1 - pct / 100.0), 2);

SELECT discount_price(100.0, 15);   -- 85.0

-- 테이블 매크로 (인라인 뷰처럼 동작)
CREATE MACRO active_dept(dept_name) AS TABLE
    SELECT * FROM employees
    WHERE active = true AND department = dept_name;

SELECT * FROM active_dept('Engineering');

-- OR REPLACE
CREATE OR REPLACE MACRO discount_price(price, pct) AS
    GREATEST(0, ROUND(price * (1 - pct / 100.0), 2));

-- 매크로 삭제
DROP MACRO discount_price;
SELECT * FROM duckdb_functions() WHERE function_type = 'macro';
```
