# 20-3. 실시간 대시보드 연동

S3 레이크의 Gold 마트가 준비됐다면 마지막 단계는 시각화다. DuckDB의 빠른 집계 성능은 대시보드 백엔드로 활용하기에 적합하다. Streamlit, Grafana, Apache Superset과의 연동 패턴을 살펴본다. 이 섹션의 코드들은 모두 앞서 만든 Gold 레이어 Parquet 파일을 직접 읽는다.

### Streamlit 대시보드

Streamlit은 Python 스크립트로 인터랙티브 대시보드를 만드는 가장 빠른 방법이다.

```python
# dashboard/app.py
import streamlit as st
import duckdb
import plotly.express as px
import plotly.graph_objects as go
from datetime import date, timedelta
from pathlib import Path

LAKE_ROOT = Path('/data/lake')

# ── 연결 캐싱 (세션당 한 번만 생성)
@st.cache_resource
def get_connection():
    con = duckdb.connect(config={'threads': 4, 'memory_limit': '4GB'})
    return con

# ── 쿼리 결과 캐싱 (5분 TTL)
@st.cache_data(ttl=300)
def load_daily_revenue(start_date: str, end_date: str, region: str = 'ALL'):
    con = get_connection()
    filter_clause = f"AND region = '{region}'" if region != 'ALL' else ''
    return con.execute(f"""
        SELECT
            order_date,
            region,
            SUM(revenue)          AS revenue,
            SUM(order_count)      AS orders,
            SUM(unique_customers) AS customers
        FROM read_parquet('{LAKE_ROOT}/gold/daily_revenue.parquet')
        WHERE order_date BETWEEN '{start_date}' AND '{end_date}'
          {filter_clause}
        GROUP BY order_date, region
        ORDER BY order_date
    """).df()


@st.cache_data(ttl=300)
def load_segment_summary():
    con = get_connection()
    return con.execute(f"""
        SELECT
            segment,
            COUNT(*)              AS customer_count,
            ROUND(AVG(monetary), 0) AS avg_revenue,
            ROUND(AVG(frequency), 1) AS avg_orders,
            ROUND(AVG(recency_days), 0) AS avg_recency_days
        FROM read_parquet('{LAKE_ROOT}/gold/customer_segments.parquet')
        GROUP BY segment
        ORDER BY avg_revenue DESC
    """).df()


# ── 페이지 설정
st.set_page_config(
    page_title='Sales Dashboard',
    page_icon='📊',
    layout='wide'
)

# ── 사이드바 필터
st.sidebar.title('필터')
date_range = st.sidebar.date_input(
    '날짜 범위',
    value=(date.today() - timedelta(days=30), date.today()),
    max_value=date.today()
)
region_options = ['ALL', 'Asia', 'Europe', 'North America', 'South America']
selected_region = st.sidebar.selectbox('지역', region_options)

start_date, end_date = date_range
df = load_daily_revenue(str(start_date), str(end_date), selected_region)

# ── KPI 카드
st.title('📊 Sales Dashboard')
col1, col2, col3, col4 = st.columns(4)

with col1:
    total_revenue = df['revenue'].sum()
    st.metric('총 매출', f'${total_revenue:,.0f}')
with col2:
    total_orders = df['orders'].sum()
    st.metric('총 주문', f'{total_orders:,}')
with col3:
    total_customers = df['customers'].sum()
    st.metric('활성 고객', f'{total_customers:,}')
with col4:
    avg_order = total_revenue / total_orders if total_orders else 0
    st.metric('평균 주문액', f'${avg_order:,.1f}')

st.divider()

# ── 매출 추이 차트
col_left, col_right = st.columns([2, 1])

with col_left:
    st.subheader('일별 매출 추이')
    fig = px.line(
        df.groupby('order_date')['revenue'].sum().reset_index(),
        x='order_date', y='revenue',
        labels={'order_date': '날짜', 'revenue': '매출'},
        color_discrete_sequence=['#636EFA']
    )
    fig.update_layout(height=350)
    st.plotly_chart(fig, use_container_width=True)

with col_right:
    st.subheader('지역별 매출 비중')
    region_df = df.groupby('region')['revenue'].sum().reset_index()
    fig = px.pie(
        region_df, values='revenue', names='region',
        hole=0.4,
        color_discrete_sequence=px.colors.qualitative.Set2
    )
    fig.update_layout(height=350)
    st.plotly_chart(fig, use_container_width=True)

# ── 고객 세분화
st.subheader('고객 세분화 (RFM)')
seg_df = load_segment_summary()

fig = px.bar(
    seg_df,
    x='segment', y='customer_count',
    color='avg_revenue',
    color_continuous_scale='Blues',
    labels={
        'segment': '세그먼트',
        'customer_count': '고객 수',
        'avg_revenue': '평균 매출'
    },
    text='customer_count'
)
fig.update_traces(texttemplate='%{text:,}', textposition='outside')
fig.update_layout(height=400)
st.plotly_chart(fig, use_container_width=True)

# ── 원시 데이터 테이블
with st.expander('데이터 테이블 보기'):
    st.dataframe(
        df.sort_values('order_date', ascending=False),
        use_container_width=True
    )
```

### FastAPI + Plotly 커스텀 대시보드

더 세밀한 제어가 필요하거나 기존 웹앱에 DuckDB를 연결할 때 사용한다.

```python
# dashboard/api_server.py
from fastapi import FastAPI, Query
from fastapi.responses import HTMLResponse
import duckdb
import plotly.graph_objects as go
import plotly.express as px
import json
from datetime import date, timedelta
from pathlib import Path

app = FastAPI(title='DuckDB Analytics API')
LAKE_ROOT = Path('/data/lake')

# 앱 수명 주기 연결 관리
_con: duckdb.DuckDBPyConnection = None

@app.on_event('startup')
def startup():
    global _con
    _con = duckdb.connect(config={'threads': 4, 'memory_limit': '8GB'})

@app.on_event('shutdown')
def shutdown():
    if _con:
        _con.close()


@app.get('/api/revenue/daily')
def daily_revenue(
    start_date: date = Query(default=date.today() - timedelta(days=30)),
    end_date:   date = Query(default=date.today()),
    region:     str  = Query(default=None)
):
    """일별 매출 JSON API"""
    where = f"AND region = '{region}'" if region else ''
    df = _con.execute(f"""
        SELECT
            order_date::VARCHAR AS date,
            region,
            ROUND(SUM(revenue), 2) AS revenue,
            SUM(order_count) AS orders
        FROM read_parquet('{LAKE_ROOT}/gold/daily_revenue.parquet')
        WHERE order_date BETWEEN '{start_date}' AND '{end_date}' {where}
        GROUP BY order_date, region
        ORDER BY order_date
    """).df()
    return df.to_dict(orient='records')


@app.get('/dashboard', response_class=HTMLResponse)
def dashboard():
    """Plotly 차트가 포함된 HTML 대시보드"""
    # 최근 30일 매출 데이터
    df = _con.execute(f"""
        SELECT
            order_date,
            region,
            SUM(revenue) AS revenue
        FROM read_parquet('{LAKE_ROOT}/gold/daily_revenue.parquet')
        WHERE order_date >= (TODAY() - INTERVAL 30 DAYS)
        GROUP BY order_date, region
        ORDER BY order_date
    """).df()

    fig = px.line(
        df, x='order_date', y='revenue', color='region',
        title='최근 30일 지역별 매출 추이'
    )
    chart_html = fig.to_html(full_html=False, include_plotlyjs='cdn')

    return f"""
    <html>
      <head>
        <title>DuckDB Dashboard</title>
        <style>
          body {{ font-family: Arial, sans-serif; margin: 20px; }}
          h1 {{ color: #333; }}
        </style>
      </head>
      <body>
        <h1>📊 Sales Analytics</h1>
        {chart_html}
      </body>
    </html>
    """
```

### Grafana 연동 (JSON API 플러그인)

```python
# dashboard/grafana_api.py - Grafana JSON 데이터소스 형식 API
from fastapi import FastAPI
from pydantic import BaseModel
from typing import List, Optional
import duckdb
from datetime import datetime
from pathlib import Path

app = FastAPI()
LAKE_ROOT = Path('/data/lake')
_con = duckdb.connect(config={'threads': 4, 'memory_limit': '4GB'})


class TimeRange(BaseModel):
    from_: str
    to: str

    class Config:
        fields = {'from_': 'from'}


class QueryRequest(BaseModel):
    targets: List[dict]
    range: dict
    intervalMs: int = 60000


@app.get('/')
def health():
    return {'status': 'ok'}


@app.post('/query')
def query(req: QueryRequest):
    """Grafana Simple JSON 형식 응답"""
    start = req.range['from'][:10]
    end   = req.range['to'][:10]
    results = []

    for target in req.targets:
        metric = target.get('target', 'revenue')

        df = _con.execute(f"""
            SELECT
                order_date,
                SUM({metric}) AS value
            FROM read_parquet('{LAKE_ROOT}/gold/daily_revenue.parquet')
            WHERE order_date BETWEEN '{start}' AND '{end}'
            GROUP BY order_date
            ORDER BY order_date
        """).df()

        # Grafana timeseries 형식: [[value, timestamp_ms], ...]
        datapoints = [
            [row['value'], int(row['order_date'].timestamp() * 1000)]
            for _, row in df.iterrows()
        ]
        results.append({'target': metric, 'datapoints': datapoints})

    return results


@app.post('/search')
def search():
    """사용 가능한 메트릭 목록 반환"""
    return ['revenue', 'orders', 'unique_customers', 'avg_order_value']
```

### 자동 새로고침 파이프라인 연동

```python
# scheduler/refresh.py - 파이프라인 완료 후 대시보드 캐시 무효화
import requests
import duckdb
from datetime import date
import logging

logger = logging.getLogger(__name__)


def run_and_refresh():
    """ETL 실행 후 Streamlit 캐시 무효화"""
    from pipeline.gold import build_daily_revenue_mart, build_customer_segments

    # Gold 마트 갱신
    build_daily_revenue_mart()
    build_customer_segments()

    # Streamlit 캐시 무효화 API 호출
    try:
        resp = requests.post('http://localhost:8501/_stcore/clear_cache')
        if resp.status_code == 200:
            logger.info("Streamlit 캐시 무효화 완료")
    except Exception as e:
        logger.warning(f"캐시 무효화 실패 (무시): {e}")

    logger.info("대시보드 데이터 갱신 완료")


if __name__ == '__main__':
    run_and_refresh()
```
