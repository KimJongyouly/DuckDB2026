# 22-C. 확장 목록과 설치 명령

부록 C는 새로운 확장을 설치하거나 플랫폼별 설치 방법, .duckdbrc 초기 설정을 구성할 때 참고하세요.

### 설치 및 관리

```sql
-- 확장 설치 및 로드
INSTALL extension_name;
LOAD extension_name;

-- 한 번에 처리 (autoload 설정)
SET autoinstall_known_extensions = true;
SET autoload_known_extensions = true;

-- 설치된 확장 목록
SELECT extension_name, loaded, installed, description
FROM duckdb_extensions()
ORDER BY loaded DESC, extension_name;

-- 확장 업데이트
UPDATE EXTENSIONS;
UPDATE EXTENSIONS (extension_name);

-- 오프라인 로컬 파일 로드
LOAD '/path/to/extension.duckdb_extension';
```

### 확장별 핵심 사용법 요약

**httpfs**

```sql
INSTALL httpfs; LOAD httpfs;

-- S3 자격증명
CREATE SECRET s3 (TYPE S3, PROVIDER CREDENTIAL_CHAIN);
CREATE SECRET s3 (TYPE S3, KEY_ID '...', SECRET '...', REGION 'ap-northeast-2');

-- HTTP 직접 읽기
SELECT * FROM read_parquet('https://example.com/data.parquet');

-- S3 읽기/쓰기
SELECT * FROM read_parquet('s3://bucket/path/*.parquet', hive_partitioning=true);
COPY t TO 's3://bucket/out.parquet' (FORMAT PARQUET);
```

**spatial**

```sql
INSTALL spatial; LOAD spatial;

-- 기하 생성
ST_Point(127.0, 37.5)
ST_GeomFromText('POINT(127.0 37.5)')
ST_GeomFromGeoJSON('{"type":"Point","coordinates":[127.0,37.5]}')

-- 측정
ST_Distance(p1, p2)
ST_Distance_Sphere(p1, p2)        -- 구면 거리 (미터)
ST_Area(polygon)
ST_Length(linestring)

-- 관계
ST_Contains(polygon, point)
ST_Intersects(g1, g2)
ST_Within(g1, g2)
ST_DWithin(g1, g2, distance)

-- 변환
ST_Transform(geom, 'EPSG:4326', 'EPSG:3857')
ST_Buffer(geom, radius)
ST_Centroid(geom)
ST_Envelope(geom)

-- 파일 읽기
SELECT * FROM ST_Read('data.geojson');
SELECT * FROM ST_Read('data.shp');
SELECT * FROM ST_Read('data.gpkg');
```

**fts (전문 검색)**

```sql
INSTALL fts; LOAD fts;

-- 인덱스 생성
PRAGMA create_fts_index('articles', 'id', 'title', 'body',
    stemmer='porter', stopwords='english', overwrite=true);

-- BM25 검색
SELECT *, fts_main_articles.match_bm25(id, '검색어') AS score
FROM articles
WHERE score IS NOT NULL
ORDER BY score DESC;
```

**excel**

```sql
INSTALL excel; LOAD excel;

-- 시트 목록
SELECT * FROM excel_sheets('report.xlsx');

-- 읽기
SELECT * FROM read_xlsx('report.xlsx');
SELECT * FROM read_xlsx('report.xlsx', sheet='Sheet2', header=true);
SELECT * FROM read_xlsx('report.xlsx', range='A1:D100');
```

**postgres**

```sql
INSTALL postgres; LOAD postgres;

-- 연결
ATTACH 'postgresql://user:pass@host:5432/dbname' AS pg (TYPE POSTGRES);
ATTACH 'postgresql://user:pass@host:5432/dbname' AS pg (TYPE POSTGRES, READ_ONLY);

-- 쿼리
SELECT * FROM pg.public.orders LIMIT 10;

-- 데이터 이동
CREATE TABLE local_copy AS SELECT * FROM pg.public.orders;
INSERT INTO pg.public.summary SELECT region, SUM(amount) FROM local GROUP BY region;

DETACH pg;
```

**sqlite**

```sql
INSTALL sqlite; LOAD sqlite;

ATTACH 'app.db' AS sqlite_db (TYPE SQLITE);
SELECT * FROM sqlite_db.main.users;
DETACH sqlite_db;
```

**iceberg**

```sql
INSTALL iceberg; LOAD iceberg;

-- 로컬 / S3
SELECT * FROM iceberg_scan('/path/to/table');
SELECT * FROM iceberg_scan('s3://bucket/table', allow_moved_paths=true);

-- REST Catalog ATTACH
ATTACH 'catalog_url' AS cat (
    TYPE ICEBERG, CLIENT_ID '...', CLIENT_SECRET '...'
);
SELECT * FROM cat.namespace.table_name;

-- 시간 여행
SELECT * FROM iceberg_scan('...', snapshot_id=12345);
```

**delta**

```sql
INSTALL delta; LOAD delta;

SELECT * FROM delta_scan('/path/to/delta_table');
SELECT * FROM delta_scan('s3://bucket/delta_table');

-- 시간 여행
SELECT * FROM delta_scan('...', version=5);
SELECT * FROM delta_scan('...', timestamp='2024-06-01');

-- 이력 조회
SELECT * FROM delta_table_history('/path/to/delta_table');
```

### .duckdbrc 권장 설정

```sql
-- ~/.duckdbrc

-- 성능
SET threads = 8;
SET memory_limit = '32GB';
SET temp_directory = '/tmp/duckdb';

-- 편의
SET enable_progress_bar = true;
.timer on
.mode duckbox
.headers on

-- 확장 자동 관리
SET autoinstall_known_extensions = true;
SET autoload_known_extensions = true;

-- 타임존
SET TimeZone = 'Asia/Seoul';
```

### 플랫폼별 설치 명령 요약

```bash
# CLI 설치
# macOS
brew install duckdb

# Linux (x86_64)
curl -fsSL https://github.com/duckdb/duckdb/releases/latest/download/duckdb_cli-linux-amd64.zip \
  | gunzip > duckdb && chmod +x duckdb && mv duckdb /usr/local/bin/

# Windows (PowerShell)
winget install DuckDB.cli

# Python
pip install duckdb

# Node.js
npm install @duckdb/node-neo-api

# R
install.packages('duckdb')

# Java (Maven)
# <dependency>
#   <groupId>org.duckdb</groupId>
#   <artifactId>duckdb_jdbc</artifactId>
#   <version>최신버전</version>
# </dependency>

# Go
# go get github.com/marcboeker/go-duckdb

# Rust (Cargo.toml)
# duckdb = { version = "최신버전", features = ["bundled"] }
```
