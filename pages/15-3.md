# 15-3. AWS / Azure 연동

### AWS 시크릿 관리 (Secret Manager)

DuckDB v0.10+는 자격증명을 `SECRET` 객체로 관리한다. SET으로 전역 설정하는 것보다 안전하고 유연하다.

```sql
-- AWS 시크릿 생성
CREATE SECRET aws_prod (
    TYPE        S3,
    PROVIDER    credential_chain,   -- 환경 변수 → 프로파일 → 인스턴스 메타데이터 순서로 시도
    REGION      'ap-northeast-2'
);

-- 키 직접 지정
CREATE SECRET aws_dev (
    TYPE             S3,
    PROVIDER         config,
    KEY_ID           'AKIAIOSFODNN7EXAMPLE',
    SECRET           'wJalrXUtnFEMI/K7MDENG/bPxRfiCYEXAMPLEKEY',
    REGION           'ap-northeast-2'
);

-- 여러 시크릿, 버킷별 시크릿 분리
CREATE SECRET s3_prod (
    TYPE    S3,
    PROVIDER credential_chain,
    REGION  'ap-northeast-2',
    SCOPE   's3://prod-bucket'      -- 이 버킷에만 사용
);

CREATE SECRET s3_dev (
    TYPE    S3,
    PROVIDER config,
    KEY_ID  'dev_key',
    SECRET  'dev_secret',
    REGION  'us-east-1',
    SCOPE   's3://dev-bucket'
);

-- 시크릿 목록
SELECT * FROM duckdb_secrets();

-- 시크릿 삭제
DROP SECRET aws_dev;

-- 시크릿을 영구 저장 (파일 DB에만 적용)
CREATE PERSISTENT SECRET aws_prod (
    TYPE     S3,
    PROVIDER credential_chain,
    REGION   'ap-northeast-2'
);
```

### AWS Glue Data Catalog

```sql
LOAD httpfs;

-- Glue 카탈로그의 Iceberg 테이블 연결
ATTACH '' AS glue_catalog (
    TYPE        ICEBERG,
    ENDPOINT    'https://glue.ap-northeast-2.amazonaws.com',
    CATALOG_TYPE 'glue',
    REGION      'ap-northeast-2'
);

-- Glue 테이블 쿼리
SELECT * FROM glue_catalog.my_database.sales_table
WHERE year = 2024
LIMIT 100;

-- AWS Athena와 동일한 데이터 공유 가능
```

### Azure Blob Storage / ADLS Gen2

```sql
INSTALL azure; LOAD azure;

-- 연결 문자열로 인증
CREATE SECRET azure_storage (
    TYPE             AZURE,
    PROVIDER         config,
    CONNECTION_STRING 'DefaultEndpointsProtocol=https;AccountName=myaccount;AccountKey=mykey;EndpointSuffix=core.windows.net'
);

-- SAS 토큰
CREATE SECRET azure_sas (
    TYPE        AZURE,
    PROVIDER    config,
    ACCOUNT_NAME 'myaccount',
    SAS_TOKEN   'sv=2023-08-03&ss=b&srt=sco&sp=...'
);

-- 관리 ID (Azure VM에서)
CREATE SECRET azure_mi (
    TYPE     AZURE,
    PROVIDER managed_identity
);

-- Blob Storage 쿼리
SELECT *
FROM read_parquet('az://mycontainer/data/sales/*.parquet');

-- ADLS Gen2
SELECT *
FROM read_parquet('abfss://mycontainer@myaccount.dfs.core.windows.net/data/*.parquet');

-- Azure 결과 저장
COPY (SELECT * FROM employees)
TO 'az://mycontainer/output/employees.parquet'
(FORMAT PARQUET, COMPRESSION ZSTD);
```

### Azure Blob + Hive 파티션

```sql
SELECT *
FROM read_parquet(
    'az://datalake/warehouse/**/*.parquet',
    hive_partitioning = true
)
WHERE year = 2024 AND region = 'Seoul';
```

### Google Cloud Storage

```sql
-- GCS: S3 호환 API (HMAC 키 사용)
CREATE SECRET gcs_secret (
    TYPE             S3,
    PROVIDER         config,
    KEY_ID           '<hmac_access_key>',
    SECRET           '<hmac_secret>',
    ENDPOINT         'storage.googleapis.com',
    URL_STYLE        'vhost',
    SCOPE            'gcs://my-bucket'
);

SELECT * FROM read_parquet('s3://my-gcs-bucket/data/*.parquet');
```
