# 05-5. 수학·통계 함수

### 기본 수학

```sql
SELECT
    ABS(-42),          -- 42     절댓값
    SIGN(-3.14),       -- -1     부호 (-1, 0, 1)
    CEIL(3.2),         -- 4      올림
    FLOOR(3.9),        -- 3      내림
    ROUND(3.567, 2),   -- 3.57   반올림 (소수점 2자리)
    ROUND(3.5),        -- 4      정수 반올림
    TRUNC(3.9),        -- 3      소수점 절사 (내림과 동일)
    TRUNC(3.567, 1),   -- 3.5    소수점 1자리 이하 절사
    MOD(17, 5),        -- 2      나머지
    17 % 5,            -- 2      나머지 (연산자)
    DIV(17, 5),        -- 3      정수 몫
    GREATEST(3,7,1,5), -- 7      최댓값 (가변 인수)
    LEAST(3,7,1,5);    -- 1      최솟값 (가변 인수)
```

### 지수와 로그

```sql
SELECT
    SQRT(144),         -- 12.0   제곱근
    CBRT(27),          -- 3.0    세제곱근
    POWER(2, 10),      -- 1024   거듭제곱
    2 ** 10,           -- 1024   거듭제곱 연산자
    EXP(1),            -- 2.718  e^x
    LN(EXP(1)),        -- 1.0    자연로그
    LOG(100),          -- 2.0    log10
    LOG2(1024),        -- 10.0   log2
    LOG(2, 1024);      -- 10.0   LOG(base, value)
```

### 삼각함수

```sql
SELECT
    PI(),              -- 3.141592...
    RADIANS(180),      -- π       도 → 라디안
    DEGREES(PI()),     -- 180.0   라디안 → 도
    SIN(PI()/2),       -- 1.0
    COS(0),            -- 1.0
    TAN(PI()/4),       -- 1.0
    ASIN(1),           -- π/2
    ACOS(1),           -- 0.0
    ATAN(1),           -- π/4
    ATAN2(1, 1),       -- π/4    y, x 순서
    SINH(1),           -- 쌍곡 사인
    COSH(1),           -- 쌍곡 코사인
    TANH(1);           -- 쌍곡 탄젠트
```

### 비트 연산

```sql
SELECT
    5 & 3,             -- 1    비트 AND
    5 | 3,             -- 7    비트 OR
    5 # 3,             -- 6    비트 XOR
    ~5,                -- -6   비트 NOT
    1 << 4,            -- 16   왼쪽 시프트
    16 >> 2,           -- 4    오른쪽 시프트
    BIT_COUNT(255);    -- 8    1인 비트 수
```

### 난수

```sql
SELECT RANDOM();                     -- 0~1 균등 분포
SELECT RANDOM() * 100;               -- 0~100 실수
SELECT FLOOR(RANDOM() * 6 + 1);      -- 1~6 주사위

-- 재현 가능한 난수 (시드)
SELECT SETSEED(0.42);
SELECT RANDOM();   -- 항상 동일한 값

-- 정규 분포 난수
SELECT NORMAL_RANDOM(mean := 100, std := 15);

-- UUID 생성
SELECT GEN_RANDOM_UUID();   -- 'a8098c1a-f86e-11da-bd1a-00112444be1e'
SELECT UUID();              -- 동일
```

### 수치 분석 함수

```sql
-- 수치 범위 클리핑
SELECT GREATEST(0, LEAST(100, value)) AS clipped  -- 0~100 사이로 제한
FROM measurements;

-- 선형 보간 (정규화)
SELECT (value - MIN(value) OVER ()) /
       (MAX(value) OVER () - MIN(value) OVER ()) AS normalized
FROM data;

-- 누적 곱
SELECT EXP(SUM(LN(ABS(value)))) AS product
FROM positive_values;
```
