# 07-2. 손상된 CSV 처리

실무 CSV 파일은 완벽하지 않다. 열이 누락되거나, 타입이 섞이거나, 인코딩이 깨지는 경우가 흔하다.

### 오류 허용 모드

```sql
-- ignore_errors: 파싱 오류가 있는 행을 건너뜀
SELECT * FROM read_csv(
    'dirty.csv',
    ignore_errors = true
);

-- store_rejects: 오류 행을 별도 테이블에 저장
SELECT * FROM read_csv(
    'dirty.csv',
    store_rejects  = true,
    rejects_table  = 'csv_errors',     -- 오류 저장 테이블
    rejects_scan   = 'csv_error_scan'  -- 오류 스캔 이름
);

-- 오류 내용 확인
SELECT * FROM csv_errors;
-- line  | column | error     | value  | file
-- ──────┼────────┼───────────┼────────┼──────────
-- 42    | amount | cast error | 'N/A'  | dirty.csv
-- 87    | date   | date error | '32/13'| dirty.csv
```

### 열 수 불일치 처리

```sql
-- 열 수가 행마다 다른 경우
SELECT * FROM read_csv(
    'ragged.csv',
    header           = true,
    null_padding     = true,   -- 부족한 열을 NULL로 채움
    ignore_errors    = true    -- 열이 너무 많은 행은 건너뜀
);

-- 예: 헤더 5열, 데이터 행이 3열인 경우
-- id | name | age | city | salary
-- 1  | Alice| 30  | NULL | NULL   ← null_padding으로 채움
```

### NULL 값 패턴 처리

```sql
-- 다양한 NULL 표현 일괄 처리
SELECT * FROM read_csv(
    'data.csv',
    nullstr = ['', 'NULL', 'null', 'NA', 'N/A', 'n/a',
               'None', 'none', '-', '--', '?', 'missing']
);

-- 빈 문자열과 NULL을 다르게 처리하고 싶을 때
-- → all_varchar=true 로 읽고 수동으로 구분
SELECT
    id,
    NULLIF(TRIM(name), '') AS name,   -- 공백 문자열 → NULL
    NULLIF(value, 'NA')   AS value
FROM read_csv('data.csv', all_varchar=true);
```

### 타입 혼합 열 처리

```sql
-- 열에 숫자와 문자가 섞인 경우
-- 예: amount 열에 "100.5", "N/A", "$200", "" 혼재

SELECT
    id,
    -- 방법 1: TRY_CAST로 안전 변환
    TRY_CAST(
        REPLACE(REPLACE(amount, '$', ''), ',', '')
        AS DECIMAL(10,2)
    ) AS amount_clean,

    -- 방법 2: 정규식으로 숫자만 추출
    TRY_CAST(
        REGEXP_EXTRACT(amount, '\d+\.?\d*')
        AS DOUBLE
    ) AS amount_num

FROM read_csv('mixed.csv', all_varchar=true, header=true);
```

### 날짜 형식 혼재

```sql
-- 같은 열에 여러 날짜 형식이 섞인 경우
SELECT
    id,
    COALESCE(
        TRY_STRPTIME(date_str, '%Y-%m-%d'),       -- 2024-03-15
        TRY_STRPTIME(date_str, '%d/%m/%Y'),       -- 15/03/2024
        TRY_STRPTIME(date_str, '%m-%d-%Y'),       -- 03-15-2024
        TRY_STRPTIME(date_str, '%d.%m.%Y'),       -- 15.03.2024
        TRY_STRPTIME(date_str, '%B %d, %Y')       -- March 15, 2024
    ) AS parsed_date,
    CASE
        WHEN TRY_STRPTIME(date_str, '%Y-%m-%d') IS NOT NULL THEN 'ISO'
        WHEN TRY_STRPTIME(date_str, '%d/%m/%Y') IS NOT NULL THEN 'EU'
        WHEN TRY_STRPTIME(date_str, '%m-%d-%Y') IS NOT NULL THEN 'US'
        ELSE 'unknown'
    END AS date_format
FROM read_csv('dates.csv', all_varchar=true);
```

### 인용 문자 이슈

```sql
-- 필드 안에 구분자가 포함된 경우 (정상 CSV)
-- "Seoul, Korea","Alice","100,000"
SELECT * FROM read_csv('quoted.csv', quote='"');

-- 인용 문자가 없는 경우
SELECT * FROM read_csv('noquote.csv', quote='');

-- 인용 문자 이스케이프 방식이 다른 경우
-- 표준: "She said ""hello"""
-- 백슬래시: "She said \"hello\""
SELECT * FROM read_csv('escaped.csv', escape='\');

-- 인용 문자가 일관성 없이 망가진 경우
SELECT * FROM read_csv(
    'broken_quotes.csv',
    ignore_errors = true,
    store_rejects = true,
    rejects_table = 'bad_rows'
);
```

### 손상 진단 워크플로우

```sql
-- Step 1: 전체를 VARCHAR로 읽어서 행 수 확인
SELECT COUNT(*) FROM read_csv('dirty.csv', all_varchar=true, ignore_errors=true);

-- Step 2: 열별 null/이상값 비율 확인
SELECT
    COUNT(*) AS total,
    COUNT(id) AS id_not_null,
    COUNT(CASE WHEN TRY_CAST(amount AS DOUBLE) IS NULL
               AND amount IS NOT NULL THEN 1 END) AS amount_bad,
    COUNT(CASE WHEN TRY_CAST(order_date AS DATE) IS NULL
               AND order_date IS NOT NULL THEN 1 END) AS date_bad
FROM read_csv('dirty.csv', all_varchar=true);

-- Step 3: 오류 행 샘플 확인
SELECT *
FROM read_csv('dirty.csv', all_varchar=true)
WHERE TRY_CAST(amount AS DOUBLE) IS NULL
  AND amount IS NOT NULL
LIMIT 20;

-- Step 4: 정제 쿼리 작성 후 테이블 생성
CREATE TABLE orders_clean AS
SELECT
    id::INTEGER AS id,
    TRY_CAST(order_date AS DATE)            AS order_date,
    TRY_CAST(REPLACE(amount,'$','') AS DECIMAL(10,2)) AS amount,
    NULLIF(TRIM(status), '')                AS status
FROM read_csv('dirty.csv', all_varchar=true)
WHERE TRY_CAST(id AS INTEGER) IS NOT NULL;  -- 필수 열이 유효한 행만
```
