### 기본 사용 (Neo API)

```javascript
import duckdb from '@duckdb/node-neo-api';

// 인스턴스 생성
const db = new duckdb.Database(':memory:');            // 인메모리
const db2 = new duckdb.Database('mydb.duckdb');        // 영구 파일
const db3 = new duckdb.Database('mydb.duckdb',
    duckdb.OPEN_READONLY);                             // 읽기 전용

// 연결 생성
const conn = await db.connect();

// DDL
await conn.run(`
    CREATE TABLE employees (
        id     INTEGER PRIMARY KEY,
        name   VARCHAR,
        dept   VARCHAR,
        salary DOUBLE
    )
`);

// INSERT
await conn.run(`
    INSERT INTO employees VALUES
        (1, 'Alice', 'Engineering', 95000),
        (2, 'Bob',   'Sales',       72000)
`);

// SELECT → 전체 결과
const rows = await conn.all('SELECT * FROM employees ORDER BY id');
console.log(rows);
// [{ id: 1, name: 'Alice', dept: 'Engineering', salary: 95000 }, ...]

// SELECT → 첫 번째 행
const row = await conn.get('SELECT * FROM employees WHERE id = ?', 1);
console.log(row);  // { id: 1, name: 'Alice', ... }

// 데이터베이스 종료
await db.close();
```

### 파라미터화와 스트리밍

```javascript
import duckdb from '@duckdb/node-neo-api';

const db = new duckdb.Database(':memory:');
const conn = await db.connect();

// 파라미터화 쿼리 (? 플레이스홀더)
const results = await conn.all(
    'SELECT * FROM employees WHERE dept = ? AND salary > ?',
    'Engineering',
    80000
);

// 스트리밍 (대용량 결과 — 메모리 효율)
const stmt = await conn.prepare(
    'SELECT * FROM read_parquet(?) WHERE amount > ?'
);

const stream = await stmt.stream('sales/*.parquet', 10000);

for await (const chunk of stream) {
    // chunk: rows 배열
    for (const row of chunk) {
        process(row);
    }
}

// Arrow 결과
const arrowResult = await conn.arrowIPCStream(
    'SELECT * FROM employees'
);
```

### Promise 기반 배치 INSERT

```javascript
import duckdb from '@duckdb/node-neo-api';

const db = new duckdb.Database('analytics.duckdb');
const conn = await db.connect();

await conn.run('BEGIN');

try {
    const stmt = await conn.prepare(
        'INSERT INTO events (ts, user_id, event_type, amount) VALUES (?, ?, ?, ?)'
    );

    const events = [
        ['2024-03-15 14:30:00', 1001, 'purchase', 99.9],
        ['2024-03-15 14:31:00', 1002, 'view',     null],
        ['2024-03-15 14:32:00', 1001, 'purchase', 149.0],
    ];

    for (const [ts, uid, type, amt] of events) {
        await stmt.run(ts, uid, type, amt);
    }

    await conn.run('COMMIT');
    console.log(`${events.length}행 삽입 완료`);

} catch (err) {
    await conn.run('ROLLBACK');
    throw err;
}
```

### Express.js API 서버 패턴

```javascript
import express from 'express';
import duckdb from '@duckdb/node-neo-api';

const app = express();
const db = new duckdb.Database('analytics.duckdb', duckdb.OPEN_READONLY);

// 부서별 통계 API
app.get('/api/departments', async (req, res) => {
    try {
        const conn = await db.connect();
        const rows = await conn.all(`
            SELECT
                dept,
                COUNT(*)    AS headcount,
                AVG(salary) AS avg_salary
            FROM employees
            GROUP BY dept
            ORDER BY headcount DESC
        `);
        await conn.close();
        res.json(rows);
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
});

// 파라미터화 검색 API
app.get('/api/employees', async (req, res) => {
    const { dept, minSalary = 0 } = req.query;
    const conn = await db.connect();

    let sql = 'SELECT * FROM employees WHERE salary >= ?';
    const params = [Number(minSalary)];

    if (dept) {
        sql += ' AND dept = ?';
        params.push(dept);
    }

    const rows = await conn.all(sql, ...params);
    await conn.close();
    res.json(rows);
});

app.listen(3000, () => console.log('서버 시작: http://localhost:3000'));
```
