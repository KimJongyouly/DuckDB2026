### 변수 주입 패턴

CLI는 SQL 변수를 직접 지원하지 않아, 셸 변수나 문자열 치환으로 주입한다.

```bash
#!/usr/bin/env bash
# etl_daily.sh

TARGET_DATE="${1:-$(date -d 'yesterday' +%Y-%m-%d)}"
OUTPUT_DIR="output/${TARGET_DATE}"
DB="pipeline.duckdb"

mkdir -p "$OUTPUT_DIR"

echo "처리 날짜: $TARGET_DATE"

# heredoc에 변수 삽입 (작은따옴표 없어야 함)
duckdb "$DB" << EOF
    -- 날짜 파라미터 주입
    CREATE OR REPLACE TEMP TABLE daily_data AS
    SELECT *
    FROM read_parquet('raw/date=*/data.parquet', hive_partitioning=true)
    WHERE date = DATE '${TARGET_DATE}';

    COPY daily_data TO '${OUTPUT_DIR}/summary.parquet'
    (FORMAT PARQUET, COMPRESSION ZSTD);

    SELECT
        '${TARGET_DATE}' AS processed_date,
        COUNT(*) AS rows_processed
    FROM daily_data;
EOF

echo "완료: $OUTPUT_DIR/summary.parquet"
```

### 다단계 파이프라인 스크립트

```bash
#!/usr/bin/env bash
# pipeline.sh — 전체 ETL 파이프라인

set -euo pipefail

DB="pipeline.duckdb"
LOG="logs/pipeline_$(date +%Y%m%d_%H%M%S).log"
mkdir -p logs reports output

log() { echo "[$(date +%H:%M:%S)] $*" | tee -a "$LOG"; }

run_sql() {
    local step="$1"
    local sql="$2"
    log "STEP: $step 시작"
    if duckdb "$DB" -bail -c "$sql" >> "$LOG" 2>&1; then
        log "STEP: $step 완료"
    else
        log "STEP: $step 실패"
        exit 1
    fi
}

# ── Step 1: 원본 적재 ──────────────────────────────────────────
run_sql "원본 적재" "
    CREATE OR REPLACE TABLE raw_sales AS
    SELECT * FROM read_csv('data/raw/*.csv',
        header=true,
        ignore_errors=true,
        union_by_name=true
    );
    SELECT '적재 완료: ' || COUNT(*) || '행' AS msg FROM raw_sales;
"

# ── Step 2: 데이터 정제 ────────────────────────────────────────
run_sql "데이터 정제" "
    CREATE OR REPLACE TABLE clean_sales AS
    SELECT
        id::INTEGER                           AS id,
        TRY_CAST(order_date AS DATE)          AS order_date,
        LOWER(TRIM(region))                   AS region,
        TRY_CAST(amount AS DECIMAL(12,2))     AS amount,
        customer_id::INTEGER                  AS customer_id
    FROM raw_sales
    WHERE TRY_CAST(id AS INTEGER) IS NOT NULL
      AND TRY_CAST(order_date AS DATE) IS NOT NULL
      AND TRY_CAST(amount AS DECIMAL(12,2)) > 0;

    SELECT
        (SELECT COUNT(*) FROM raw_sales) AS raw_rows,
        COUNT(*) AS clean_rows,
        ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM raw_sales), 1)
            AS clean_pct
    FROM clean_sales;
"

# ── Step 3: 집계 ───────────────────────────────────────────────
run_sql "집계" "
    CREATE OR REPLACE TABLE monthly_summary AS
    SELECT
        DATE_TRUNC('month', order_date) AS month,
        region,
        COUNT(*)                        AS order_count,
        COUNT(DISTINCT customer_id)     AS unique_customers,
        SUM(amount)                     AS revenue,
        AVG(amount)                     AS avg_order
    FROM clean_sales
    GROUP BY ALL;
"

# ── Step 4: 리포트 출력 ────────────────────────────────────────
log "리포트 생성 중..."

duckdb "$DB" << 'EOF' >> "$LOG" 2>&1
    .mode csv
    .headers on
    .output reports/monthly_summary.csv
    SELECT * FROM monthly_summary ORDER BY month, region;
    .output stdout

    COPY monthly_summary TO 'output/monthly_summary.parquet'
    (FORMAT PARQUET, COMPRESSION ZSTD);
EOF

log "파이프라인 완료"
log "리포트: reports/monthly_summary.csv"
log "Parquet: output/monthly_summary.parquet"
```

### 데이터 품질 검증 스크립트

```bash
#!/usr/bin/env bash
# validate.sh — 데이터 품질 게이트

DB="analytics.duckdb"
FAILED=0

check() {
    local name="$1"
    local sql="$2"
    local expected="$3"

    result=$(duckdb -csv -noheader "$DB" -c "$sql")

    if [ "$result" = "$expected" ]; then
        echo "  ✓ PASS: $name"
    else
        echo "  ✗ FAIL: $name (예상: $expected, 실제: $result)"
        FAILED=$((FAILED + 1))
    fi
}

echo "=== 데이터 품질 검증 ==="

# 행 수 검증
check "최소 행 수" \
    "SELECT (COUNT(*) >= 10000)::INTEGER FROM clean_sales" \
    "1"

# NULL 검증
check "order_date NULL 없음" \
    "SELECT (COUNT(*) FILTER (WHERE order_date IS NULL) = 0)::INTEGER FROM clean_sales" \
    "1"

# 범위 검증
check "amount 양수" \
    "SELECT (MIN(amount) > 0)::INTEGER FROM clean_sales" \
    "1"

# 날짜 범위 검증
check "날짜 범위 정상" \
    "SELECT (MAX(order_date) <= CURRENT_DATE)::INTEGER FROM clean_sales" \
    "1"

# 중복 검증
check "ID 중복 없음" \
    "SELECT (COUNT(*) = COUNT(DISTINCT id))::INTEGER FROM clean_sales" \
    "1"

# 참조 무결성
check "고객 참조 정상" \
    "SELECT (COUNT(*) FILTER (WHERE customer_id NOT IN (SELECT id FROM customers)) = 0)::INTEGER FROM clean_sales" \
    "1"

echo ""
if [ "$FAILED" -eq 0 ]; then
    echo "모든 검증 통과 ✓"
    exit 0
else
    echo "${FAILED}개 검증 실패 ✗"
    exit 1
fi
```

### CSV/Parquet 변환 유틸리티

```bash
#!/usr/bin/env bash
# convert.sh — 포맷 변환 유틸리티

usage() {
    echo "사용법: $0 <입력파일> <출력파일>"
    echo "  예)  $0 data.csv data.parquet"
    echo "       $0 data.parquet output.csv"
    echo "       $0 'data/*.csv' merged.parquet"
    exit 1
}

[ $# -lt 2 ] && usage

INPUT="$1"
OUTPUT="$2"
EXT_OUT="${OUTPUT##*.}"

echo "변환: $INPUT → $OUTPUT"

case "$EXT_OUT" in
    parquet)
        duckdb -c "
            COPY (SELECT * FROM read_csv_auto('${INPUT}'))
            TO '${OUTPUT}' (FORMAT PARQUET, COMPRESSION ZSTD);
        "
        ;;
    csv)
        duckdb -c "
            COPY (SELECT * FROM read_parquet('${INPUT}'))
            TO '${OUTPUT}' (FORMAT CSV, HEADER true);
        "
        ;;
    json)
        duckdb -c "
            COPY (SELECT * FROM read_parquet('${INPUT}'))
            TO '${OUTPUT}' (FORMAT JSON, ARRAY true);
        "
        ;;
    *)
        echo "지원하지 않는 출력 형식: $EXT_OUT"
        exit 1
        ;;
esac

# 결과 크기 출력
SIZE=$(du -sh "$OUTPUT" | cut -f1)
echo "완료: $OUTPUT ($SIZE)"
```

### Make를 활용한 파이프라인

```makefile
# Makefile — DuckDB 파이프라인 관리

DB        := analytics.duckdb
DATA_DIR  := data
OUTPUT    := output
REPORTS   := reports

.PHONY: all clean validate load transform export

all: validate export

# 원본 데이터 적재
load: $(DB)
	@echo "원본 데이터 적재..."
	duckdb $(DB) -bail -f sql/01_load.sql

# 데이터 변환
transform: load
	@echo "데이터 변환..."
	duckdb $(DB) -bail -f sql/02_transform.sql

# 데이터 품질 검증
validate: transform
	@echo "품질 검증..."
	bash scripts/validate.sh

# 결과 내보내기
export: validate
	@mkdir -p $(REPORTS) $(OUTPUT)
	duckdb $(DB) -bail -f sql/03_export.sql

# 특정 날짜 재처리
reprocess:
	@[ -n "$(DATE)" ] || (echo "DATE= 필요. 예: make reprocess DATE=2024-01-15"; exit 1)
	duckdb $(DB) -bail -c "
		DELETE FROM clean_sales WHERE order_date = DATE '$(DATE)';
	"
	DATE=$(DATE) bash scripts/etl_daily.sh $(DATE)

# DB 초기화
$(DB):
	duckdb $(DB) -bail -f sql/00_schema.sql

# 정리
clean:
	rm -f $(OUTPUT)/*.parquet $(REPORTS)/*.csv

# 전체 초기화
reset: clean
	rm -f $(DB)
```

```bash
# 사용
make all                           # 전체 파이프라인
make load                          # 적재만
make validate                      # 검증만
make reprocess DATE=2024-01-15     # 특정 날짜 재처리
make reset                         # 초기화
```

### Python에서 CLI 호출

```python
import subprocess
import shlex

def run_duckdb(sql: str, db: str = ':memory:', fmt: str = 'csv') -> str:
    """DuckDB CLI를 subprocess로 실행"""
    cmd = ['duckdb', f'-{fmt}', '-noheader', '-bail', db, '-c', sql]
    result = subprocess.run(
        cmd,
        capture_output=True,
        text=True,
        check=True
    )
    return result.stdout.strip()

# 사용
count = run_duckdb("SELECT COUNT(*) FROM 'data.parquet'")
print(f"행 수: {count}")

# SQL 파일 실행
def run_sql_file(sql_file: str, db: str) -> int:
    """SQL 파일 실행, 종료 코드 반환"""
    result = subprocess.run(
        ['duckdb', db, '-bail', '-f', sql_file],
        capture_output=True,
        text=True
    )
    if result.returncode != 0:
        print(f"오류:\n{result.stderr}")
    return result.returncode

# CI에서 검증
import sys
rc = run_sql_file('tests/data_quality.sql', 'analytics.duckdb')
sys.exit(rc)
```

### CI/CD에서 DuckDB 활용

```yaml
# .github/workflows/data_quality.yml
name: 데이터 품질 검증

on:
  push:
    paths:
      - 'data/**'
      - 'sql/**'

jobs:
  validate:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: DuckDB 설치
        run: |
          curl -fsSL https://install.duckdb.org | sh
          echo "$HOME/.duckdb/cli" >> $GITHUB_PATH

      - name: 스키마 초기화
        run: duckdb test.duckdb -bail -f sql/schema.sql

      - name: 샘플 데이터 적재
        run: |
          duckdb test.duckdb -bail -c "
            COPY employees FROM 'data/sample_employees.csv' (HEADER true);
            COPY sales FROM 'data/sample_sales.csv' (HEADER true);
          "

      - name: 데이터 품질 검증
        run: duckdb test.duckdb -bail -f tests/quality_checks.sql

      - name: 쿼리 성능 검증
        run: |
          duckdb test.duckdb -bail -stats -c "
            SELECT region, SUM(amount)
            FROM sales
            GROUP BY region;
          "

      - name: 정리
        if: always()
        run: rm -f test.duckdb
```

```sql
-- tests/quality_checks.sql

-- 행 수 체크
SELECT CASE
    WHEN COUNT(*) >= 100 THEN 'PASS'
    ELSE ERROR('행 수 부족: ' || COUNT(*))
END AS row_count_check
FROM employees;

-- NULL 체크
SELECT CASE
    WHEN COUNT(*) FILTER (WHERE name IS NULL) = 0 THEN 'PASS'
    ELSE ERROR('name에 NULL 존재')
END AS null_check
FROM employees;

-- 범위 체크
SELECT CASE
    WHEN MIN(salary) > 0 AND MAX(salary) < 100000000 THEN 'PASS'
    ELSE ERROR('salary 범위 이상: min=' || MIN(salary) || ' max=' || MAX(salary))
END AS range_check
FROM employees;
```

### 유용한 원라이너 모음

```bash
# 파일 빠른 탐색
duckdb -c "DESCRIBE SELECT * FROM 'data.parquet'"
duckdb -c "SELECT COUNT(*) FROM 'data/*.parquet'"
duckdb -c "SELECT * FROM 'data.csv' LIMIT 5"

# 파일 크기별 행 수
duckdb -csv -c "
    SELECT filename, COUNT(*) AS rows
    FROM read_parquet('data/*.parquet', filename=true)
    GROUP BY filename ORDER BY rows DESC
"

# CSV → Parquet 원라이너
duckdb -c "COPY (SELECT * FROM read_csv_auto('input.csv')) TO 'output.parquet' (FORMAT PARQUET, COMPRESSION ZSTD)"

# Parquet 스키마 확인
duckdb -c "SELECT * FROM parquet_schema('data.parquet')"

# 두 파일 비교 (행 수 차이)
duckdb -csv -c "
    SELECT 'before' AS label, COUNT(*) FROM 'before.parquet'
    UNION ALL
    SELECT 'after', COUNT(*) FROM 'after.parquet'
"

# 디렉터리 내 모든 Parquet 요약
duckdb -c "
    SELECT
        REGEXP_EXTRACT(file_name, '[^/]+$') AS file,
        SUM(row_group_num_rows) AS rows,
        ROUND(SUM(total_compressed_size)/1024/1024.0, 1) AS mb
    FROM parquet_metadata('data/*.parquet')
    GROUP BY file_name
    ORDER BY rows DESC
"

# 중복 행 탐지
duckdb -c "
    SELECT *, COUNT(*) AS cnt
    FROM 'data.parquet'
    GROUP BY ALL HAVING COUNT(*) > 1
    LIMIT 10
"

# 열별 NULL 비율
duckdb -c "
    SELECT column_name,
           null_percentage
    FROM (SUMMARIZE SELECT * FROM 'data.parquet')
    WHERE null_percentage > 0
    ORDER BY null_percentage DESC
"

# SUMMARIZE: 전체 통계 한 번에
duckdb -c "SUMMARIZE SELECT * FROM 'data.parquet'"
```
