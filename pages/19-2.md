# 19-2. 로깅과 모니터링

DuckDB는 내장 로그 시스템이 없으므로, 애플리케이션 레이어에서 실행 쿼리, 소요 시간, 메모리 사용량을 직접 계측해야 한다.

### 쿼리 로깅 미들웨어

```python
import duckdb
import time
import logging
import functools
from typing import Any

# 로거 설정
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s %(levelname)s [%(name)s] %(message)s',
    datefmt='%Y-%m-%dT%H:%M:%S'
)
logger = logging.getLogger('duckdb')


class LoggedConnection:
    """쿼리 실행을 자동으로 로깅하는 DuckDB 연결 래퍼"""

    def __init__(self, *args, **kwargs):
        self._con = duckdb.connect(*args, **kwargs)
        self._query_count = 0
        self._total_time = 0.0

    def execute(self, query: str, params=None):
        self._query_count += 1
        short_query = ' '.join(query.split())[:120]
        start = time.perf_counter()

        try:
            result = self._con.execute(query, params or [])
            elapsed = time.perf_counter() - start
            self._total_time += elapsed

            # 느린 쿼리 경고 (100ms 초과)
            if elapsed > 0.1:
                logger.warning(
                    f"SLOW_QUERY elapsed={elapsed*1000:.1f}ms query={short_query!r}"
                )
            else:
                logger.debug(
                    f"QUERY elapsed={elapsed*1000:.1f}ms query={short_query!r}"
                )
            return result

        except Exception as e:
            elapsed = time.perf_counter() - start
            logger.error(
                f"QUERY_ERROR elapsed={elapsed*1000:.1f}ms "
                f"error={type(e).__name__}: {e} query={short_query!r}"
            )
            raise

    def stats(self):
        return {
            'query_count': self._query_count,
            'total_time_ms': round(self._total_time * 1000, 1),
            'avg_time_ms': round(
                self._total_time / self._query_count * 1000, 1
            ) if self._query_count else 0,
        }

    def __getattr__(self, name):
        return getattr(self._con, name)

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self._con.close()


# 사용 예시
with LoggedConnection('/data/analytics.db', config={'threads': 4}) as con:
    con.execute("SELECT COUNT(*) FROM sales").fetchone()
    con.execute("SELECT region, SUM(amount) FROM sales GROUP BY region").fetchall()
    print(con.stats())
# {'query_count': 2, 'total_time_ms': 42.3, 'avg_time_ms': 21.1}
```

### 쿼리 실행 기록 테이블

```python
# 쿼리 실행 이력을 DuckDB 자체에 저장
class AuditedConnection:
    """쿼리 실행 감사 로그를 별도 DB에 저장"""

    def __init__(self, main_db: str, audit_db: str = ':memory:'):
        self._con = duckdb.connect(main_db)
        self._audit = duckdb.connect(audit_db)
        self._audit.execute("""
            CREATE TABLE IF NOT EXISTS query_log (
                id          INTEGER PRIMARY KEY,
                executed_at TIMESTAMPTZ DEFAULT now(),
                query       VARCHAR,
                params      VARCHAR,
                elapsed_ms  DOUBLE,
                rows_affected INTEGER,
                success     BOOLEAN,
                error_msg   VARCHAR
            )
        """)
        self._seq = 0

    def execute(self, query: str, params=None):
        self._seq += 1
        start = time.perf_counter()
        error_msg = None
        rows = None

        try:
            result = self._con.execute(query, params or [])
            elapsed = (time.perf_counter() - start) * 1000
            try:
                rows = result.fetchall()
                rows_affected = len(rows)
                # 결과를 다시 담을 수 있도록 재실행 (단순 로깅 목적)
            except Exception:
                rows_affected = 0
            success = True
        except Exception as e:
            elapsed = (time.perf_counter() - start) * 1000
            error_msg = str(e)
            rows_affected = 0
            success = False
            raise
        finally:
            self._audit.execute("""
                INSERT INTO query_log
                    (id, query, params, elapsed_ms, rows_affected, success, error_msg)
                VALUES (?, ?, ?, ?, ?, ?, ?)
            """, [
                self._seq,
                query[:2000],
                str(params)[:500] if params else None,
                round(elapsed, 2),
                rows_affected,
                success,
                error_msg,
            ])

        return self._con.execute(query, params or [])

    def slow_queries(self, threshold_ms: float = 100):
        """느린 쿼리 목록 반환"""
        return self._audit.execute("""
            SELECT id, executed_at, elapsed_ms, query
            FROM query_log
            WHERE elapsed_ms > ?
            ORDER BY elapsed_ms DESC
        """, [threshold_ms]).fetchdf()
```

### Prometheus 메트릭 노출

```python
# metrics.py - Prometheus 메트릭 연동
from prometheus_client import Counter, Histogram, Gauge, start_http_server
import duckdb
import time
import threading

# 메트릭 정의
QUERY_COUNT = Counter(
    'duckdb_queries_total',
    'Total number of DuckDB queries',
    ['status']   # success / error
)
QUERY_DURATION = Histogram(
    'duckdb_query_duration_seconds',
    'Query execution duration',
    buckets=[.001, .005, .01, .05, .1, .5, 1, 5, 10]
)
MEMORY_USAGE = Gauge(
    'duckdb_memory_usage_bytes',
    'Current DuckDB memory usage'
)
ACTIVE_CONNECTIONS = Gauge(
    'duckdb_active_connections',
    'Number of active DuckDB connections'
)


class MonitoredConnection:
    _instance_count = 0
    _lock = threading.Lock()

    def __init__(self, *args, **kwargs):
        self._con = duckdb.connect(*args, **kwargs)
        with MonitoredConnection._lock:
            MonitoredConnection._instance_count += 1
        ACTIVE_CONNECTIONS.set(MonitoredConnection._instance_count)

    def execute(self, query: str, params=None):
        with QUERY_DURATION.time():
            try:
                result = self._con.execute(query, params or [])
                QUERY_COUNT.labels(status='success').inc()
                return result
            except Exception:
                QUERY_COUNT.labels(status='error').inc()
                raise

    def update_memory_metric(self):
        try:
            row = self._con.execute(
                "SELECT memory_usage FROM duckdb_memory()"
            ).fetchone()
            if row:
                MEMORY_USAGE.set(row[0])
        except Exception:
            pass

    def close(self):
        self._con.close()
        with MonitoredConnection._lock:
            MonitoredConnection._instance_count -= 1
        ACTIVE_CONNECTIONS.set(MonitoredConnection._instance_count)


# 메모리 메트릭 주기적 갱신 (백그라운드)
def start_metrics_collector(con: MonitoredConnection, interval: int = 15):
    def collect():
        while True:
            con.update_memory_metric()
            time.sleep(interval)
    t = threading.Thread(target=collect, daemon=True)
    t.start()


if __name__ == '__main__':
    start_http_server(9090)   # Prometheus 스크레이핑 엔드포인트
    con = MonitoredConnection('/data/analytics.db')
    start_metrics_collector(con)
    print("Metrics server running on :9090")
```

# 메모리 경고 모니터 (임계값 초과 시 알림)
import threading
import time

def memory_watchdog(con, threshold_pct=80, interval=10):
    """메모리 사용률이 임계값을 넘으면 경고 로그"""
    def watch():
        while True:
            try:
                row = con.execute("""
                    SELECT
                        round(memory_usage * 100.0 / memory_limit, 1) AS pct,
                        memory_usage,
                        memory_limit
                    FROM duckdb_memory()
                """).fetchone()
                if row and row[0] > threshold_pct:
                    logger.warning(
                        f"MEMORY_HIGH usage={row[0]}% "
                        f"({row[1]/1e9:.1f}GB / {row[2]/1e9:.1f}GB)"
                    )
            except Exception:
                pass
            time.sleep(interval)

    t = threading.Thread(target=watch, daemon=True)
    t.start()
