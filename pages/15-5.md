# 15-5. Full-Text Search

`fts` 확장은 DuckDB에 전문 검색(Full-Text Search) 기능을 추가한다. 역인덱스(Inverted Index)를 생성해 키워드 검색을 빠르게 처리한다.

### 설치와 인덱스 생성

```sql
INSTALL fts; LOAD fts;

-- 테이블 생성
CREATE TABLE articles (
    id      INTEGER PRIMARY KEY,
    title   VARCHAR,
    body    VARCHAR,
    author  VARCHAR,
    tags    VARCHAR
);

INSERT INTO articles VALUES
    (1, 'DuckDB 입문', 'DuckDB는 분석용 인메모리 데이터베이스입니다. SQL을 사용합니다.', '김철수', 'database sql'),
    (2, 'Parquet 완전 정복', 'Parquet는 열 지향 파일 형식입니다. 압축률이 높습니다.', '이영희', 'file format'),
    (3, 'Python 데이터 분석', 'Pandas와 DuckDB를 함께 사용하면 강력합니다.', '박민준', 'python pandas'),
    (4, '데이터 레이크 설계', 'Iceberg와 Delta Lake는 레이크하우스 포맷입니다.', '최지은', 'lakehouse'),
    (5, 'SQL 최적화', 'SQL 쿼리 튜닝은 인덱스와 실행계획 분석에서 시작합니다.', '홍길동', 'sql optimization');

-- FTS 인덱스 생성
-- PRAGMA create_fts_index(테이블, PK열, 검색열들...)
PRAGMA create_fts_index('articles', 'id', 'title', 'body', 'author', 'tags');
```

### 전문 검색 쿼리

```sql
-- 기본 검색: FTS 인덱스 활용
SELECT id, title, score
FROM (
    SELECT *, fts_main_articles.match_bm25(id, 'DuckDB') AS score
    FROM articles
) ranked
WHERE score IS NOT NULL
ORDER BY score DESC;
-- id | title              | score
-- ───┼────────────────────┼──────
-- 1  | DuckDB 입문        | 0.89
-- 3  | Python 데이터 분석 | 0.42

-- 여러 단어 검색 (AND)
SELECT id, title,
    fts_main_articles.match_bm25(id, 'DuckDB SQL') AS score
FROM articles
WHERE score IS NOT NULL
ORDER BY score DESC;

-- 함수 이름 규칙: fts_main_<테이블명>.match_bm25(<pk>, <쿼리>)
```

### 검색 옵션

```sql
-- 필드 가중치 조정 (title이 body보다 중요)
SELECT id, title,
    fts_main_articles.match_bm25(
        id,
        'SQL 분석',
        fields   := 'title,body',
        k        := 1.2,          -- 빈도 포화 파라미터 (기본 1.2)
        b        := 0.75,         -- 문서 길이 정규화 (기본 0.75)
        conjunctive := true       -- AND 검색 (모든 단어 포함)
    ) AS score
FROM articles
WHERE score IS NOT NULL
ORDER BY score DESC;

-- 단순 포함 검색 (BM25 점수 없이)
SELECT * FROM articles
WHERE title ILIKE '%DuckDB%'
   OR body  ILIKE '%DuckDB%';

-- FTS + 필터 결합
SELECT id, title, score
FROM (
    SELECT *, fts_main_articles.match_bm25(id, 'SQL') AS score
    FROM articles
) t
WHERE score IS NOT NULL
  AND author = '김철수'   -- 추가 필터
ORDER BY score DESC;
```

### 인덱스 관리

```sql
-- 인덱스 재생성 (데이터 변경 후)
PRAGMA drop_fts_index('articles');
PRAGMA create_fts_index('articles', 'id', 'title', 'body', 'author', 'tags');

-- 스테머(어간 추출) 언어 설정
PRAGMA create_fts_index(
    'articles', 'id', 'title', 'body',
    stemmer  := 'korean',   -- 'english', 'none' 등
    stopwords := 'english', -- 불용어 처리
    lower    := true,       -- 소문자 변환
    strip_accents := true,  -- 악센트 제거
    overwrite := true       -- 기존 인덱스 덮어쓰기
);

-- 인덱스 정보 확인
SELECT * FROM fts_main_articles.docs;
SELECT * FROM fts_main_articles.terms LIMIT 20;
```

### 실전: 검색 API 패턴

```python
import duckdb

con = duckdb.connect('articles.duckdb')
con.execute("LOAD fts")

def search_articles(query: str, top_k: int = 10):
    """BM25 기반 문서 검색"""
    return con.execute(f"""
        SELECT
            id,
            title,
            author,
            ROUND(score, 4) AS relevance,
            -- 하이라이팅 (매칭 위치 표시)
            CASE
                WHEN body ILIKE '%{query}%'
                THEN SUBSTRING(body,
                     GREATEST(1, STRPOS(LOWER(body), LOWER('{query}')) - 50),
                     150) || '...'
                ELSE LEFT(body, 100) || '...'
            END AS snippet
        FROM (
            SELECT *,
                fts_main_articles.match_bm25(id, '{query}') AS score
            FROM articles
        ) t
        WHERE score IS NOT NULL
        ORDER BY score DESC
        LIMIT {top_k}
    """).df()

results = search_articles('DuckDB 분석')
print(results)
```
