# 03-7. 집합 연산

집합 연산은 두 SELECT 결과를 합치거나 교차시키거나 빼는 연산이다. 열 수와 타입이 호환되어야 한다.

```
집합 연산 시각화
────────────────────────────────────────
A = {1, 2, 3, 4}
B = {3, 4, 5, 6}

UNION ALL     = {1,2,3,4,3,4,5,6}  (모두, 중복 포함)
UNION         = {1,2,3,4,5,6}      (모두, 중복 제거)
INTERSECT     = {3,4}              (공통 요소)
INTERSECT ALL = {3,4}              (공통, 중복 보존)
EXCEPT        = {1,2}              (A에만 있는 것)
EXCEPT ALL    = {1,2}              (A에만, 중복 보존)
```

### UNION

```sql
-- UNION: 중복 제거 (느림)
SELECT id, name FROM employees_2023
UNION
SELECT id, name FROM employees_2024;

-- UNION ALL: 중복 포함 (빠름, 대부분의 경우 권장)
SELECT 'Q1' AS quarter, revenue FROM sales_q1
UNION ALL
SELECT 'Q2', revenue FROM sales_q2
UNION ALL
SELECT 'Q3', revenue FROM sales_q3
UNION ALL
SELECT 'Q4', revenue FROM sales_q4;
```

### INTERSECT

```sql
-- 두 쿼리에 모두 존재하는 행
SELECT product_id FROM orders_2023
INTERSECT
SELECT product_id FROM orders_2024;
-- 2023년과 2024년 모두 주문된 상품

-- INTERSECT ALL: 중복 횟수까지 고려
SELECT category FROM popular_items_region_A
INTERSECT ALL
SELECT category FROM popular_items_region_B;
```

### EXCEPT (MINUS)

```sql
-- 첫 번째 결과에만 있는 행
SELECT customer_id FROM all_customers
EXCEPT
SELECT customer_id FROM repeat_customers;
-- 한 번만 구매한 고객

-- 순서 중요: A EXCEPT B ≠ B EXCEPT A
SELECT product_id FROM catalog
EXCEPT
SELECT product_id FROM discontinued;
-- 단종되지 않은 상품 목록
```

### 집합 연산 결합과 주의사항

```sql
-- 여러 집합 연산 결합 (괄호로 우선순위 명시 권장)
(SELECT id FROM table_a
 UNION ALL
 SELECT id FROM table_b)
EXCEPT
SELECT id FROM table_c;

-- ORDER BY와 LIMIT은 전체 결과에 적용
SELECT name FROM employees_kr
UNION ALL
SELECT name FROM employees_us
ORDER BY name
LIMIT 20;

-- 열 이름은 첫 번째 SELECT 기준
SELECT dept_id AS id, dept_name AS name FROM departments
UNION ALL
SELECT emp_id, emp_name FROM employees;
-- 결과 열 이름: id, name (첫 번째 SELECT 기준)
```

### 실전 활용 패턴

```sql
-- 패턴 1: 전년 대비 신규/이탈 고객 분석
WITH
new_customers AS (
    SELECT customer_id FROM purchases_2024
    EXCEPT
    SELECT customer_id FROM purchases_2023
),
churned_customers AS (
    SELECT customer_id FROM purchases_2023
    EXCEPT
    SELECT customer_id FROM purchases_2024
),
retained_customers AS (
    SELECT customer_id FROM purchases_2023
    INTERSECT
    SELECT customer_id FROM purchases_2024
)
SELECT '신규' AS type, COUNT(*) AS cnt FROM new_customers
UNION ALL
SELECT '이탈',          COUNT(*)        FROM churned_customers
UNION ALL
SELECT '유지',          COUNT(*)        FROM retained_customers;

-- 패턴 2: 여러 지역 데이터 통합 후 집계
WITH all_sales AS (
    SELECT 'Seoul'  AS region, * FROM sales_seoul
    UNION ALL
    SELECT 'Busan',            * FROM sales_busan
    UNION ALL
    SELECT 'Incheon',          * FROM sales_incheon
)
SELECT region, SUM(amount) AS total
FROM all_sales
GROUP BY region
ORDER BY total DESC;
```
