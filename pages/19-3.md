# 19-3. 보안 고려사항

배포를 마쳤다면 이제 보안을 강화할 차례다. DuckDB는 파일 시스템과 네트워크에 직접 접근할 수 있는 강력한 기능을 갖추고 있다. 이 유연함은 개발 편의를 높이지만, 신뢰할 수 없는 입력을 처리하거나 공유 환경에서 운영할 때는 동시에 공격 표면이 된다. 아래 위협들은 실제로 발생한 사례를 바탕으로 한 것이므로, 각 대응책을 꼼꼼히 적용하기 바란다.

### SQL 인젝션 방지

SQL 인젝션은 가장 오래되고 가장 흔한 공격 방식이다. 예를 들어, 지역별 매출을 조회하는 API에 사용자가 `region` 파라미터로 `Asia'; DROP TABLE sales; --` 를 넣으면 어떻게 될까? 문자열 포매팅으로 쿼리를 만드는 코드에서는 `sales` 테이블 전체가 삭제될 수 있다. 파라미터 바인딩은 이런 공격을 원천 차단한다.

```python
import duckdb

con = duckdb.connect()

# 취약한 코드: 문자열 포매팅으로 쿼리 구성
region = "Asia'; DROP TABLE sales; --"

# 절대 금지
bad_query = f"SELECT * FROM sales WHERE region = '{region}'"
# → SQL 인젝션 발생 가능

# 안전한 방법 1: 파라미터 바인딩 (?)
safe_result = con.execute(
    "SELECT * FROM sales WHERE region = ?",
    [region]
).fetchall()

# 안전한 방법 2: named 파라미터 ($name)
safe_result = con.execute(
    "SELECT * FROM sales WHERE region = $region AND year = $year",
    {'region': region, 'year': 2024}
).fetchall()

# 동적 컬럼/테이블명은 화이트리스트로 검증
ALLOWED_COLUMNS = {'region', 'year', 'amount', 'status'}
ALLOWED_TABLES = {'sales', 'customers', 'orders'}

def safe_group_by(table: str, column: str):
    if table not in ALLOWED_TABLES:
        raise ValueError(f"허용되지 않은 테이블: {table}")
    if column not in ALLOWED_COLUMNS:
        raise ValueError(f"허용되지 않은 컬럼: {column}")
    # 검증된 이름만 포매팅에 사용
    return con.execute(
        f"SELECT {column}, SUM(amount) FROM {table} GROUP BY {column}"
    ).fetchall()
```

### 파일 시스템 접근 제한

DuckDB는 쿼리 안에서 파일 경로를 직접 사용할 수 있다는 점이 편리하지만, 사용자 입력이 파일 경로로 이어지면 경로 탈출(path traversal) 공격에 노출된다. 예를 들어, 사용자가 파일명으로 `../../etc/passwd`를 입력하면 시스템 계정 정보가 담긴 파일을 읽어올 수 있다. `realpath`로 경로를 정규화한 뒤 허용된 디렉터리 안에 있는지 반드시 확인해야 한다.

```python
import os
import re

# DuckDB는 파일 경로를 쿼리 내에서 직접 사용할 수 있어
# 사용자 입력이 파일 경로로 이어지면 경로 탈출 공격에 취약

BASE_DATA_DIR = '/data/allowed'

def safe_read_parquet(filename: str):
    # 경로 구성 요소 검증
    if not re.match(r'^[a-zA-Z0-9_\-]+\.parquet$', filename):
        raise ValueError(f"유효하지 않은 파일명: {filename}")

    full_path = os.path.realpath(os.path.join(BASE_DATA_DIR, filename))

    # 경로 탈출 방지 (realpath로 정규화 후 확인)
    if not full_path.startswith(BASE_DATA_DIR):
        raise PermissionError(f"접근 거부: {filename}")

    return con.execute(
        "SELECT * FROM read_parquet(?)",
        [full_path]
    ).fetchdf()
```

### 외부 접근 비활성화

```sql
-- 외부 파일/네트워크 접근 완전 차단 (샌드박스 환경)
SET enable_external_access = false;

-- HTTP/S3 접근 차단 (httpfs 확장 없이도 적용)
-- enable_external_access = false 설정 후에는
-- read_parquet('s3://...'), read_csv('http://...') 모두 실패

-- 설정 잠금 (이후 변경 불가)
SET lock_configuration = true;
```

```python
# 샌드박스 연결 생성
sandbox_con = duckdb.connect(config={
    'enable_external_access': False,
    'threads': 2,
    'memory_limit': '1GB',
})

# 이 연결에서는 로컬 파일만 접근 가능
sandbox_con.execute("SELECT 42")           # 정상
sandbox_con.execute("SELECT * FROM read_parquet('s3://bucket/file.parquet')")
# → Error: External access is disabled
```

### 자격증명 보호

```python
# 나쁜 예: 코드에 자격증명 하드코딩
con.execute("""
    CREATE SECRET s3 (
        TYPE S3,
        KEY_ID 'AKIAXXXXXXXXXXXXXXXX',
        SECRET 'wJalrXUtnFEMI/K7MDENG/bPxRfiCYEXAMPLEKEY'
    )
""")

# 좋은 예 1: 환경 변수
import os
key_id = os.environ['AWS_ACCESS_KEY_ID']
secret = os.environ['AWS_SECRET_ACCESS_KEY']
con.execute("CREATE SECRET s3 (TYPE S3, KEY_ID ?, SECRET ?)", [key_id, secret])

# 좋은 예 2: IAM 역할 (EC2/ECS/Lambda - 가장 안전)
con.execute("CREATE SECRET s3 (TYPE S3, PROVIDER CREDENTIAL_CHAIN)")

# 좋은 예 3: AWS Secrets Manager에서 동적 로드
import boto3, json

def get_db_credentials(secret_name: str) -> dict:
    client = boto3.client('secretsmanager', region_name='ap-northeast-2')
    response = client.get_secret_value(SecretId=secret_name)
    return json.loads(response['SecretString'])

creds = get_db_credentials('prod/duckdb/s3-credentials')
con.execute("""
    CREATE OR REPLACE SECRET s3 (
        TYPE S3, KEY_ID ?, SECRET ?, REGION ?
    )
""", [creds['key_id'], creds['secret'], creds['region']])
```

### Parquet 암호화

```sql
-- 민감한 데이터 컬럼 암호화
COPY (SELECT * FROM customers)
TO 'customers_encrypted.parquet' (
    FORMAT PARQUET,
    ENCRYPTION_CONFIG {
        footer_key: 'footer_key_name',
        column_keys: {
            ssn_key: ['ssn', 'credit_card'],
            pii_key: ['email', 'phone', 'address']
        }
    }
);

-- 복호화 읽기 (키 제공 필요)
SELECT *
FROM read_parquet('customers_encrypted.parquet',
    encryption_config => {
        footer_key: 'footer_key_name',
        column_keys: {
            ssn_key: ['ssn', 'credit_card'],
            pii_key: ['email', 'phone', 'address']
        }
    }
);
```
