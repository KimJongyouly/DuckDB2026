### ATTACH로 MySQL 연결

```sql
-- 기본 연결
ATTACH 'host=localhost user=root password=secret database=mydb'
AS mysql_db (TYPE MYSQL);

-- TCP 포트 지정
ATTACH 'host=mysql-server port=3306 user=analyst password=pass database=sales'
AS mysql_db (TYPE MYSQL);

-- URL 형식
ATTACH 'mysql://analyst:pass@localhost:3306/sales'
AS mysql_db (TYPE MYSQL);

-- 읽기 전용
ATTACH 'mysql://user:pass@prod-mysql/production'
AS mysql_prod (TYPE MYSQL, READ_ONLY);

-- 연결 확인
SHOW DATABASES;
SHOW ALL TABLES IN mysql_db;
```

### 테이블 탐색과 쿼리

```sql
-- 테이블 구조
DESCRIBE mysql_db.sales;

-- 기본 쿼리
SELECT * FROM mysql_db.orders LIMIT 10;

SELECT
    YEAR(order_date) AS year,
    MONTH(order_date) AS month,
    SUM(total_price) AS monthly_revenue
FROM mysql_db.orders
GROUP BY 1, 2
ORDER BY 1, 2;

-- MySQL + DuckDB 로컬 테이블 조인
SELECT
    m.product_id,
    m.product_name,
    l.category,
    SUM(m.sales_count) AS total_sales
FROM mysql_db.product_sales m
JOIN local_product_categories l ON m.product_id = l.id
GROUP BY 1, 2, 3
ORDER BY total_sales DESC;

-- 필터 푸시다운
SELECT order_id, amount
FROM mysql_db.orders
WHERE status = 'completed'
  AND order_date BETWEEN '2024-01-01' AND '2024-12-31';
```

### MySQL → DuckDB 마이그레이션

```sql
-- 전체 테이블 마이그레이션
CREATE TABLE orders_local AS
SELECT * FROM mysql_db.orders;

-- 타입 명시 마이그레이션 (MySQL TINYINT(1) → BOOLEAN 등 타입 불일치 해소)
CREATE TABLE customers_local AS
SELECT
    id,
    name,
    email,
    is_active::BOOLEAN AS is_active,   -- TINYINT(1) → BOOLEAN
    created_at::TIMESTAMP AS created_at
FROM mysql_db.customers;

-- 여러 테이블 일괄 마이그레이션
CREATE TABLE products_local AS SELECT * FROM mysql_db.products;
CREATE TABLE categories_local AS SELECT * FROM mysql_db.categories;
CREATE TABLE inventory_local AS SELECT * FROM mysql_db.inventory;

-- 마이그레이션 검증
SELECT
    'mysql' AS source,
    COUNT(*) AS rows
FROM mysql_db.orders
UNION ALL
SELECT 'duckdb', COUNT(*) FROM orders_local;
```

### 증분 동기화 패턴

```python
import duckdb
from datetime import datetime, timedelta

con = duckdb.connect('analytics.duckdb')
con.execute("LOAD mysql")
con.execute("""
    ATTACH 'mysql://etl_user:pass@mysql-prod/ecommerce'
    AS mysql_prod (TYPE MYSQL, READ_ONLY)
""")

def sync_incremental(table: str, ts_col: str):
    """변경된 데이터만 DuckDB로 동기화"""
    # 마지막 동기화 시점 조회
    last_sync = con.execute(f"""
        SELECT COALESCE(MAX({ts_col}), '1970-01-01'::TIMESTAMP)
        FROM {table}_local
    """).fetchone()[0]

    # 새 데이터만 가져오기
    new_rows = con.execute(f"""
        SELECT COUNT(*) FROM mysql_prod.{table}
        WHERE {ts_col} > '{last_sync}'
    """).fetchone()[0]

    if new_rows == 0:
        print(f"  {table}: 새 데이터 없음")
        return

    con.execute(f"""
        INSERT INTO {table}_local
        SELECT * FROM mysql_prod.{table}
        WHERE {ts_col} > '{last_sync}'
    """)
    print(f"  {table}: {new_rows:,}행 동기화 완료")

sync_incremental('orders', 'updated_at')
sync_incremental('customers', 'updated_at')

con.execute("DETACH mysql_prod")
```

### 연결 풀 관리 (Python)

```python
import duckdb
from contextlib import contextmanager

class DBConnector:
    """멀티 DB 연결 관리자"""

    def __init__(self, duckdb_path: str = ':memory:'):
        self.con = duckdb.connect(duckdb_path)
        self.con.execute("LOAD postgres; LOAD mysql; LOAD sqlite;")
        self._attached = set()

    def attach_postgres(self, name: str, dsn: str, read_only: bool = True):
        mode = "READ_ONLY" if read_only else ""
        self.con.execute(f"""
            ATTACH '{dsn}' AS {name}
            (TYPE POSTGRES{', ' + mode if mode else ''})
        """)
        self._attached.add(name)

    def attach_mysql(self, name: str, dsn: str, read_only: bool = True):
        mode = "READ_ONLY" if read_only else ""
        self.con.execute(f"""
            ATTACH '{dsn}' AS {name}
            (TYPE MYSQL{', ' + mode if mode else ''})
        """)
        self._attached.add(name)

    def attach_sqlite(self, name: str, path: str, read_only: bool = True):
        mode = "READ_ONLY" if read_only else ""
        self.con.execute(f"""
            ATTACH '{path}' AS {name}
            (TYPE SQLITE{', ' + mode if mode else ''})
        """)
        self._attached.add(name)

    def detach_all(self):
        for name in list(self._attached):
            try:
                self.con.execute(f"DETACH {name}")
            except Exception:
                pass
        self._attached.clear()

    def query(self, sql: str):
        return self.con.execute(sql).df()

    def close(self):
        self.detach_all()
        self.con.close()

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self.close()


# 사용 예
with DBConnector('analytics.duckdb') as db:
    db.attach_postgres('pg', 'postgresql://user:pass@host/db')
    db.attach_mysql('mysql', 'mysql://user:pass@host/db')
    db.attach_sqlite('cache', 'local_cache.sqlite')

    result = db.query("""
        SELECT
            p.customer_id,
            p.name,
            COUNT(m.order_id) AS orders
        FROM pg.customers p
        LEFT JOIN mysql.orders m ON p.customer_id = m.customer_id
        GROUP BY p.customer_id, p.name
        ORDER BY orders DESC
        LIMIT 10
    """)
    print(result)
```
