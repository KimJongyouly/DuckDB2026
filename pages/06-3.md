# 06-3. 인덱스와 제약조건

### DuckDB의 인덱스 철학

DuckDB는 OLAP 데이터베이스다. 전통적인 OLTP 데이터베이스처럼 B-Tree 인덱스를 폭넓게 활용하지 않는다.

```
┌────────────────────────────────────────────────────────────┐
│               DuckDB 인덱스 특성                            │
├────────────────────────────────────────────────────────────┤
│  기본 스캔 방식: 열 지향 스캔 + 존 맵 (Zone Map)            │
│  인덱스 효과: 포인트 조회나 범위 조회에서 성능 개선 가능     │
│  대용량 분석: 풀 스캔이 오히려 더 효율적인 경우 많음         │
│  ART 인덱스: 기본키·고유 제약에 자동 생성                   │
└────────────────────────────────────────────────────────────┘
```

### 인덱스 생성과 삭제

```sql
-- 단일 열 인덱스
CREATE INDEX idx_emp_dept ON employees (department);

-- 복합 인덱스
CREATE INDEX idx_emp_dept_salary ON employees (department, salary DESC);

-- 고유 인덱스
CREATE UNIQUE INDEX idx_emp_email ON employees (email);

-- 조건부 인덱스 (부분 인덱스)
CREATE INDEX idx_active_emp ON employees (hire_date)
WHERE active = true;

-- IF NOT EXISTS
CREATE INDEX IF NOT EXISTS idx_emp_dept ON employees (department);

-- 인덱스 삭제
DROP INDEX idx_emp_dept;
DROP INDEX IF EXISTS idx_emp_dept;

-- 인덱스 목록 조회
SELECT * FROM duckdb_indexes();
-- 또는
PRAGMA show_indexes('employees');
```

### 제약조건

```sql
-- PRIMARY KEY
CREATE TABLE products (
    id   INTEGER PRIMARY KEY,
    name VARCHAR NOT NULL
);

-- 복합 기본키
CREATE TABLE order_items (
    order_id   INTEGER,
    product_id INTEGER,
    quantity   INTEGER,
    PRIMARY KEY (order_id, product_id)
);

-- UNIQUE
CREATE TABLE users (
    id    INTEGER PRIMARY KEY,
    email VARCHAR UNIQUE,
    phone VARCHAR,
    UNIQUE (email, phone)   -- 복합 고유 제약
);

-- CHECK
CREATE TABLE employees (
    id     INTEGER PRIMARY KEY,
    salary DECIMAL CHECK (salary >= 0),
    grade  INTEGER CHECK (grade BETWEEN 1 AND 10),
    CONSTRAINT chk_reasonable_salary CHECK (salary < 10000000)
);

-- FOREIGN KEY
CREATE TABLE orders (
    id          INTEGER PRIMARY KEY,
    customer_id INTEGER REFERENCES customers(id),
    product_id  INTEGER,
    FOREIGN KEY (product_id) REFERENCES products(id)
        ON UPDATE CASCADE
        ON DELETE SET NULL
);

-- NOT NULL
CREATE TABLE articles (
    id      INTEGER PRIMARY KEY,
    title   VARCHAR NOT NULL,
    content VARCHAR NOT NULL,
    author  VARCHAR
);
```

### 외래키 액션

```sql
-- ON DELETE / ON UPDATE 옵션
FOREIGN KEY (dept_id) REFERENCES departments(id)
    ON DELETE CASCADE      -- 부모 삭제 시 자식도 삭제
    ON DELETE SET NULL     -- 부모 삭제 시 NULL로 설정
    ON DELETE SET DEFAULT  -- 부모 삭제 시 기본값으로 설정
    ON DELETE RESTRICT     -- 자식이 있으면 부모 삭제 금지
    ON DELETE NO ACTION    -- 제약 검사를 트랜잭션 끝으로 미룸

-- 외래키 검사 비활성화 (대량 로드 시 성능)
SET enable_foreign_keys = false;
-- 대량 INSERT ...
SET enable_foreign_keys = true;
```

### 제약조건 조회

```sql
-- 테이블 제약조건 확인
SELECT * FROM duckdb_constraints()
WHERE table_name = 'employees';

-- 컬럼 정보 조회
DESCRIBE employees;
PRAGMA table_info('employees');
SELECT * FROM information_schema.columns
WHERE table_name = 'employees';
```
