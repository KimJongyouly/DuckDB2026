# 05-4. JSON 함수

DuckDB는 JSON을 일급 타입으로 지원한다. 외부 파서 없이 SQL 안에서 JSON을 파싱·생성·탐색할 수 있다.

### JSON 읽기와 타입

```sql
-- JSON 타입
SELECT '{"name": "Alice", "age": 30}'::JSON;

-- JSON 파일 읽기
SELECT * FROM read_json('data.json');
SELECT * FROM read_json_auto('data.json');  -- 스키마 자동 감지

-- JSON 배열 파일
SELECT * FROM read_json('logs.json', format='array');
-- newline-delimited JSON (NDJSON)
SELECT * FROM read_json('events.ndjson', format='newline_delimited');
```

### 필드 접근

```sql
-- -> : JSON 반환 (체이닝 가능)
-- ->> : VARCHAR 반환 (최종 추출 시)
SELECT
    data->'name'           AS name_json,   -- JSON 타입
    data->>'name'          AS name_str,    -- VARCHAR 타입
    data->'address'->>'city' AS city,      -- 중첩 접근
    data->2                AS third_item   -- 배열 인덱스 (0-based)
FROM (
    SELECT '{"name":"Alice","address":{"city":"Seoul"},"scores":[9,8,10]}'::JSON AS data
);

-- JSON_EXTRACT: 명시적 추출
SELECT JSON_EXTRACT(data, '$.name');          -- JSON 타입
SELECT JSON_EXTRACT_STRING(data, '$.name');   -- VARCHAR 타입

-- JSONPath 표현식
SELECT JSON_EXTRACT(data, '$.address.city');
SELECT JSON_EXTRACT(data, '$.scores[0]');       -- 배열 첫 번째 (0-based)
SELECT JSON_EXTRACT(data, '$.scores[*]');       -- 배열 모든 원소
SELECT JSON_EXTRACT(data, '$..name');           -- 재귀 탐색
```

### JSON 생성

```sql
-- 객체 생성
SELECT JSON_OBJECT('name', 'Alice', 'age', 30);
-- {"name":"Alice","age":30}

-- 배열 생성
SELECT JSON_ARRAY(1, 'two', true, NULL);
-- [1,"two",true,null]

-- 행을 JSON으로 변환
SELECT ROW_TO_JSON({'name': 'Alice', 'age': 30});
-- {"name":"Alice","age":30}

-- 집계로 JSON 배열 생성
SELECT JSON_GROUP_ARRAY(name ORDER BY salary DESC)
FROM employees
GROUP BY department;

-- 집계로 JSON 객체 생성
SELECT JSON_GROUP_OBJECT(name, salary)
FROM employees
GROUP BY department;
```

### JSON 변환 및 검사

```sql
-- JSON → 테이블 (unnest)
SELECT * FROM JSON_EACH('{"a":1,"b":2,"c":3}');
-- key | value
-- ────┼──────
-- a   | 1
-- b   | 2
-- c   | 3

SELECT * FROM JSON_ARRAY_ELEMENTS('[1,2,3,4,5]');
-- value
-- ─────
-- 1
-- 2
-- ...

-- JSON 타입 확인
SELECT JSON_TYPE('{"key":"val"}');   -- 'OBJECT'
SELECT JSON_TYPE('[1,2,3]');         -- 'ARRAY'
SELECT JSON_TYPE('"hello"');         -- 'VARCHAR'
SELECT JSON_TYPE('42');              -- 'UBIGINT'
SELECT JSON_TYPE('true');            -- 'BOOLEAN'
SELECT JSON_TYPE('null');            -- 'NULL'

-- JSON 유효성 검사
SELECT JSON_VALID('{"valid": true}');     -- true
SELECT JSON_VALID('{invalid json}');      -- false

-- JSON 키 목록
SELECT JSON_KEYS('{"a":1,"b":2}');       -- ['a', 'b']

-- JSON 구조 확인
SELECT JSON_STRUCTURE('{"name":"Alice","age":30}');
-- {"name":"VARCHAR","age":"UBIGINT"}
```

### JSON → 관계형 변환

```sql
-- JSON 열을 여러 열로 펼치기
CREATE TABLE raw_events (
    id   INTEGER,
    data JSON
);
INSERT INTO raw_events VALUES
    (1, '{"user":"alice","action":"login","ts":"2024-03-15"}'),
    (2, '{"user":"bob","action":"purchase","amount":99.9}');

SELECT
    id,
    data->>'user'    AS user_name,
    data->>'action'  AS action,
    TRY_CAST(data->>'amount' AS DECIMAL(10,2)) AS amount
FROM raw_events;

-- UNNEST + JSON_ARRAY_ELEMENTS
SELECT id, elem->>'name' AS tag
FROM articles,
     JSON_ARRAY_ELEMENTS(tags_json) AS elem;
```
