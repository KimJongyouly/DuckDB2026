# 18-3. 대용량 데이터 처리 전략

앞서 벡터화 실행과 메모리 설정으로 쿼리 자체를 빠르게 만드는 법을 익혔다. 이번에는 데이터 규모 자체가 크다는 근본적인 문제를 다룬다. 수억 행, 수백 GB 규모의 데이터를 다룰 때는 단순히 쿼리를 최적화하는 것 이상의 전략이 필요하다. 파일 포맷 선택, 파티셔닝 설계, 배치 처리 분할, 그리고 스필 관리를 함께 고려해야 한다.

### 파일 포맷 선택 전략

```
┌──────────────┬───────────┬───────────┬──────────────────────────┐
│ 포맷          │ 읽기 속도  │ 쓰기 속도  │ 최적 사용 시나리오        │
├──────────────┼───────────┼───────────┼──────────────────────────┤
│ Parquet      │ ★★★★★    │ ★★★★☆    │ 분석 쿼리, 컬럼 선택      │
│ CSV          │ ★★★☆☆    │ ★★★★★    │ 데이터 교환, 단순 적재    │
│ JSON(NDJSON) │ ★★☆☆☆    │ ★★★☆☆    │ 반정형 데이터             │
│ Delta Lake   │ ★★★★☆    │ ★★★☆☆    │ ACID 쓰기, 시간 여행      │
│ Iceberg      │ ★★★★☆    │ ★★★☆☆    │ 대형 카탈로그, 멀티엔진   │
└──────────────┴───────────┴───────────┴──────────────────────────┘
```

```sql
-- Parquet 최적 쓰기 설정
COPY (SELECT * FROM large_table)
TO 'output.parquet' (
    FORMAT PARQUET,
    CODEC 'ZSTD',        -- 압축률과 속도의 균형
    ROW_GROUP_SIZE 100000,  -- 행 그룹 크기 (기본 122880)
    COMPRESSION_LEVEL 3  -- ZSTD 압축 레벨 (1=빠름, 22=고압축)
);

-- 압축 코덱별 특성
-- SNAPPY: 빠른 압축/해제, 보통 압축률 (기본값)
-- ZSTD:   좋은 압축률, 적당한 속도 (추천)
-- GZIP:   높은 압축률, 느린 속도
-- LZ4:    매우 빠른 압축/해제, 낮은 압축률
```

### 파티셔닝으로 I/O 최소화

파일 포맷을 Parquet으로 정했다면, 이제 그 파일들을 어떻게 배치할지가 중요해진다. 파티셔닝은 데이터를 논리적인 폴더 구조로 나눠 저장하는 방법이다. 예를 들어 연도별·월별로 파티션을 나누면, 특정 달의 데이터를 조회할 때 다른 달의 파일은 아예 열지 않아 I/O가 획기적으로 줄어든다.

```sql
-- 날짜 기반 파티셔닝 (가장 일반적인 패턴)
COPY (
    SELECT *, year(order_date) AS year, month(order_date) AS month
    FROM orders
)
TO '/data/orders_partitioned' (
    FORMAT PARQUET,
    PARTITION_BY (year, month),
    OVERWRITE_OR_IGNORE true
);

-- 디렉터리 구조:
-- /data/orders_partitioned/
--   year=2024/
--     month=1/data_0.parquet
--     month=2/data_0.parquet
--   year=2023/
--     ...

-- 파티션 프루닝: 2024년 1월 데이터만 읽음 (나머지 파일 스킵)
SELECT SUM(amount)
FROM read_parquet('/data/orders_partitioned/**/*.parquet', hive_partitioning=true)
WHERE year = 2024 AND month = 1;
```

```python
# 파티션 프루닝 효과 측정
import duckdb
import time

con = duckdb.connect()
con.execute("SET enable_progress_bar = false")

# 전체 스캔 시간
start = time.time()
con.execute("""
    SELECT SUM(amount)
    FROM read_parquet('/data/orders_partitioned/**/*.parquet', hive_partitioning=true)
""").fetchone()
full_scan = time.time() - start

# 파티션 프루닝 적용 시간
start = time.time()
con.execute("""
    SELECT SUM(amount)
    FROM read_parquet('/data/orders_partitioned/**/*.parquet', hive_partitioning=true)
    WHERE year = 2024 AND month = 1
""").fetchone()
pruned_scan = time.time() - start

print(f"전체 스캔: {full_scan:.2f}s")
print(f"파티션 프루닝: {pruned_scan:.2f}s")
print(f"속도 향상: {full_scan / pruned_scan:.1f}x")
```

### 배치 처리로 메모리 절약

파티셔닝으로 읽어야 할 데이터를 줄였다면, 이번에는 결과를 출력할 때 메모리를 어떻게 아낄지를 다룬다. 메모리보다 큰 데이터셋을 처리할 때는 배치로 나눠 처리한다.

```python
import duckdb

con = duckdb.connect()
con.execute("SET memory_limit = '4GB'")

# 방법 1: fetchmany() 스트리밍
result = con.execute("""
    SELECT customer_id, SUM(amount) AS total
    FROM read_parquet('s3://bucket/sales/*.parquet')
    GROUP BY customer_id
""")

output_rows = 0
with open('output.csv', 'w') as f:
    f.write("customer_id,total\n")
    while True:
        batch = result.fetchmany(10_000)
        if not batch:
            break
        for row in batch:
            f.write(f"{row[0]},{row[1]}\n")
        output_rows += len(batch)

print(f"처리 완료: {output_rows:,}행")
```

```python
# 방법 2: fetch_arrow_reader() Arrow 스트림 (더 효율적)
import pyarrow.parquet as pq

reader = con.execute("""
    SELECT *
    FROM read_parquet('s3://bucket/large_table/*.parquet')
    WHERE status = 'active'
""").fetch_arrow_reader(rows_per_batch=50_000)

writer = None
for batch in reader:
    if writer is None:
        writer = pq.ParquetWriter('output.parquet', batch.schema)
    writer.write_table(batch.to_pyarrow_table() if hasattr(batch, 'to_pyarrow_table') else batch)

if writer:
    writer.close()
```

```python
# 방법 3: 날짜 범위로 배치 분할
from datetime import date, timedelta

def process_date_range(con, start_date, end_date, output_dir):
    """날짜 범위별 배치 처리"""
    current = start_date
    while current <= end_date:
        next_date = current + timedelta(days=1)
        output_path = f"{output_dir}/{current.strftime('%Y-%m-%d')}.parquet"

        con.execute(f"""
            COPY (
                SELECT *
                FROM read_parquet('s3://bucket/events/**/*.parquet', hive_partitioning=true)
                WHERE event_date = '{current}'
            ) TO '{output_path}' (FORMAT PARQUET, CODEC 'ZSTD')
        """)

        print(f"완료: {current}")
        current = next_date

process_date_range(
    con,
    date(2024, 1, 1),
    date(2024, 12, 31),
    '/data/processed'
)
```

### Zone Map과 필터 푸시다운 활용

Parquet의 Zone Map(열 통계)을 최대한 활용하면 Row Group 단위로 스킵할 수 있다.

```sql
-- Zone Map이 효과적인 경우: 정렬된 컬럼에 범위 필터
-- Parquet을 쓸 때 정렬 순서를 고려해 쓰면 Row Group 스킵 효과 극대화

-- 날짜 컬럼 기준으로 정렬 후 저장
COPY (
    SELECT * FROM events ORDER BY event_date, user_id
)
TO 'events_sorted.parquet' (FORMAT PARQUET, CODEC 'ZSTD');

-- 이후 날짜 범위 쿼리는 해당 Row Group만 읽음
SELECT COUNT(*)
FROM read_parquet('events_sorted.parquet')
WHERE event_date BETWEEN '2024-06-01' AND '2024-06-30';

-- Row Group 통계 확인
SELECT
    row_group_id,
    row_group_num_rows,
    stats_min,
    stats_max,
    stats_min_value,
    stats_max_value
FROM parquet_metadata('events_sorted.parquet')
WHERE name = 'event_date'
ORDER BY row_group_id;
```

### 집계 결과 캐싱

반복적으로 사용되는 집계 결과는 중간 테이블로 저장한다.

```sql
-- 집계 결과를 로컬 테이블로 저장
CREATE TABLE daily_summary AS
SELECT
    DATE_TRUNC('day', event_time) AS event_date,
    region,
    COUNT(*) AS event_count,
    COUNT(DISTINCT user_id) AS unique_users,
    SUM(revenue) AS daily_revenue
FROM read_parquet('s3://bucket/events/**/*.parquet', hive_partitioning=true)
WHERE event_date >= '2024-01-01'
GROUP BY 1, 2;

-- 이후 쿼리는 캐시된 테이블을 사용 (훨씬 빠름)
SELECT
    DATE_TRUNC('month', event_date) AS month,
    region,
    SUM(event_count) AS monthly_events,
    SUM(unique_users) AS total_users
FROM daily_summary
GROUP BY 1, 2
ORDER BY 1, 2;
```

### 인덱스 활용 전략

파티셔닝과 배치 처리로 데이터 규모 문제를 해결했다면, 마지막으로 특정 행을 정확히 찾아야 하는 상황에서 인덱스가 어떤 역할을 하는지 살펴보자. DuckDB는 ART(Adaptive Radix Tree) 인덱스를 지원하지만, 분석 쿼리보다 포인트 룩업에 유리하다.

```sql
-- 기본 키 / 유니크 인덱스 (자동 생성)
CREATE TABLE customers (
    id BIGINT PRIMARY KEY,   -- 자동으로 ART 인덱스 생성
    email VARCHAR UNIQUE,    -- 자동으로 ART 인덱스 생성
    region VARCHAR
);

-- 명시적 인덱스 생성
CREATE INDEX idx_orders_customer ON orders(customer_id);
CREATE INDEX idx_orders_date ON orders(order_date);

-- 인덱스가 유리한 경우: 포인트 룩업
SELECT * FROM customers WHERE id = 12345;         -- INDEX_SCAN
SELECT * FROM orders WHERE customer_id = 12345;   -- INDEX_SCAN

-- 인덱스가 불리한 경우: 대용량 범위 스캔
SELECT SUM(amount) FROM orders WHERE order_date >= '2024-01-01';
-- → 전체 데이터의 상당 부분을 읽어야 하면 SEQ_SCAN이 더 빠름
```

### 워크로드 프로파일링 종합 예시

```python
import duckdb
import json
import time

def profile_query(con, query, label=""):
    """쿼리 실행 계획과 실행 시간을 함께 측정"""
    # 실행 계획 출력
    plan = con.execute(f"EXPLAIN ANALYZE {query}").fetchall()
    print(f"\n{'='*60}")
    print(f"쿼리: {label or query[:50]}")
    print(f"{'='*60}")
    for row in plan:
        print(row[1])

    # 실제 실행 시간 측정 (3회 평균)
    times = []
    for _ in range(3):
        start = time.perf_counter()
        con.execute(query).fetchall()
        times.append(time.perf_counter() - start)

    avg_time = sum(times) / len(times)
    print(f"\n평균 실행 시간: {avg_time*1000:.1f}ms (3회 평균)")
    return avg_time


con = duckdb.connect()
con.execute("SET threads = 8")
con.execute("SET memory_limit = '16GB'")

# 최적화 전후 비교
slow_query = """
    SELECT *
    FROM orders o, customers c
    WHERE o.customer_id = c.id
      AND o.status = 'completed'
"""

fast_query = """
    SELECT o.order_id, o.amount, c.region
    FROM orders o
    JOIN customers c ON o.customer_id = c.id
    WHERE o.status = 'completed'
      AND o.order_date >= '2024-01-01'
"""

t1 = profile_query(con, slow_query, "최적화 전 (SELECT *, CROSS JOIN)")
t2 = profile_query(con, fast_query, "최적화 후 (컬럼 선택, 조건 추가)")
print(f"\n속도 향상: {t1/t2:.1f}x")
```

### 대용량 처리 체크리스트

아래 항목들은 독립적으로 적용할 수 있으며, 이 중 한 가지만 실천해도 수배의 성능 향상을 기대할 수 있다. 현재 가장 병목이 되는 항목부터 하나씩 체크해 보자.
