# 14-1. Java (JDBC)

Java 생태계는 JDBC(Java Database Connectivity)라는 표준 데이터베이스 인터페이스를 오랫동안 사용해왔다. DuckDB는 JDBC 드라이버를 공식 제공하므로, 기존 Java·Spring 프로젝트에 DuckDB를 추가할 때 코드 변경을 최소화할 수 있다.

### 의존성 추가

```xml
<!-- Maven pom.xml -->
<dependency>
    <groupId>org.duckdb</groupId>
    <artifactId>duckdb_jdbc</artifactId>
    <version>1.1.0</version>
</dependency>
```

```kotlin
// Gradle (Kotlin DSL)
dependencies {
    implementation("org.duckdb:duckdb_jdbc:1.1.0")
}
```

### 기본 연결과 쿼리

```java
import org.duckdb.DuckDBConnection;
import java.sql.*;

public class DuckDBExample {
    public static void main(String[] args) throws Exception {

        // 드라이버 로드
        Class.forName("org.duckdb.DuckDBDriver");

        // 인메모리 연결
        Connection conn = DriverManager.getConnection("jdbc:duckdb:");

        // 영구 파일 연결
        Connection conn2 = DriverManager.getConnection("jdbc:duckdb:mydb.duckdb");

        // 읽기 전용
        Properties props = new Properties();
        props.setProperty("duckdb.read_only", "true");
        Connection roConn = DriverManager.getConnection("jdbc:duckdb:mydb.duckdb", props);

        // DDL 실행
        try (Statement stmt = conn.createStatement()) {
            stmt.execute("""
                CREATE TABLE employees (
                    id      INTEGER PRIMARY KEY,
                    name    VARCHAR NOT NULL,
                    dept    VARCHAR,
                    salary  DOUBLE
                )
            """);
        }

        // 다중 행 INSERT
        try (Statement stmt = conn.createStatement()) {
            stmt.execute("""
                INSERT INTO employees VALUES
                    (1, 'Alice', 'Engineering', 95000),
                    (2, 'Bob',   'Sales',       72000),
                    (3, 'Carol', 'Engineering', 105000)
            """);
        }

        // SELECT + ResultSet
        try (Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(
                "SELECT dept, AVG(salary) AS avg_sal FROM employees GROUP BY dept")) {

            ResultSetMetaData meta = rs.getMetaData();
            int cols = meta.getColumnCount();

            // 헤더 출력
            for (int i = 1; i <= cols; i++) {
                System.out.printf("%-15s", meta.getColumnName(i));
            }
            System.out.println();

            // 데이터 출력
            while (rs.next()) {
                System.out.printf("%-15s%.2f%n",
                    rs.getString("dept"),
                    rs.getDouble("avg_sal"));
            }
        }

        conn.close();
    }
}
```

### PreparedStatement (파라미터화)

```java
// 파라미터화 쿼리 (SQL 인젝션 방지)
String sql = "SELECT * FROM employees WHERE dept = ? AND salary > ?";
try (PreparedStatement ps = conn.prepareStatement(sql)) {
    ps.setString(1, "Engineering");
    ps.setDouble(2, 80000.0);

    try (ResultSet rs = ps.executeQuery()) {
        while (rs.next()) {
            System.out.printf("ID: %d, Name: %s, Salary: %.0f%n",
                rs.getInt("id"),
                rs.getString("name"),
                rs.getDouble("salary"));
        }
    }
}

// 배치 INSERT
String insertSql = "INSERT INTO employees VALUES (?, ?, ?, ?)";
try (PreparedStatement ps = conn.prepareStatement(insertSql)) {
    conn.setAutoCommit(false);  // 트랜잭션 시작

    int[][] data = {{4, 0, 0, 0}, {5, 0, 0, 0}};
    String[][] strData = {
        {"4", "David", "Marketing", "68000"},
        {"5", "Eve",   "Engineering", "88000"}
    };

    for (String[] row : strData) {
        ps.setInt(1, Integer.parseInt(row[0]));
        ps.setString(2, row[1]);
        ps.setString(3, row[2]);
        ps.setDouble(4, Double.parseDouble(row[3]));
        ps.addBatch();
    }

    ps.executeBatch();
    conn.commit();
}
```

### 연결 풀 (HikariCP)

```java
import com.zaxxer.hikari.HikariConfig;
import com.zaxxer.hikari.HikariDataSource;

HikariConfig config = new HikariConfig();
config.setJdbcUrl("jdbc:duckdb:analytics.duckdb");
config.setMaximumPoolSize(1);          // DuckDB는 단일 쓰기 연결
config.setConnectionTestQuery("SELECT 1");

HikariDataSource ds = new HikariDataSource(config);

try (Connection conn = ds.getConnection();
     Statement stmt = conn.createStatement();
     ResultSet rs = stmt.executeQuery("SELECT COUNT(*) FROM sales")) {
    rs.next();
    System.out.println(rs.getLong(1));
}
```
