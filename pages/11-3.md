# 11-3. INSERT 문과 Appender API

DuckDB 자체 테이블에 대량 데이터를 적재할 때는 일반 INSERT 외에 성능 최적화된 방법들이 있다. 특히 Python에서 수백만 건을 빠르게 넣어야 할 때 Appender API가 큰 차이를 만든다.

### INSERT 성능 비교

```
대량 적재 방법별 성능 (100만 행 기준, 근사치)
──────────────────────────────────────────────────────
방법                          속도
─────────────────────────     ──────────
단건 INSERT (루프)            매우 느림 (~1000행/초)
다중 행 INSERT                보통       (~10만행/초)
COPY FROM / read_parquet      빠름       (~100만행/초)
Appender API                  매우 빠름  (~200만행/초)
INSERT SELECT FROM 파일        빠름       (~100만행/초)
```

### 단일·다중 행 INSERT

```sql
-- 단건 INSERT (소량 데이터용)
INSERT INTO employees (id, name, department, salary)
VALUES (1, 'Alice', 'Engineering', 95000);

-- 다중 행 INSERT (권장: 한 번에 묶어서)
INSERT INTO employees (id, name, department, salary)
VALUES
    (2, 'Bob',   'Engineering', 82000),
    (3, 'Carol', 'Sales',       75000),
    (4, 'David', 'Marketing',   68000),
    (5, 'Eve',   'Engineering', 105000);

-- INSERT SELECT: 다른 테이블/파일에서
INSERT INTO employees
SELECT id, name, department, salary
FROM read_csv('new_hires.csv', header=true);

-- INSERT SELECT WITH RETURNING
INSERT INTO employees (name, department, salary)
SELECT name, department, salary
FROM staging_employees
WHERE validated = true
RETURNING id, name;
```

### COPY FROM: 파일에서 고속 적재

```sql
-- CSV에서 적재
COPY employees FROM 'employees.csv' (HEADER true, DELIMITER ',');

-- Parquet에서 적재
COPY employees FROM 'employees.parquet' (FORMAT PARQUET);

-- 와일드카드
COPY sales FROM 'data/*.parquet' (FORMAT PARQUET);

-- INSERT SELECT FROM이 COPY FROM보다 유연함
INSERT INTO employees
SELECT * FROM read_parquet('employees.parquet')
WHERE department = 'Engineering';  -- 조건 필터 가능
```

### Appender API (Python)

Appender는 DuckDB가 제공하는 고성능 스트리밍 삽입 API다. 내부적으로 배치 버퍼를 관리해 대량 데이터를 가장 빠르게 삽입한다.

```python
import duckdb

con = duckdb.connect('mydb.duckdb')

# 테이블 생성
con.execute("""
    CREATE TABLE IF NOT EXISTS events (
        id        BIGINT,
        ts        TIMESTAMP,
        user_id   INTEGER,
        event     VARCHAR,
        amount    DOUBLE
    )
""")

# Appender 사용
with con.cursor() as cur:
    appender = duckdb.Appender(con, 'events')

    for i in range(1_000_000):
        appender.append_row(
            i,                                    # id
            '2024-03-15 14:30:00',               # ts
            i % 10000,                            # user_id
            'purchase' if i % 3 == 0 else 'view',# event
            round(i * 0.99 % 1000, 2)            # amount
        )

    appender.close()  # 반드시 close() 호출 (flush)

print(con.execute("SELECT COUNT(*) FROM events").fetchone()[0])
# 1000000
```

### Appender 고급 패턴

```python
import duckdb
import csv
from datetime import datetime

con = duckdb.connect('analytics.duckdb')

con.execute("""
    CREATE TABLE IF NOT EXISTS page_views (
        view_id   BIGINT,
        ts        TIMESTAMP,
        page      VARCHAR,
        user_id   INTEGER,
        duration  INTEGER,
        referrer  VARCHAR
    )
""")

# 대용량 CSV → Appender (메모리 효율적)
def load_csv_with_appender(csv_path: str, table: str):
    appender = duckdb.Appender(con, table)
    count = 0

    with open(csv_path, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            appender.append_row(
                int(row['view_id']),
                datetime.fromisoformat(row['ts']),
                row['page'],
                int(row['user_id']) if row['user_id'] else None,
                int(row['duration']) if row['duration'] else None,
                row.get('referrer') or None
            )
            count += 1
            if count % 100_000 == 0:
                print(f"  {count:,}행 처리 중...")

    appender.close()
    print(f"완료: {count:,}행 적재")

load_csv_with_appender('huge_page_views.csv', 'page_views')
```

```python
# 스키마를 가진 Appender (타입 안전)
import duckdb
from duckdb import typing as T

con = duckdb.connect()

con.execute("""
    CREATE TABLE sensor_data (
        sensor_id  INTEGER,
        ts         TIMESTAMP,
        value      DOUBLE,
        unit       VARCHAR,
        quality    TINYINT
    )
""")

# 컨텍스트 매니저 패턴
with duckdb.Appender(con, 'sensor_data') as appender:
    # 시뮬레이션: 10개 센서 × 100만 타임스탬프
    from datetime import datetime, timedelta
    import random

    base_ts = datetime(2024, 1, 1)
    for i in range(1_000_000):
        appender.append_row(
            i % 10,                                  # sensor_id
            base_ts + timedelta(seconds=i),          # ts
            round(20 + random.gauss(0, 2), 3),       # value
            'Celsius',                               # unit
            1 if random.random() > 0.01 else 0       # quality
        )
# with 블록 탈출 시 자동 close() (flush)

print(con.execute("SELECT COUNT(*) FROM sensor_data").fetchone()[0])
```

### INSERT 성능 최적화 패턴

```python
import duckdb
import pandas as pd
import pyarrow as pa

con = duckdb.connect('analytics.duckdb')

# 방법 1: DataFrame 등록 후 INSERT SELECT (권장)
df = pd.read_parquet('large_data.parquet')
con.register('df_staging', df)
con.execute("INSERT INTO target_table SELECT * FROM df_staging")

# 방법 2: Arrow Table 등록 (메모리 Zero-copy)
arrow_table = pa.parquet.read_table('large_data.parquet')
con.register('arrow_staging', arrow_table)
con.execute("INSERT INTO target_table SELECT * FROM arrow_staging")

# 방법 3: 파일에서 직접 (가장 간단)
con.execute("""
    INSERT INTO target_table
    SELECT * FROM read_parquet('large_data.parquet')
""")

# 방법 4: 트랜잭션으로 묶어 배치 INSERT (소량이지만 여러 번)
con.execute("BEGIN")
for batch in get_data_batches():
    con.executemany(
        "INSERT INTO events VALUES (?, ?, ?, ?)",
        batch
    )
con.execute("COMMIT")
```

### executemany: 파라미터화 배치 INSERT

```python
import duckdb

con = duckdb.connect()
con.execute("""
    CREATE TABLE orders (
        id      INTEGER,
        date    DATE,
        amount  DOUBLE,
        status  VARCHAR
    )
""")

# 리스트 of 튜플로 배치 INSERT
data = [
    (1, '2024-01-05', 15000.0, 'completed'),
    (2, '2024-01-06', 8500.0,  'pending'),
    (3, '2024-01-07', 22000.0, 'completed'),
]

con.executemany(
    "INSERT INTO orders VALUES (?, ?, ?, ?)",
    data
)

# DataFrame의 itertuples로
import pandas as pd
df = pd.read_csv('orders.csv')
con.executemany(
    "INSERT INTO orders VALUES (?, ?, ?, ?)",
    df.itertuples(index=False, name=None)
)
```

### 대량 적재 방법 선택 가이드

```
┌──────────────────────────────────────────────────────────────┐
│              대량 적재 방법 선택 기준                          │
├─────────────────────────┬────────────────────────────────── ─┤
│ 상황                     │ 권장 방법                           │
├─────────────────────────┼────────────────────────────────── ─┤
│ 파일(CSV/Parquet)에서    │ INSERT SELECT FROM read_*()        │
│ DataFrame/Arrow에서      │ register() + INSERT SELECT         │
│ 스트리밍·이터레이터에서  │ Appender API                       │
│ 외부 API·실시간 이벤트   │ Appender API                       │
│ 소량 (<1000행)           │ INSERT VALUES (다중 행)             │
│ 파라미터화 배치           │ executemany()                      │
│ 조건부 삽입               │ INSERT SELECT WHERE / UPSERT       │
└─────────────────────────┴────────────────────────────────── ─┘
```

### INSERT와 레이크하우스의 연결

```python
import duckdb
from deltalake import DeltaTable, write_deltalake

con = duckdb.connect()
con.execute("LOAD iceberg; LOAD delta;")

# 패턴: DuckDB에서 변환 → Delta Lake로 저장
def etl_to_delta(source_path: str, target_path: str, partition_col: str):
    """DuckDB로 변환 후 Delta Lake에 저장"""

    # DuckDB에서 변환
    arrow_result = con.execute(f"""
        SELECT
            id,
            DATE_TRUNC('day', ts)::DATE AS event_date,
            user_id,
            event_type,
            SUM(amount) AS daily_amount,
            COUNT(*)    AS event_count
        FROM read_parquet('{source_path}')
        GROUP BY 1, 2, 3, 4
    """).arrow()

    # Delta Lake로 저장
    write_deltalake(
        target_path,
        arrow_result,
        mode='append',
        partition_by=[partition_col]
    )
    print(f"Delta 저장 완료: {arrow_result.num_rows:,}행")

etl_to_delta('raw/*.parquet', 'delta/daily_summary', 'event_date')

# 저장된 Delta 테이블을 DuckDB로 검증
result = con.execute("""
    SELECT event_date, SUM(daily_amount) AS total
    FROM delta_scan('delta/daily_summary')
    GROUP BY event_date
    ORDER BY event_date DESC
    LIMIT 7
""").df()
print(result)
```
