# 14-5. WebAssembly

### 개요

DuckDB-WASM은 DuckDB를 WebAssembly로 컴파일해 브라우저 안에서 실행한다. 서버 없이 클라이언트 사이드에서 대용량 데이터를 분석할 수 있다.

```
┌────────────────────────────────────────────────────────────┐
│                 DuckDB-WASM 아키텍처                         │
├────────────────────────────────────────────────────────────┤
│  Browser                                                   │
│  ┌──────────────────────────────────────────────────────┐  │
│  │  JavaScript / TypeScript 앱                           │  │
│  │         ↕ async API                                  │  │
│  │  DuckDB-WASM (Web Worker에서 실행)                    │  │
│  │  ┌──────────────────────────────────────────────┐    │  │
│  │  │  DuckDB C++ 코어 (WASM 컴파일)                │    │  │
│  │  └──────────────────────────────────────────────┘    │  │
│  │         ↕ Fetch / OPFS (Origin Private File System)  │  │
│  │  원격 Parquet / S3 파일 직접 쿼리                     │  │
│  └──────────────────────────────────────────────────────┘  │
└────────────────────────────────────────────────────────────┘
```

### 기본 사용

```javascript
import * as duckdb from '@duckdb/duckdb-wasm';

async function initDuckDB() {
    // WASM 번들 선택 (자동)
    const JSDELIVR_BUNDLES = duckdb.getJsDelivrBundles();
    const bundle = await duckdb.selectBundle(JSDELIVR_BUNDLES);

    // Web Worker에서 실행
    const worker_url = URL.createObjectURL(
        new Blob([`importScripts("${bundle.mainWorker}")`],
                 { type: 'text/javascript' })
    );

    const worker = new Worker(worker_url);
    const logger = new duckdb.ConsoleLogger();
    const db = new duckdb.AsyncDuckDB(logger, worker);

    await db.instantiate(bundle.mainModule, bundle.pthreadWorker);

    return db;
}

async function main() {
    const db = await initDuckDB();
    const conn = await db.connect();

    // DDL + DML
    await conn.query(`
        CREATE TABLE sales (
            id     INTEGER,
            region VARCHAR,
            amount DOUBLE
        )
    `);

    await conn.query(`
        INSERT INTO sales VALUES
            (1, 'Seoul',  15000),
            (2, 'Busan',   8500),
            (3, 'Seoul',  22000),
            (4, 'Incheon',11000)
    `);

    // SELECT → Apache Arrow Table
    const result = await conn.query(
        'SELECT region, SUM(amount) AS total FROM sales GROUP BY region'
    );

    // Arrow → JavaScript 배열
    const rows = result.toArray().map(row => row.toJSON());
    console.log(rows);
    // [{ region: 'Seoul', total: 37000 }, { region: 'Busan', total: 8500 }, ...]

    await conn.close();
    await db.terminate();
}

main().catch(console.error);
```

### 원격 Parquet 파일 쿼리

DuckDB-WASM의 핵심 기능 — 서버에 업로드하지 않고 URL의 Parquet를 직접 쿼리한다.

```javascript
import * as duckdb from '@duckdb/duckdb-wasm';

async function queryRemoteParquet(db) {
    const conn = await db.connect();

    // HTTP Range 요청으로 필요한 청크만 다운로드 (전체 다운로드 불필요)
    const result = await conn.query(`
        SELECT
            region,
            COUNT(*) AS orders,
            SUM(amount) AS revenue
        FROM read_parquet('https://example.com/data/sales_2024.parquet')
        WHERE order_date >= '2024-01-01'
        GROUP BY region
        ORDER BY revenue DESC
    `);

    return result.toArray().map(r => r.toJSON());
}

// S3 (퍼블릭 버킷)
async function queryS3(db) {
    const conn = await db.connect();

    await conn.query("LOAD httpfs");
    await conn.query("SET s3_region = 'ap-northeast-2'");

    const result = await conn.query(`
        SELECT * FROM read_parquet('s3://my-public-bucket/data/*.parquet')
        LIMIT 1000
    `);

    return result.toArray().map(r => r.toJSON());
}
```

### 로컬 파일 처리 (File API)

```javascript
// 사용자가 드래그앤드롭하거나 선택한 파일 처리
async function processUploadedFile(db, file) {
    const conn = await db.connect();

    // 파일을 DuckDB-WASM 가상 파일시스템에 등록
    await db.registerFileHandle(
        file.name,               // 가상 경로
        file,                    // File 객체
        duckdb.DuckDBDataProtocol.BROWSER_FILEREADER,
        true
    );

    // 등록된 파일 쿼리
    const ext = file.name.split('.').pop().toLowerCase();
    let query;

    if (ext === 'parquet') {
        query = `SELECT * FROM read_parquet('${file.name}') LIMIT 100`;
    } else if (ext === 'csv') {
        query = `SELECT * FROM read_csv_auto('${file.name}') LIMIT 100`;
    } else if (ext === 'json' || ext === 'ndjson') {
        query = `SELECT * FROM read_json_auto('${file.name}') LIMIT 100`;
    }

    const result = await conn.query(query);
    await conn.close();

    return result.toArray().map(r => r.toJSON());
}

// HTML에서 사용
document.getElementById('fileInput').addEventListener('change', async (e) => {
    const file = e.target.files[0];
    if (!file) return;

    const rows = await processUploadedFile(db, file);
    renderTable(rows);  // 결과 테이블 렌더링
});
```

### React 컴포넌트 패턴

```jsx
import { useState, useEffect, useCallback } from 'react';
import * as duckdb from '@duckdb/duckdb-wasm';

// DuckDB 초기화 훅
function useDuckDB() {
    const [db, setDb] = useState(null);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
        let mounted = true;

        async function init() {
            const bundles = duckdb.getJsDelivrBundles();
            const bundle = await duckdb.selectBundle(bundles);
            const workerUrl = URL.createObjectURL(
                new Blob([`importScripts("${bundle.mainWorker}")`],
                         { type: 'text/javascript' })
            );
            const worker = new Worker(workerUrl);
            const db = new duckdb.AsyncDuckDB(
                new duckdb.ConsoleLogger(), worker
            );
            await db.instantiate(bundle.mainModule);

            if (mounted) {
                setDb(db);
                setLoading(false);
            }
        }

        init();
        return () => { mounted = false; };
    }, []);

    return { db, loading };
}

// 데이터 탐색 컴포넌트
function DataExplorer({ parquetUrl }) {
    const { db, loading } = useDuckDB();
    const [data, setData] = useState([]);
    const [sql, setSql] = useState('');
    const [error, setError] = useState(null);

    const runQuery = useCallback(async (query) => {
        if (!db) return;
        try {
            const conn = await db.connect();
            const result = await conn.query(query);
            setData(result.toArray().map(r => r.toJSON()));
            setError(null);
            await conn.close();
        } catch (err) {
            setError(err.message);
        }
    }, [db]);

    // URL의 Parquet 로드
    useEffect(() => {
        if (db && parquetUrl) {
            runQuery(`
                SELECT * FROM read_parquet('${parquetUrl}') LIMIT 50
            `);
        }
    }, [db, parquetUrl]);

    if (loading) return <div>DuckDB 초기화 중...</div>;

    return (
        <div>
            <textarea
                value={sql}
                onChange={e => setSql(e.target.value)}
                placeholder="SQL 쿼리 입력..."
                rows={4}
                style={{ width: '100%' }}
            />
            <button onClick={() => runQuery(sql)}>실행</button>

            {error && <div style={{ color: 'red' }}>{error}</div>}

            <table>
                <thead>
                    <tr>
                        {data[0] && Object.keys(data[0]).map(k => (
                            <th key={k}>{k}</th>
                        ))}
                    </tr>
                </thead>
                <tbody>
                    {data.map((row, i) => (
                        <tr key={i}>
                            {Object.values(row).map((v, j) => (
                                <td key={j}>{String(v ?? '')}</td>
                            ))}
                        </tr>
                    ))}
                </tbody>
            </table>
        </div>
    );
}

export default DataExplorer;
```

### Observable / Observablehq 패턴

Observable 노트북에서 DuckDB-WASM을 사용하면 데이터 탐색을 인터랙티브하게 공유할 수 있다.

```javascript
// Observable 셀
import * as duckdb from "@duckdb/duckdb-wasm";

const db = await (async () => {
    const bundle = await duckdb.selectBundle(duckdb.getJsDelivrBundles());
    const worker = new Worker(URL.createObjectURL(
        new Blob([`importScripts("${bundle.mainWorker}")`])
    ));
    const db = new duckdb.AsyncDuckDB(new duckdb.VoidLogger(), worker);
    await db.instantiate(bundle.mainModule);
    return db;
})();

// Observable 셀: 쿼리
const conn = await db.connect();
const result = await conn.query(`
    SELECT region, SUM(amount) AS revenue
    FROM read_parquet('https://observablehq.com/@user/sales-data/sales.parquet')
    GROUP BY region
    ORDER BY revenue DESC
`);
result.toArray().map(r => r.toJSON())
```

### WASM 성능 최적화

```javascript
// 팁 1: Web Worker 분리 (메인 스레드 블로킹 방지)
// → 기본적으로 Worker에서 실행되므로 별도 처리 불필요

// 팁 2: OPFS (Origin Private File System) 영구 저장
async function withPersistentDB() {
    const db = await initDuckDB();
    const conn = await db.connect();

    // OPFS에 파일 시스템 마운트 (Chrome 86+)
    await db.open({
        path: 'analytics.duckdb',
        accessMode: duckdb.DuckDBAccessMode.READ_WRITE,
    });

    return conn;
}

// 팁 3: 쿼리 결과 캐싱
const queryCache = new Map();

async function cachedQuery(conn, sql) {
    if (queryCache.has(sql)) {
        return queryCache.get(sql);
    }
    const result = await conn.query(sql);
    const rows = result.toArray().map(r => r.toJSON());
    queryCache.set(sql, rows);
    return rows;
}

// 팁 4: CDN 번들 선택 (크기 vs 기능)
// duckdb-eh.wasm   : 예외 처리 포함 (권장, ~5MB)
// duckdb-mvp.wasm  : 최소 기능 (~3MB, 구형 브라우저)
// duckdb-coi.wasm  : SharedArrayBuffer 사용 (멀티스레드)
```
