# 21-1. 대용량 피처 엔지니어링

피처 엔지니어링은 원시 데이터에서 모델이 학습할 수 있는 수치형 표현을 만드는 과정이다. 집계, 시계열 지연값, 인코딩, 정규화까지 DuckDB SQL로 처리하면 Python 루프 없이 멀티스레드로 병렬 계산된다. Pandas로 같은 로직을 구현하면 행별로 반복문을 돌거나 `apply`를 사용해야 하지만, DuckDB는 이를 단일 SQL 쿼리로 표현하고 내부적으로 벡터화 실행한다.

### 기본 집계 피처

```sql
-- 고객별 구매 행동 피처 생성
CREATE OR REPLACE TABLE features_customer AS
SELECT
    customer_id,

    -- 거래량 피처
    COUNT(*)                                        AS order_count,
    COUNT(DISTINCT DATE_TRUNC('month', order_date)) AS active_months,
    COUNT(DISTINCT product_category)                AS category_diversity,

    -- 금액 피처
    SUM(amount)                                     AS total_spend,
    AVG(amount)                                     AS avg_order_value,
    STDDEV(amount)                                  AS std_order_value,
    MIN(amount)                                     AS min_order_value,
    MAX(amount)                                     AS max_order_value,
    QUANTILE_CONT(amount, 0.25)                     AS q25_order_value,
    QUANTILE_CONT(amount, 0.75)                     AS q75_order_value,

    -- 시간 피처
    MIN(order_date)                                 AS first_order_date,
    MAX(order_date)                                 AS last_order_date,
    DATE_DIFF('day', MIN(order_date), MAX(order_date)) AS customer_lifespan_days,
    DATE_DIFF('day', MAX(order_date), CURRENT_DATE)    AS days_since_last_order,

    -- 비율 피처
    COUNT(*) FILTER (WHERE status = 'COMPLETED')
        * 1.0 / COUNT(*)                            AS completion_rate,
    COUNT(*) FILTER (WHERE status = 'REFUNDED')
        * 1.0 / COUNT(*)                            AS refund_rate

FROM orders
GROUP BY customer_id;
```

### 윈도우 기반 시계열 피처

시계열 데이터에서 과거 N일/N회의 패턴을 피처로 만드는 것이 핵심이다.

```sql
-- 거래 이력 기반 롤링 피처
CREATE OR REPLACE TABLE features_timeseries AS
WITH daily_customer AS (
    SELECT
        customer_id,
        order_date,
        SUM(amount) AS daily_spend,
        COUNT(*)    AS daily_orders
    FROM orders
    WHERE status = 'COMPLETED'
    GROUP BY customer_id, order_date
)
SELECT
    customer_id,
    order_date,
    daily_spend,

    -- 과거 7일 롤링 집계
    SUM(daily_spend) OVER w7   AS spend_last_7d,
    AVG(daily_spend) OVER w7   AS avg_spend_last_7d,
    COUNT(*)         OVER w7   AS active_days_last_7d,

    -- 과거 30일 롤링 집계
    SUM(daily_spend) OVER w30  AS spend_last_30d,
    AVG(daily_spend) OVER w30  AS avg_spend_last_30d,

    -- 전일 대비 변화율
    (daily_spend - LAG(daily_spend, 1) OVER wp) /
        NULLIF(LAG(daily_spend, 1) OVER wp, 0)  AS spend_change_pct,

    -- 이동 평균과의 편차 (이상치 탐지용)
    daily_spend - AVG(daily_spend) OVER w7       AS spend_vs_7d_avg

FROM daily_customer
WINDOW
    wp  AS (PARTITION BY customer_id ORDER BY order_date
            ROWS BETWEEN UNBOUNDED PRECEDING AND 1 PRECEDING),
    w7  AS (PARTITION BY customer_id ORDER BY order_date
            RANGE BETWEEN INTERVAL 6 DAYS PRECEDING AND CURRENT ROW),
    w30 AS (PARTITION BY customer_id ORDER BY order_date
            RANGE BETWEEN INTERVAL 29 DAYS PRECEDING AND CURRENT ROW)
ORDER BY customer_id, order_date;
```

### 범주형 인코딩

```sql
-- 원-핫 인코딩 (카디널리티가 낮은 경우)
CREATE OR REPLACE TABLE features_ohe AS
SELECT
    order_id,
    customer_id,
    amount,

    -- 지역 원-핫 인코딩
    (region = 'Asia')          ::INTEGER AS region_asia,
    (region = 'Europe')        ::INTEGER AS region_europe,
    (region = 'North America') ::INTEGER AS region_north_america,
    (region = 'South America') ::INTEGER AS region_south_america,

    -- 요일 원-핫 인코딩
    (DAYOFWEEK(order_date) = 1)::INTEGER AS is_sunday,
    (DAYOFWEEK(order_date) = 2)::INTEGER AS is_monday,
    (DAYOFWEEK(order_date) = 6)::INTEGER AS is_friday,
    (DAYOFWEEK(order_date) IN (1,7))::INTEGER AS is_weekend

FROM orders;


-- 타겟 인코딩 (카디널리티가 높은 경우, 훈련 데이터에서만 계산)
CREATE OR REPLACE TABLE target_encoding_region AS
SELECT
    region,
    AVG(label) AS region_target_mean,
    COUNT(*)   AS region_count
FROM orders
WHERE split = 'train'   -- 반드시 훈련 데이터만 사용
GROUP BY region;

-- 인코딩 적용
SELECT
    o.*,
    COALESCE(te.region_target_mean, 0.5) AS region_encoded   -- 미등장 범주는 전체 평균으로
FROM orders o
LEFT JOIN target_encoding_region te ON o.region = te.region;


-- 빈도 인코딩 (카테고리 등장 빈도로 대체)
SELECT
    o.*,
    fc.category_freq
FROM orders o
LEFT JOIN (
    SELECT
        product_category,
        COUNT(*) * 1.0 / SUM(COUNT(*)) OVER () AS category_freq
    FROM orders
    GROUP BY product_category
) fc ON o.product_category = fc.product_category;
```

> **주의:** 타겟 인코딩을 적용할 때 전체 데이터의 레이블 평균을 사용하면 **데이터 누수(Data Leakage)**가 발생한다. 검증/테스트 데이터의 정답이 훈련 피처에 스며들어 모델이 실제보다 훨씬 좋은 성능을 보이게 되고, 실전 배포 후 성능이 급격히 떨어지는 원인이 된다. 반드시 `WHERE split = 'train'` 조건으로 훈련 데이터만 사용해 인코딩 테이블을 만들고, 그 값을 검증/테스트 세트에 JOIN 방식으로 적용해야 한다.

### 수치형 변환 및 정규화

```sql
-- 정규화 파라미터를 훈련 데이터에서 계산
CREATE OR REPLACE TABLE normalization_stats AS
SELECT
    AVG(amount)         AS amount_mean,
    STDDEV(amount)      AS amount_std,
    MIN(amount)         AS amount_min,
    MAX(amount)         AS amount_max,
    QUANTILE_CONT(amount, 0.01) AS amount_p1,
    QUANTILE_CONT(amount, 0.99) AS amount_p99
FROM orders
WHERE split = 'train';   -- 훈련 데이터 통계만 사용 (데이터 누수 방지)


-- Z-score 표준화 적용
SELECT
    order_id,
    customer_id,

    -- Z-score 표준화 (평균=0, 표준편차=1)
    (amount - ns.amount_mean) / NULLIF(ns.amount_std, 0) AS amount_zscore,

    -- Min-Max 정규화 (0~1 범위)
    (amount - ns.amount_min) /
        NULLIF(ns.amount_max - ns.amount_min, 0)          AS amount_minmax,

    -- 로그 변환 (우편포 분포 완화)
    LOG(amount + 1)                                        AS amount_log,

    -- 이상치 클리핑 (1~99 퍼센타일 범위로)
    LEAST(GREATEST(amount, ns.amount_p1), ns.amount_p99)   AS amount_clipped,

    -- 버킷 변환 (분위수 기반)
    NTILE(10) OVER (ORDER BY amount)                       AS amount_decile

FROM orders, normalization_stats ns;
```

### 시간 피처 분해

```sql
-- 날짜/시간에서 주기적 피처 추출
SELECT
    order_id,
    order_date,
    order_time,

    -- 달력 피처
    YEAR(order_date)                    AS year,
    MONTH(order_date)                   AS month,
    DAY(order_date)                     AS day_of_month,
    DAYOFWEEK(order_date)               AS day_of_week,   -- 1=일, 7=토
    DAYOFYEAR(order_date)               AS day_of_year,
    WEEKOFYEAR(order_date)              AS week_of_year,
    QUARTER(order_date)                 AS quarter,

    -- 주기적 인코딩 (sin/cos 변환으로 12월→1월 연속성 유지)
    SIN(2 * PI() * MONTH(order_date) / 12)  AS month_sin,
    COS(2 * PI() * MONTH(order_date) / 12)  AS month_cos,
    SIN(2 * PI() * DAYOFWEEK(order_date) / 7) AS dow_sin,
    COS(2 * PI() * DAYOFWEEK(order_date) / 7) AS dow_cos,

    -- 비즈니스 피처
    (DAYOFWEEK(order_date) IN (1,7))::INTEGER AS is_weekend,
    (MONTH(order_date) IN (11,12))   ::INTEGER AS is_holiday_season,
    HOUR(order_time)                          AS hour_of_day,
    (HOUR(order_time) BETWEEN 9 AND 18)::INTEGER AS is_business_hours

FROM orders;
```

### 텍스트 피처 (간단한 수준)

```sql
-- 상품명 텍스트에서 수치 피처 추출
SELECT
    product_id,
    product_name,

    -- 기본 텍스트 통계
    LENGTH(product_name)                AS name_length,
    LENGTH(product_name)
        - LENGTH(REPLACE(product_name, ' ', '')) + 1 AS word_count,

    -- 특정 키워드 포함 여부
    (LOWER(product_name) LIKE '%premium%')::INTEGER  AS has_premium,
    (LOWER(product_name) LIKE '%organic%')::INTEGER  AS has_organic,
    (LOWER(product_name) LIKE '%xl%'
     OR LOWER(product_name) LIKE '%large%')::INTEGER AS is_large_size,

    -- 숫자 포함 여부 (버전, 크기 등)
    (REGEXP_MATCHES(product_name, '\d+'))::INTEGER    AS has_number,

    -- 첫 단어 (브랜드명 추출용)
    SPLIT_PART(product_name, ' ', 1)                  AS first_word

FROM products;
```

### 피처 선택: 상관관계 및 분산 필터

```sql
-- 피처 간 상관관계 행렬
SELECT
    CORR(total_spend,           avg_order_value)   AS spend_vs_avg,
    CORR(total_spend,           order_count)       AS spend_vs_count,
    CORR(days_since_last_order, completion_rate)   AS recency_vs_completion,
    CORR(customer_lifespan_days, total_spend)      AS lifespan_vs_spend
FROM features_customer;


-- 분산이 0인 (상수) 피처 탐지 및 제거 후보
SELECT
    'order_count'      AS feature, STDDEV(order_count)      AS std FROM features_customer
UNION ALL
SELECT 'total_spend',              STDDEV(total_spend)             FROM features_customer
UNION ALL
SELECT 'completion_rate',          STDDEV(completion_rate)         FROM features_customer
ORDER BY std ASC;


-- 결측률 높은 피처 목록
SELECT
    COUNT(*) AS total_rows,
    COUNT(avg_order_value)    AS non_null_avg_order,
    COUNT(std_order_value)    AS non_null_std_order,
    ROUND(100.0 * (COUNT(*) - COUNT(std_order_value)) / COUNT(*), 1)
        AS std_order_null_pct
FROM features_customer;
```
