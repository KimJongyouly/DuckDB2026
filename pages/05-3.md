# 05-3. 날짜/시간 함수

날짜·시간 데이터는 시계열 분석, 코호트 분석, 기간 집계 등 대부분의 분석 작업에서 빠지지 않는다. DuckDB는 표준 SQL의 날짜 함수를 모두 지원하면서, `STRFTIME`·`STRPTIME`·`DATE_TRUNC` 같이 실무에서 자주 쓰이는 편의 함수도 풍부하게 갖추고 있다.

### 현재 시각과 생성

```sql
SELECT CURRENT_DATE;        -- 오늘 날짜 (DATE)
SELECT CURRENT_TIME;        -- 현재 시간 (TIME)
SELECT CURRENT_TIMESTAMP;   -- 현재 날짜+시간 (TIMESTAMP)
SELECT NOW();               -- CURRENT_TIMESTAMP와 동일
SELECT TODAY();             -- CURRENT_DATE와 동일
SELECT YESTERDAY();         -- 어제 날짜
SELECT TOMORROW();          -- 내일 날짜

-- 날짜 생성
SELECT MAKE_DATE(2024, 3, 15);          -- 2024-03-15
SELECT MAKE_TIME(14, 30, 0);            -- 14:30:00
SELECT MAKE_TIMESTAMP(2024, 3, 15, 14, 30, 0);  -- 2024-03-15 14:30:00
```

### 부분 추출

```sql
SELECT
    EXTRACT(YEAR    FROM ts) AS year,
    EXTRACT(MONTH   FROM ts) AS month,
    EXTRACT(DAY     FROM ts) AS day,
    EXTRACT(HOUR    FROM ts) AS hour,
    EXTRACT(MINUTE  FROM ts) AS minute,
    EXTRACT(SECOND  FROM ts) AS second,
    EXTRACT(DOW     FROM ts) AS day_of_week,   -- 0=일요일
    EXTRACT(DOY     FROM ts) AS day_of_year,
    EXTRACT(WEEK    FROM ts) AS iso_week,
    EXTRACT(QUARTER FROM ts) AS quarter,
    EXTRACT(EPOCH   FROM ts) AS unix_epoch
FROM (SELECT TIMESTAMP '2024-03-15 14:30:00' AS ts);

-- 단축 함수 (더 간결)
SELECT
    YEAR(ts), MONTH(ts), DAY(ts),
    HOUR(ts), MINUTE(ts), SECOND(ts),
    DAYOFWEEK(ts),     -- 0=일요일
    DAYOFYEAR(ts),
    WEEKOFYEAR(ts),
    QUARTER(ts)
FROM ...;
```

### 절사와 반올림

```sql
SELECT DATE_TRUNC('year',    ts);   -- 연도 초 (2024-01-01 00:00:00)
SELECT DATE_TRUNC('quarter', ts);   -- 분기 초 (2024-01-01)
SELECT DATE_TRUNC('month',   ts);   -- 월 초   (2024-03-01)
SELECT DATE_TRUNC('week',    ts);   -- 주 초   (2024-03-11, 월요일)
SELECT DATE_TRUNC('day',     ts);   -- 자정    (2024-03-15 00:00:00)
SELECT DATE_TRUNC('hour',    ts);   -- 정각    (2024-03-15 14:00:00)
SELECT DATE_TRUNC('minute',  ts);   -- 분 단위 (2024-03-15 14:30:00)

-- 마지막 날
SELECT LAST_DAY(DATE '2024-02-01');  -- 2024-02-29 (윤년)
SELECT LAST_DAY(DATE '2024-03-15');  -- 2024-03-31
```

### 산술 연산

```sql
-- 날짜 더하기/빼기
SELECT DATE '2024-03-15' + INTERVAL '1 year';    -- 2025-03-15
SELECT DATE '2024-03-15' + INTERVAL '2 months';  -- 2024-05-15
SELECT DATE '2024-03-15' + INTERVAL '10 days';   -- 2024-03-25
SELECT DATE '2024-03-15' - INTERVAL '1 week';    -- 2024-03-08

-- 날짜 차이
SELECT DATE '2024-03-15' - DATE '2024-01-01';    -- 74 (일수)
SELECT DATEDIFF('day',   DATE '2024-01-01', DATE '2024-03-15');  -- 74
SELECT DATEDIFF('month', DATE '2024-01-01', DATE '2024-03-15');  -- 2
SELECT DATEDIFF('year',  DATE '2020-06-15', DATE '2024-03-15');  -- 3

-- AGE: 사람이 읽기 좋은 경과 시간
SELECT AGE(DATE '2024-03-15', DATE '1990-06-20');
-- INTERVAL '33 years 8 months 23 days'

-- 타임스탬프 차이
SELECT TIMESTAMPDIFF('hour', ts1, ts2);
SELECT TIMESTAMPDIFF('minute', ts1, ts2);
```

### 포맷과 파싱

```sql
-- 날짜 → 문자열
SELECT STRFTIME(DATE '2024-03-15', '%Y년 %m월 %d일');   -- '2024년 03월 15일'
SELECT STRFTIME(NOW(), '%Y-%m-%d %H:%M:%S');             -- '2024-03-15 14:30:00'

-- 포맷 코드
-- %Y: 4자리 연도   %y: 2자리 연도
-- %m: 월 (01~12)   %d: 일 (01~31)
-- %H: 시 (00~23)   %I: 시 (01~12)
-- %M: 분 (00~59)   %S: 초 (00~59)
-- %A: 요일명       %a: 요일약자
-- %B: 월 이름      %b: 월 약자
-- %p: AM/PM        %j: 연중 일수

-- 문자열 → 날짜 (STRPTIME)
SELECT STRPTIME('15/03/2024', '%d/%m/%Y');          -- DATE
SELECT STRPTIME('Mar 15, 2024', '%b %d, %Y');       -- DATE
SELECT STRPTIME('2024-03-15 14:30', '%Y-%m-%d %H:%M'); -- TIMESTAMP
```

### 타임존

```sql
-- 타임존 변환
SELECT TIMESTAMPTZ '2024-03-15 05:30:00+00:00'
    AT TIME ZONE 'Asia/Seoul';       -- 2024-03-15 14:30:00 (UTC+9)

SELECT CONVERT_TZ(
    TIMESTAMP '2024-03-15 14:30:00',
    'Asia/Seoul', 'America/New_York'
);

-- 현재 타임존
SELECT CURRENT_SETTING('TimeZone');

-- 세션 타임존 설정
SET TimeZone = 'Asia/Seoul';
```

### 유닉스 시간 변환

```sql
SELECT TO_TIMESTAMP(1710504600);          -- 초 → TIMESTAMPTZ
SELECT EPOCH_MS(1710504600000);           -- 밀리초 → TIMESTAMP
SELECT EPOCH_US(1710504600000000);        -- 마이크로초 → TIMESTAMP

SELECT EPOCH(TIMESTAMP '2024-03-15 14:30:00');  -- TIMESTAMP → 초
SELECT EPOCH_MS(TIMESTAMP '2024-03-15');        -- TIMESTAMP → 밀리초
```
