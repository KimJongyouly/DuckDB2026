# 04-3. ENUM, BLOB, ARRAY

### ENUM: 열거형 타입

ENUM은 미리 정의된 문자열 집합 중 하나만 저장할 수 있는 타입이다. 내부적으로 정수로 저장해 메모리를 절약하고 비교 속도를 높인다.

```sql
-- ENUM 타입 생성 (데이터베이스 수준)
CREATE TYPE status_t AS ENUM ('active', 'inactive', 'pending', 'banned');
CREATE TYPE day_of_week AS ENUM (
    'Monday', 'Tuesday', 'Wednesday', 'Thursday',
    'Friday', 'Saturday', 'Sunday'
);

-- ENUM 타입 사용
CREATE TABLE users (
    id     INTEGER,
    name   VARCHAR,
    status status_t DEFAULT 'pending'
);

INSERT INTO users VALUES (1, 'Alice', 'active');
INSERT INTO users VALUES (2, 'Bob', 'pending');

-- ENUM은 정의된 값만 허용
INSERT INTO users VALUES (3, 'Carol', 'unknown');  -- 오류!

-- ENUM 조회 및 비교
SELECT * FROM users WHERE status = 'active';
SELECT * FROM users WHERE status != 'banned';

-- ENUM 값 목록 조회
SELECT enum_range(NULL::status_t);    -- ['active', 'inactive', 'pending', 'banned']
SELECT enum_first(NULL::status_t);    -- 'active'
SELECT enum_last(NULL::status_t);     -- 'banned'
SELECT enum_code('active'::status_t); -- 0 (내부 정수 코드)

-- ENUM 타입 수정
ALTER TYPE status_t ADD VALUE 'suspended';
ALTER TYPE status_t RENAME VALUE 'banned' TO 'blocked';

-- ENUM 삭제
DROP TYPE status_t;
```

```
ENUM의 장점
────────────────────────────────────────────
VARCHAR 'active' 저장   → 6 bytes
ENUM   'active' 저장    → 1~2 bytes (내부 정수)

카디널리티 낮고 반복 등장하는 문자열 열에 유리
예: 상태, 카테고리, 요일, 등급 등
```

### BLOB: 이진 데이터

BLOB(Binary Large Object)은 원시 이진 데이터를 저장한다.

```sql
-- BLOB 리터럴: \x 헥사 표기
SELECT '\xDEADBEEF'::BLOB;
SELECT '\x48656C6C6F'::BLOB;    -- 'Hello'의 이진 표현

-- 문자열 ↔ BLOB 변환
SELECT encode('Hello'::BLOB);                  -- 'SGVsbG8=' (Base64)
SELECT decode('SGVsbG8=')::VARCHAR;            -- 'Hello'
SELECT 'Hello'::BLOB::VARCHAR;                 -- 'Hello'

-- BLOB 함수
SELECT octet_length('\xDEADBEEF'::BLOB);  -- 4 (바이트 수)

-- 실용 예: 이미지나 파일 해시 저장
CREATE TABLE file_registry (
    path    VARCHAR,
    content BLOB,
    md5     VARCHAR GENERATED ALWAYS AS (md5(content::VARCHAR))
);
```

### ARRAY: 고정 길이 배열

ARRAY는 LIST와 달리 크기가 컴파일 타임에 고정된다.

```sql
-- ARRAY 타입 정의 (크기 고정)
CREATE TABLE vectors (
    id     INTEGER,
    v3     FLOAT[3],    -- 3차원 벡터
    matrix INTEGER[2][3]  -- 2×3 행렬
);

INSERT INTO vectors VALUES
    (1, [1.0, 2.0, 3.0], [[1,2,3],[4,5,6]]);

-- ARRAY 접근
SELECT v3[1], v3[2], v3[3] FROM vectors;

-- ARRAY vs LIST
```

```
┌───────────────────────────────────────────────────────────┐
│                   ARRAY vs LIST 비교                       │
├───────────────────────┬─────────────────────────────────── ┤
│ ARRAY                 │ LIST                               │
├───────────────────────┼─────────────────────────────────── ┤
│ 고정 크기 (컴파일 시)  │ 가변 크기 (런타임)                   │
│ FLOAT[3], INT[2][3]   │ FLOAT[], INTEGER[]                 │
│ 벡터·행렬 연산에 유리  │ 가변 목록·태그 등에 유리             │
│ 머신러닝 임베딩 저장   │ 일반 배열 처리                       │
└───────────────────────┴─────────────────────────────────── ┘
```

```sql
-- 실용 예: 벡터 유사도 검색 (VSS 확장과 함께)
INSTALL vss;
LOAD vss;

CREATE TABLE embeddings (
    id    INTEGER,
    text  VARCHAR,
    vec   FLOAT[384]   -- 384차원 임베딩
);

-- 코사인 유사도
SELECT id, text,
    array_cosine_similarity(vec, [0.1, 0.2, ...]::FLOAT[384]) AS sim
FROM embeddings
ORDER BY sim DESC
LIMIT 5;
```
