# 15-4. Spatial: 공간 데이터 분석

`spatial` 확장은 DuckDB에 GIS(지리 정보 시스템) 기능을 추가한다. PostGIS와 유사한 API로 공간 데이터를 SQL에서 직접 분석할 수 있다.

### 설치와 기본 지오메트리

```sql
INSTALL spatial; LOAD spatial;

-- 기본 지오메트리 생성
SELECT ST_Point(126.9780, 37.5665) AS seoul;        -- 점 (경도, 위도)
SELECT ST_MakeLine(
    ST_Point(126.978, 37.566),
    ST_Point(129.075, 35.179)
) AS line;                                          -- 선분

SELECT ST_MakePolygon(
    ST_GeomFromText('LINESTRING(0 0, 4 0, 4 4, 0 4, 0 0)')
) AS square;                                        -- 다각형

-- WKT (Well-Known Text)로 생성
SELECT ST_GeomFromText('POINT(126.9780 37.5665)') AS point;
SELECT ST_GeomFromText('LINESTRING(0 0, 1 1, 2 0)') AS line;
SELECT ST_GeomFromText('POLYGON((0 0,4 0,4 4,0 4,0 0))') AS poly;

-- GeoJSON으로 생성
SELECT ST_GeomFromGeoJSON('{"type":"Point","coordinates":[126.978,37.566]}');
```

### 공간 측정 함수

```sql
-- 거리 계산 (단위: 원본 좌표계)
SELECT ST_Distance(
    ST_Point(126.978, 37.566),   -- 서울
    ST_Point(129.075, 35.179)    -- 부산
) AS distance_deg;              -- 도(degree) 단위

-- 실제 거리 (미터) — 구면 거리
SELECT ST_Distance_Sphere(
    ST_Point(126.978, 37.566),
    ST_Point(129.075, 35.179)
) / 1000 AS distance_km;        -- ~328 km

-- 면적
SELECT ST_Area(
    ST_GeomFromText('POLYGON((127 37, 128 37, 128 38, 127 38, 127 37))')
) AS area_deg2;

-- 길이
SELECT ST_Length(
    ST_GeomFromText('LINESTRING(0 0, 3 4)')
) AS length;                    -- 5.0

-- 경계 박스
SELECT ST_Extent(geom) FROM locations;

-- 무게중심
SELECT ST_Centroid(
    ST_GeomFromText('POLYGON((0 0,4 0,4 4,0 4,0 0))')
);
-- POINT(2 2)
```

### 공간 관계 함수

```sql
-- 포함 여부
SELECT ST_Contains(
    ST_GeomFromText('POLYGON((0 0,4 0,4 4,0 4,0 0))'),
    ST_Point(2, 2)
) AS contains;  -- true

SELECT ST_Within(
    ST_Point(2, 2),
    ST_GeomFromText('POLYGON((0 0,4 0,4 4,0 4,0 0))')
) AS within;    -- true

-- 교차 여부
SELECT ST_Intersects(
    ST_GeomFromText('LINESTRING(0 0,2 2)'),
    ST_GeomFromText('LINESTRING(0 2,2 0)')
) AS intersects;  -- true

-- 접촉 여부
SELECT ST_Touches(geom_a, geom_b) FROM boundaries;

-- 중첩 여부 (면적)
SELECT ST_Overlaps(poly_a, poly_b) FROM regions;

-- 포함 (WITHIN의 역)
SELECT ST_Contains(polygon, point) FROM ...;
```

### 공간 변환 함수

```sql
-- 버퍼 (반경 N 단위 영역 생성)
SELECT ST_Buffer(
    ST_Point(126.978, 37.566),
    0.01                         -- ~1km (도 단위 근사)
) AS buffer_1km;

-- 유니온 (합집합)
SELECT ST_Union(geom_a, geom_b) FROM regions WHERE id IN (1, 2);

-- 교집합
SELECT ST_Intersection(geom_a, geom_b) FROM ...;

-- 차집합
SELECT ST_Difference(geom_a, geom_b) FROM ...;

-- 단순화 (점 수 줄이기)
SELECT ST_Simplify(complex_geom, 0.001) FROM large_polygons;

-- 볼록 껍질
SELECT ST_ConvexHull(geom) FROM scatter_points;

-- 좌표 반전 (위도·경도 순서 변경)
SELECT ST_FlipCoordinates(geom) FROM ...;
```

### 좌표계 변환 (PROJ)

```sql
-- EPSG 코드로 좌표계 변환
-- WGS84(4326) → 웹 메르카토르(3857)
SELECT ST_Transform(
    ST_Point(126.978, 37.566),
    'EPSG:4326',
    'EPSG:3857'
) AS web_mercator;

-- 한국 좌표계 (GRS80 중부원점 → WGS84)
SELECT ST_Transform(geom, 'EPSG:5179', 'EPSG:4326') AS wgs84
FROM korea_data;
```

### 공간 파일 읽기

```sql
-- GeoJSON 파일 직접 읽기
SELECT * FROM ST_Read('seoul_districts.geojson');

-- Shapefile 읽기
SELECT * FROM ST_Read('admin_boundaries.shp');

-- FlatGeobuf
SELECT * FROM ST_Read('buildings.fgb');

-- GeoPackage
SELECT * FROM ST_Read('data.gpkg', layer='buildings');

-- 여러 파일
SELECT * FROM ST_Read(['a.geojson', 'b.geojson']);

-- 필드 확인
SELECT * FROM ST_ReadMeta('data.geojson');

-- 공간 파일 → Parquet 변환 (고성능 저장)
COPY (SELECT * FROM ST_Read('seoul.geojson'))
TO 'seoul.parquet' (FORMAT PARQUET);
```

### 실전: 반경 내 매장 찾기

```sql
-- 고객 위치와 매장 테이블
CREATE TABLE stores (
    id      INTEGER,
    name    VARCHAR,
    lat     DOUBLE,
    lon     DOUBLE,
    geom    GEOMETRY
);

CREATE TABLE customers (
    id      INTEGER,
    name    VARCHAR,
    lat     DOUBLE,
    lon     DOUBLE
);

-- 데이터 삽입 (geom 열 생성)
INSERT INTO stores
SELECT id, name, lat, lon, ST_Point(lon, lat)
FROM (VALUES
    (1, '강남점', 37.498, 127.028),
    (2, '홍대점', 37.557, 126.924),
    (3, '신촌점', 37.555, 126.937)
) t(id, name, lat, lon);

-- 고객 위치에서 반경 3km 이내 매장 (구면 거리)
WITH customer_loc AS (
    SELECT ST_Point(126.970, 37.540) AS cust_geom  -- 고객 위치
)
SELECT
    s.name AS store_name,
    ROUND(ST_Distance_Sphere(s.geom, c.cust_geom) / 1000, 2) AS dist_km
FROM stores s, customer_loc c
WHERE ST_Distance_Sphere(s.geom, c.cust_geom) <= 3000  -- 3km
ORDER BY dist_km;

-- 행정구역별 매장 수
SELECT
    d.district_name,
    COUNT(s.id) AS store_count
FROM districts d
LEFT JOIN stores s ON ST_Within(s.geom, d.geom)
GROUP BY d.district_name
ORDER BY store_count DESC;
```

### 실전: 공간 집계

```sql
-- 격자(Grid) 기반 밀도 분석
WITH grid AS (
    -- 0.01도 격자 (약 1km)
    SELECT
        FLOOR(lon / 0.01) * 0.01 AS grid_lon,
        FLOOR(lat / 0.01) * 0.01 AS grid_lat,
        COUNT(*) AS event_count
    FROM events
    GROUP BY 1, 2
)
SELECT
    ST_MakeEnvelope(grid_lon, grid_lat,
                   grid_lon + 0.01, grid_lat + 0.01) AS cell_geom,
    event_count
FROM grid
ORDER BY event_count DESC
LIMIT 100;

-- 폴리곤 내 이벤트 수 집계
SELECT
    r.region_name,
    COUNT(e.id) AS events
FROM regions r
LEFT JOIN events e ON ST_Within(
    ST_Point(e.lon, e.lat), r.geom
)
GROUP BY r.region_name
ORDER BY events DESC;
```
