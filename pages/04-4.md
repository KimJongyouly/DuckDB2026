# 04-4. 타입 캐스팅과 변환

### 캐스팅 방법

DuckDB에는 타입 변환을 위한 세 가지 문법이 있다.

```sql
-- 방법 1: :: 연산자 (PostgreSQL 스타일, 가장 간결)
SELECT '42'::INTEGER;
SELECT 3.14::VARCHAR;
SELECT '2024-03-15'::DATE;
SELECT '{"key": "value"}'::JSON;

-- 방법 2: CAST() 함수 (SQL 표준)
SELECT CAST('42' AS INTEGER);
SELECT CAST(3.14 AS VARCHAR);

-- 방법 3: 타입 생성자 함수
SELECT INTEGER '42';
SELECT DATE '2024-03-15';
SELECT TIMESTAMP '2024-03-15 14:30:00';
```

### 안전한 캐스팅: TRY_CAST

```sql
-- CAST: 실패 시 오류 발생
SELECT CAST('abc' AS INTEGER);        -- 오류!

-- TRY_CAST: 실패 시 NULL 반환
SELECT TRY_CAST('abc' AS INTEGER);    -- NULL
SELECT TRY_CAST('42' AS INTEGER);     -- 42
SELECT TRY_CAST('2024-13-99' AS DATE); -- NULL (잘못된 날짜)

-- 실전: 더티 데이터 처리
SELECT
    raw_value,
    TRY_CAST(raw_value AS DECIMAL(10,2)) AS amount,
    CASE
        WHEN TRY_CAST(raw_value AS DECIMAL(10,2)) IS NULL
        THEN '변환 실패'
        ELSE '정상'
    END AS status
FROM raw_transactions;
```

### 묵시적 캐스팅 (Implicit Cast)

DuckDB는 타입이 자동으로 변환되는 경우가 있다.

```sql
-- 정수 → 실수 (자동 확장)
SELECT 1 + 2.5;          -- 3.5 (INTEGER → DOUBLE)

-- 작은 정수 → 큰 정수
SELECT 100::TINYINT + 200::SMALLINT;  -- SMALLINT로 자동 확장

-- 문자열 리터럴 → 날짜 (맥락에 따라)
SELECT * FROM events WHERE date_col = '2024-03-15';
-- '2024-03-15'가 DATE로 자동 변환

-- 주의: VARCHAR → INTEGER는 묵시적 변환 안 됨
SELECT '42' + 1;    -- 오류! 명시적 캐스팅 필요
SELECT '42'::INTEGER + 1;  -- 43
```

### 숫자 타입 간 변환

```sql
-- 정수 → 정수 (손실 가능)
SELECT 300::TINYINT;        -- 오류! (TINYINT 최대 127)
SELECT TRY_CAST(300 AS TINYINT);  -- NULL

-- 실수 → 정수 (소수점 절사)
SELECT 3.9::INTEGER;        -- 3 (내림)
SELECT FLOOR(3.9)::INTEGER; -- 3
SELECT CEIL(3.2)::INTEGER;  -- 4
SELECT ROUND(3.5)::INTEGER; -- 4

-- DECIMAL 정밀도 변환
SELECT 3.14159::DECIMAL(5,2);   -- 3.14 (잘림)
SELECT 999.99::DECIMAL(4,2);    -- 오류! (전체 4자리 초과)
```

### 날짜/시간 변환

```sql
-- 문자열 → 날짜
SELECT '2024-03-15'::DATE;
SELECT '15/03/2024'::DATE;   -- 오류! 형식 맞춰야 함
SELECT STRPTIME('15/03/2024', '%d/%m/%Y')::DATE;  -- 올바른 방법

-- 날짜 → 타임스탬프
SELECT DATE '2024-03-15'::TIMESTAMP;   -- 2024-03-15 00:00:00

-- 타임스탬프 → 날짜 (시간 부분 제거)
SELECT TIMESTAMP '2024-03-15 14:30:00'::DATE;  -- 2024-03-15

-- 유닉스 타임스탬프 변환
SELECT to_timestamp(1710504600);          -- UNIX 초 → TIMESTAMP
SELECT epoch_ms(1710504600000);           -- UNIX 밀리초 → TIMESTAMP
SELECT epoch(TIMESTAMP '2024-03-15 14:30:00');  -- TIMESTAMP → UNIX 초

-- 타임존 변환
SELECT TIMESTAMPTZ '2024-03-15 14:30:00+00:00'
    AT TIME ZONE 'Asia/Seoul';    -- UTC → KST
```

### 복합 타입 변환

```sql
-- 문자열 → JSON
SELECT '{"name": "Alice", "age": 30}'::JSON;

-- JSON 필드 추출
SELECT ('{"name": "Alice", "age": 30}'::JSON)->>'name';  -- 'Alice'
SELECT ('{"scores": [9, 8, 10]}'::JSON)->>'scores';      -- '[9,8,10]'

-- 리스트 → 문자열
SELECT list_aggregate([1,2,3], 'string_agg', ',');  -- '1,2,3'
SELECT array_to_string([1,2,3], ', ');               -- '1, 2, 3'

-- 문자열 → 리스트
SELECT STRING_SPLIT('a,b,c', ',');      -- ['a', 'b', 'c']
SELECT STRING_SPLIT_REGEX('1 2  3', '\s+');  -- ['1', '2', '3']

-- STRUCT → JSON
SELECT row_to_json({'name': 'Alice', 'age': 30});
-- '{"name":"Alice","age":30}'
```

### 타입 검사 함수

```sql
-- 타입 이름 확인
SELECT typeof(42);                        -- 'INTEGER'
SELECT typeof(3.14);                      -- 'DOUBLE'
SELECT typeof('hello');                   -- 'VARCHAR'
SELECT typeof(DATE '2024-03-15');         -- 'DATE'
SELECT typeof([1, 2, 3]);                 -- 'INTEGER[]'
SELECT typeof({'a': 1});                  -- 'STRUCT(a INTEGER)'

-- NULL 여부
SELECT value IS NULL, value IS NOT NULL FROM t;

-- 숫자 판별
SELECT isnan(0.0/0.0);    -- true
SELECT isinf(1.0/0.0);    -- true
SELECT isfinite(42.0);    -- true
```

### 캐스팅 규칙 요약

```
암묵적 캐스팅 허용 방향 (자동으로 확장)
────────────────────────────────────────────────────
TINYINT → SMALLINT → INTEGER → BIGINT → HUGEINT
INTEGER → FLOAT → DOUBLE
FLOAT → DOUBLE
DATE → TIMESTAMP
VARCHAR 리터럴 → DATE/TIMESTAMP (문맥에 따라)

명시적 캐스팅 필요 (:: 또는 CAST)
────────────────────────────────────────────────────
VARCHAR → INTEGER/FLOAT/DATE
DOUBLE → INTEGER (소수점 절사)
TIMESTAMP → DATE (시간 제거)
LIST → VARCHAR
```
