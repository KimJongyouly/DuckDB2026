# 20-2. S3 + Parquet + DuckDB 분석 환경

로컬 레이크에서 구조와 흐름을 검증했다면, 이제 동일한 Bronze → Silver → Gold 패턴을 클라우드로 확장할 차례다. DuckDB의 `httpfs` 확장을 사용하면 S3 파일을 로컬에 다운로드하지 않고 필요한 Row Group만 선택적으로 읽어올 수 있어 비용과 처리 시간을 동시에 줄인다.

클라우드 스토리지 위에 데이터를 쌓고 DuckDB로 직접 쿼리하는 패턴은 별도의 데이터 웨어하우스 없이도 페타바이트급 분석 환경을 구성할 수 있다.

### 분석 환경 아키텍처

```
┌─────────────────────────────────────────────────────────────────┐
│                      S3 데이터 레이크                            │
│                                                                  │
│  s3://my-lake/                                                   │
│  ├── bronze/   (원본 CSV, JSON → Parquet 변환)                   │
│  ├── silver/   (정제 Parquet, Hive 파티셔닝)                     │
│  └── gold/     (집계 마트 Parquet)                               │
└──────────────────────────┬──────────────────────────────────────┘
                           │  httpfs 확장
                           │  S3 Range Request
                           ▼
┌─────────────────────────────────────────────────────────────────┐
│                    DuckDB (로컬 / 컨테이너)                       │
│                                                                  │
│  - Zone Map으로 불필요한 Row Group 스킵                          │
│  - Hive 파티션 프루닝으로 불필요한 파일 스킵                      │
│  - Projection Pushdown으로 필요한 컬럼만 전송                    │
│  - 멀티스레드 병렬 다운로드                                       │
└─────────────────────────────────────────────────────────────────┘
```

### S3 연결 설정

```python
# config/s3_setup.py
import duckdb
import os


def create_s3_connection(profile: str = 'prod') -> duckdb.DuckDBPyConnection:
    con = duckdb.connect(config={
        'threads': os.cpu_count(),
        'memory_limit': '16GB',
        'temp_directory': '/tmp/duckdb_spill',
    })

    con.execute("INSTALL httpfs; LOAD httpfs;")

    if os.environ.get('AWS_EXECUTION_ENV'):
        # EC2/ECS: IAM 역할 자동 감지
        con.execute("""
            CREATE OR REPLACE SECRET s3 (
                TYPE S3,
                PROVIDER CREDENTIAL_CHAIN
            )
        """)
    else:
        # 로컬 개발: 환경 변수 또는 ~/.aws/credentials
        con.execute(f"""
            CREATE OR REPLACE SECRET s3 (
                TYPE S3,
                PROVIDER CREDENTIAL_CHAIN,
                CHAIN 'env;config'
            )
        """)

    # HTTP 재시도 및 캐시 설정
    con.execute("SET http_retries = 3")
    con.execute("SET http_retry_wait_ms = 1000")

    return con
```

### S3 기반 ETL 파이프라인

```python
# pipeline/s3_etl.py
import duckdb
from config.s3_setup import create_s3_connection
from datetime import date, timedelta

LAKE_BUCKET = 's3://my-data-lake'


def s3_bronze_to_silver(target_date: date):
    """S3 Bronze → Silver 변환 (서버리스 ETL)"""
    con = create_s3_connection()

    bronze_path = (
        f"{LAKE_BUCKET}/bronze/orders"
        f"/year={target_date.year}"
        f"/month={target_date.month:02d}"
        f"/day={target_date.day:02d}"
        f"/*.parquet"
    )
    silver_path = (
        f"{LAKE_BUCKET}/silver/orders"
        f"/year={target_date.year}"
        f"/month={target_date.month:02d}"
        f"/day={target_date.day:02d}"
        f"/orders_clean.parquet"
    )

    # S3 파일 존재 확인
    files = con.execute(f"SELECT * FROM glob('{bronze_path}')").fetchall()
    if not files:
        print(f"처리할 파일 없음: {bronze_path}")
        return

    print(f"처리 대상: {len(files)}개 파일")

    # S3 → S3 직접 변환 (로컬 다운로드 없이)
    con.execute(f"""
        COPY (
            WITH cleaned AS (
                SELECT
                    TRY_CAST(order_id    AS BIGINT)       AS order_id,
                    TRY_CAST(customer_id AS BIGINT)       AS customer_id,
                    TRY_CAST(amount      AS DECIMAL(12,2)) AS amount,
                    TRY_CAST(order_date  AS DATE)         AS order_date,
                    UPPER(TRIM(status))                   AS status,
                    UPPER(TRIM(region))                   AS region
                FROM read_parquet('{bronze_path}')
                WHERE order_id IS NOT NULL
                  AND amount > 0
            )
            SELECT * FROM cleaned
            QUALIFY ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY order_date DESC) = 1
        )
        TO '{silver_path}' (FORMAT PARQUET, CODEC 'ZSTD')
    """)
    print(f"Silver 저장 완료: {silver_path}")


def s3_adhoc_analysis(start_date: date, end_date: date):
    """S3 Silver 데이터 임시 분석"""
    con = create_s3_connection()

    result = con.execute(f"""
        SELECT
            DATE_TRUNC('week', order_date) AS week,
            region,
            COUNT(*)                        AS orders,
            COUNT(DISTINCT customer_id)     AS unique_customers,
            ROUND(SUM(amount), 2)           AS revenue,
            ROUND(AVG(amount), 2)           AS avg_order
        FROM read_parquet(
            '{LAKE_BUCKET}/silver/orders/**/*.parquet',
            hive_partitioning = true
        )
        WHERE order_date BETWEEN '{start_date}' AND '{end_date}'
          AND status = 'COMPLETED'
        GROUP BY 1, 2
        ORDER BY 1, revenue DESC
    """).df()

    return result
```

### S3 파티션 관리와 소형 파일 병합

> **주의:** S3 환경에서 파이프라인을 오래 운영하면 파티션마다 수백 개의 작은 Parquet 파일이 쌓이는 "소형 파일 문제(small file problem)"가 발생한다. 파일 수가 많아지면 S3 LIST 요청 비용이 증가하고 DuckDB의 파티션 프루닝 효율도 떨어진다. 월 1회 이상 `compact_silver_partition`을 실행해 파티션을 정리하는 것을 권장한다.

```python
# maintenance/compact.py - 소형 Parquet 파일 병합
import duckdb
from config.s3_setup import create_s3_connection

LAKE_BUCKET = 's3://my-data-lake'


def compact_silver_partition(year: int, month: int):
    """월별 Silver 파티션의 소형 파일을 하나로 병합"""
    con = create_s3_connection()

    source_path = f"{LAKE_BUCKET}/silver/orders/year={year}/month={month:02d}/**/*.parquet"
    target_path = f"{LAKE_BUCKET}/silver/orders_compacted/year={year}/month={month:02d}/data.parquet"

    # 파일 크기 확인
    files = con.execute(f"""
        SELECT
            filename,
            file_size_bytes / 1e6 AS size_mb
        FROM parquet_file_metadata('{source_path}')
        ORDER BY size_mb
    """).fetchdf()

    print(f"병합 대상: {len(files)}개 파일, "
          f"총 {files['size_mb'].sum():.1f}MB")

    # 모든 파일을 하나로 병합
    con.execute(f"""
        COPY (
            SELECT * FROM read_parquet('{source_path}', union_by_name=true)
            ORDER BY order_date, customer_id
        )
        TO '{target_path}' (
            FORMAT PARQUET,
            CODEC 'ZSTD',
            ROW_GROUP_SIZE 100000
        )
    """)
    print(f"병합 완료: {target_path}")


def list_small_partitions(threshold_mb: float = 10.0):
    """소형 파일이 많은 파티션 목록 조회"""
    con = create_s3_connection()

    return con.execute(f"""
        SELECT
            regexp_extract(filename, 'year=(\d+)/month=(\d+)', 0) AS partition,
            COUNT(*) AS file_count,
            ROUND(SUM(file_size_bytes) / 1e6, 1) AS total_mb,
            ROUND(AVG(file_size_bytes) / 1e6, 1) AS avg_file_mb
        FROM parquet_file_metadata(
            '{LAKE_BUCKET}/silver/orders/**/*.parquet'
        )
        GROUP BY 1
        HAVING avg_file_mb < ?
        ORDER BY avg_file_mb
    """, [threshold_mb]).df()
```

### 비용 최적화: S3 요청 수 줄이기

```python
def cost_optimized_query(con, date_filter: str):
    """S3 비용을 최소화하는 쿼리 패턴"""

    # 나쁜 예: 전체 파일 스캔 후 Python 필터
    # → 모든 파일을 S3에서 다운로드
    bad = con.execute(f"""
        SELECT * FROM read_parquet('{LAKE_BUCKET}/silver/orders/**/*.parquet')
    """).df()
    result = bad[bad['order_date'] >= date_filter]   # Python 필터 (이미 전부 다운로드됨)

    # 좋은 예: Hive 파티셔닝 + WHERE로 S3 요청 최소화
    # → 해당 날짜 파티션 파일만 다운로드
    good = con.execute(f"""
        SELECT *
        FROM read_parquet(
            '{LAKE_BUCKET}/silver/orders/**/*.parquet',
            hive_partitioning = true
        )
        WHERE order_date >= '{date_filter}'
    """).df()

    return good
```
