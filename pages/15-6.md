# 15-6. Excel 읽기

`excel` 확장은 `.xlsx` / `.xls` 파일을 직접 쿼리할 수 있게 한다.

### 설치와 기본 읽기

```sql
INSTALL excel; LOAD excel;

-- Excel 파일 읽기 (첫 번째 시트)
SELECT * FROM read_xlsx('sales.xlsx');

-- 기본 옵션
SELECT * FROM read_xlsx(
    'sales.xlsx',
    header     = true,     -- 첫 행을 헤더로 (기본 true)
    sheet      = 'Sheet1', -- 시트 이름 지정
    range      = 'A1:F100' -- 특정 범위만
);

-- 시트 번호로 지정 (0-based)
SELECT * FROM read_xlsx('report.xlsx', sheet = 0);  -- 첫 번째 시트
SELECT * FROM read_xlsx('report.xlsx', sheet = 1);  -- 두 번째 시트
```

### 시트 탐색

```sql
-- 시트 목록 확인
SELECT * FROM excel_sheets('report.xlsx');
-- sheet_name | sheet_index | num_rows | num_cols
-- ───────────┼─────────────┼──────────┼─────────
-- 매출현황   | 0           | 1200     | 8
-- 비용내역   | 1           | 450      | 6
-- 요약       | 2           | 50       | 4

-- 모든 시트 읽어서 통합
SELECT '매출현황' AS sheet, * FROM read_xlsx('report.xlsx', sheet='매출현황')
UNION ALL
SELECT '비용내역', * FROM read_xlsx('report.xlsx', sheet='비용내역');
```

### 범위와 타입 지정

```sql
-- 특정 범위만 읽기
SELECT * FROM read_xlsx(
    'data.xlsx',
    range = 'B2:G500'    -- B2부터 G500까지
);

-- 헤더 행 건너뛰기 (병합 셀 등 복잡한 헤더)
SELECT * FROM read_xlsx(
    'complex.xlsx',
    header = false,       -- 헤더 없이 읽기
    range  = 'A3:H1000'  -- 실제 데이터 시작 행부터
);

-- 타입 명시 (자동 감지 실패 시)
SELECT * FROM read_xlsx(
    'sales.xlsx',
    types = {
        'order_date': 'DATE',
        'amount':     'DOUBLE',
        'quantity':   'INTEGER'
    }
);
```

### 손상된 Excel 처리

```sql
-- 오류 행 건너뛰기
SELECT * FROM read_xlsx('dirty.xlsx', ignore_errors = true);

-- 모두 VARCHAR로 읽어서 수동 변환
SELECT
    id::INTEGER,
    TRY_CAST(order_date AS DATE) AS order_date,
    TRY_CAST(REPLACE(amount, ',', '') AS DOUBLE) AS amount,
    product_name
FROM read_xlsx('data.xlsx', all_varchar = true);
```

### Excel → Parquet 변환 파이프라인

```sql
-- 단순 변환
COPY (SELECT * FROM read_xlsx('monthly_report.xlsx'))
TO 'monthly_report.parquet' (FORMAT PARQUET, COMPRESSION ZSTD);

-- 정제 후 변환 (여러 시트 통합)
CREATE TABLE all_sales AS
SELECT '2024-Q1' AS quarter, * FROM read_xlsx('Q1_sales.xlsx')
UNION ALL
SELECT '2024-Q2',            * FROM read_xlsx('Q2_sales.xlsx')
UNION ALL
SELECT '2024-Q3',            * FROM read_xlsx('Q3_sales.xlsx');

COPY all_sales TO 'sales_2024.parquet' (FORMAT PARQUET);

-- 날짜별 파티션으로 변환
COPY (
    SELECT
        *,
        YEAR(order_date)  AS year,
        MONTH(order_date) AS month
    FROM read_xlsx('sales.xlsx')
    WHERE TRY_CAST(order_date AS DATE) IS NOT NULL
) TO 'output/' (
    FORMAT       PARQUET,
    PARTITION_BY (year, month),
    COMPRESSION  ZSTD
);
```

### Python으로 Excel 일괄 처리

```python
import duckdb
from pathlib import Path

con = duckdb.connect()
con.execute("LOAD excel")

# 디렉터리 내 모든 Excel 파일을 Parquet로 변환
excel_dir = Path('excel_files')
output_dir = Path('parquet_files')
output_dir.mkdir(exist_ok=True)

for xlsx_file in sorted(excel_dir.glob('*.xlsx')):
    output_path = output_dir / xlsx_file.with_suffix('.parquet').name
    print(f"변환: {xlsx_file.name} → {output_path.name}")

    try:
        # 시트 목록 확인
        sheets = con.execute(
            f"SELECT sheet_name FROM excel_sheets('{xlsx_file}')"
        ).fetchall()
        print(f"  시트: {[s[0] for s in sheets]}")

        # 첫 번째 시트만 변환
        con.execute(f"""
            COPY (
                SELECT * FROM read_xlsx('{xlsx_file}', ignore_errors=true)
            )
            TO '{output_path}' (FORMAT PARQUET, COMPRESSION ZSTD)
        """)

        row_count = con.execute(
            f"SELECT COUNT(*) FROM '{output_path}'"
        ).fetchone()[0]
        print(f"  완료: {row_count:,}행")

    except Exception as e:
        print(f"  오류: {e}")
```

### 확장 조합 실전 패턴

#### S3 Parquet → 분석 → Excel 리포트

```python
import duckdb

con = duckdb.connect()
con.execute("LOAD httpfs; LOAD excel;")

# S3에서 읽어 분석
summary = con.execute("""
    SELECT
        region,
        DATE_TRUNC('month', order_date) AS month,
        COUNT(*) AS orders,
        SUM(amount) AS revenue,
        AVG(amount) AS avg_order
    FROM read_parquet(
        's3://my-bucket/sales/year=2024/**/*.parquet',
        hive_partitioning = true
    )
    GROUP BY region, month
    ORDER BY month, revenue DESC
""").df()

# 결과를 Excel로 저장 (openpyxl 활용)
import openpyxl
from openpyxl.styles import Font, PatternFill, Alignment
from openpyxl.utils.dataframe import dataframe_to_rows

wb = openpyxl.Workbook()
ws = wb.active
ws.title = "월별 지역 매출"

# 헤더 스타일
header_fill = PatternFill(fill_type='solid', fgColor='4472C4')
header_font = Font(color='FFFFFF', bold=True)

for r in dataframe_to_rows(summary, index=False, header=True):
    ws.append(r)

for cell in ws[1]:
    cell.fill = header_fill
    cell.font = header_font
    cell.alignment = Alignment(horizontal='center')

wb.save('monthly_regional_report.xlsx')
print("리포트 저장 완료: monthly_regional_report.xlsx")
```

#### GeoJSON + S3 공간 분석

```sql
LOAD httpfs;
LOAD spatial;

-- S3의 공간 데이터와 HTTP GeoJSON 결합
WITH
store_locs AS (
    SELECT
        store_id,
        ST_Point(longitude, latitude) AS geom
    FROM read_parquet('s3://retail-data/stores/locations.parquet')
),
districts AS (
    SELECT
        properties->>'district' AS district_name,
        ST_GeomFromGeoJSON(geometry::VARCHAR) AS geom
    FROM read_json(
        'https://geojson.example.com/seoul_districts.geojson',
        format = 'auto'
    )
)
SELECT
    d.district_name,
    COUNT(s.store_id) AS store_count
FROM districts d
LEFT JOIN store_locs s ON ST_Within(s.geom, d.geom)
GROUP BY d.district_name
ORDER BY store_count DESC;
```
