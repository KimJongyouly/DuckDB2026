```go
package main

import (
    "database/sql"
    "fmt"
    "log"

    _ "github.com/marcboeker/go-duckdb"
)

func main() {
    // 인메모리 연결
    db, err := sql.Open("duckdb", "")
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // 영구 파일 연결
    // db, err := sql.Open("duckdb", "mydb.duckdb")

    // DDL
    _, err = db.Exec(`
        CREATE TABLE employees (
            id     INTEGER PRIMARY KEY,
            name   VARCHAR NOT NULL,
            dept   VARCHAR,
            salary DOUBLE
        )
    `)
    if err != nil {
        log.Fatal(err)
    }

    // INSERT (파라미터화)
    stmt, err := db.Prepare(
        "INSERT INTO employees VALUES (?, ?, ?, ?)")
    if err != nil {
        log.Fatal(err)
    }
    defer stmt.Close()

    data := [][]interface{}{
        {1, "Alice", "Engineering", 95000.0},
        {2, "Bob",   "Sales",       72000.0},
        {3, "Carol", "Engineering", 105000.0},
    }
    for _, row := range data {
        _, err = stmt.Exec(row...)
        if err != nil {
            log.Fatal(err)
        }
    }

    // SELECT
    rows, err := db.Query(`
        SELECT dept, AVG(salary) AS avg_sal
        FROM employees
        GROUP BY dept
        ORDER BY avg_sal DESC
    `)
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()

    fmt.Printf("%-20s %10s\n", "부서", "평균 급여")
    fmt.Println("------------------------------")
    for rows.Next() {
        var dept string
        var avgSal float64
        if err := rows.Scan(&dept, &avgSal); err != nil {
            log.Fatal(err)
        }
        fmt.Printf("%-20s %10.0f\n", dept, avgSal)
    }

    // 파일 직접 쿼리
    var count int64
    err = db.QueryRow(
        "SELECT COUNT(*) FROM read_parquet('data/*.parquet')").Scan(&count)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("\nParquet 행 수: %d\n", count)
}
```

```rust
use duckdb::{Connection, Result, params};

fn main() -> Result<()> {
    // 인메모리 연결
    let conn = Connection::open_in_memory()?;

    // 영구 파일 연결
    // let conn = Connection::open("mydb.duckdb")?;

    // DDL
    conn.execute_batch("
        CREATE TABLE employees (
            id     INTEGER PRIMARY KEY,
            name   VARCHAR NOT NULL,
            dept   VARCHAR,
            salary DOUBLE
        );
        INSERT INTO employees VALUES
            (1, 'Alice', 'Engineering', 95000),
            (2, 'Bob',   'Sales',       72000),
            (3, 'Carol', 'Engineering', 105000);
    ")?;

    // SELECT
    let mut stmt = conn.prepare(
        "SELECT dept, AVG(salary) AS avg_sal FROM employees GROUP BY dept"
    )?;

    let rows = stmt.query_map([], |row| {
        Ok((
            row.get::<_, String>(0)?,   // dept
            row.get::<_, f64>(1)?,      // avg_sal
        ))
    })?;

    println!("{:<20} {:>10}", "부서", "평균 급여");
    println!("{}", "-".repeat(32));
    for row in rows {
        let (dept, avg_sal) = row?;
        println!("{:<20} {:>10.0}", dept, avg_sal);
    }

    // 파라미터화 쿼리
    let dept = "Engineering";
    let min_salary: f64 = 80000.0;

    let mut stmt = conn.prepare(
        "SELECT id, name, salary FROM employees WHERE dept = ?1 AND salary > ?2"
    )?;

    let engineers: Vec<(i32, String, f64)> = stmt.query_map(
        params![dept, min_salary],
        |row| Ok((row.get(0)?, row.get(1)?, row.get(2)?))
    )?.filter_map(|r| r.ok()).collect();

    println!("\n{}의 고연봉자:", dept);
    for (id, name, salary) in &engineers {
        println!("  ID: {}, Name: {}, Salary: {:.0}", id, name, salary);
    }

    // 파일 직접 쿼리
    let count: i64 = conn.query_row(
        "SELECT COUNT(*) FROM read_parquet('data/*.parquet')",
        [],
        |row| row.get(0)
    )?;
    println!("\nParquet 행 수: {}", count);

    Ok(())
}
```

```rust
// Rust Arrow 연동
use duckdb::{Connection, Result};
use duckdb::arrow::record_batch::RecordBatch;

fn main() -> Result<()> {
    let conn = Connection::open_in_memory()?;

    conn.execute_batch("
        CREATE TABLE t AS
        SELECT range AS id, random() AS val
        FROM range(100);
    ")?;

    // Arrow RecordBatch로 결과 받기
    let mut stmt = conn.prepare("SELECT * FROM t")?;
    let batches: Vec<RecordBatch> = stmt.query_arrow([])?.collect();

    for batch in &batches {
        println!("배치: {}행 × {}열", batch.num_rows(), batch.num_columns());
    }

    Ok(())
}
```

```r
# R data.frame ↔ DuckDB 연동
library(duckdb)

con <- dbConnect(duckdb())

# data.frame → DuckDB 테이블
sales_df <- data.frame(
    id     = 1:5,
    region = c("Seoul", "Busan", "Seoul", "Incheon", "Busan"),
    amount = c(15000, 8500, 22000, 11000, 18500)
)

# register: data.frame을 뷰로 등록
duckdb_register(con, "sales_view", sales_df)

result <- dbGetQuery(con, "
    SELECT region, SUM(amount) AS total
    FROM sales_view
    GROUP BY region
    ORDER BY total DESC
")
print(result)

# DuckDB → data.frame
big_df <- dbGetQuery(con,
    "SELECT * FROM read_parquet('data/*.parquet')"
)

# Arrow 기반 고속 전송
library(arrow)
arrow_tbl <- dbGetQuery(con,
    "SELECT * FROM large_table",
    arrow = TRUE         # arrow::Table로 반환
)

# dplyr 연동 (dbplyr)
library(dplyr)

tbl_emp <- tbl(con, "employees")

# dplyr 문법이 SQL로 자동 변환
result <- tbl_emp |>
    filter(dept == "Engineering", salary > 80000) |>
    select(name, salary) |>
    arrange(desc(salary)) |>
    collect()              # 실제 실행

print(result)

# 연결 종료
dbDisconnect(con, shutdown = TRUE)
```

```r
# R에서 CSV/Parquet 직접 분석
library(duckdb)
con <- dbConnect(duckdb())

# 파일 직접 쿼리 (테이블 생성 불필요)
result <- dbGetQuery(con, "
    SELECT
        YEAR(order_date) AS year,
        region,
        SUM(amount) AS revenue
    FROM read_parquet('sales/*.parquet')
    WHERE order_date >= '2024-01-01'
    GROUP BY ALL
    ORDER BY year, revenue DESC
")

# ggplot2로 시각화
library(ggplot2)
ggplot(result, aes(x = region, y = revenue / 1e6, fill = factor(year))) +
    geom_col(position = "dodge") +
    labs(title = "지역·연도별 매출", y = "매출 (백만원)", fill = "연도") +
    theme_minimal()
```
