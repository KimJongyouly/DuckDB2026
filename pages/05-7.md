# 05-7. 윈도우 함수 심화

3장에서 윈도우 함수의 기본을 다뤘다면, 여기서는 고급 패턴과 성능 최적화를 집중적으로 살펴본다.

### 윈도우 정의 재사용 (WINDOW 절)

같은 윈도우를 여러 함수에서 반복 정의하면 코드가 지저분해진다. WINDOW 절로 이름을 붙여 재사용할 수 있다.

```sql
SELECT
    name, department, salary, hire_date,
    RANK()        OVER w AS rank,
    DENSE_RANK()  OVER w AS dense_rank,
    ROW_NUMBER()  OVER w AS row_num,
    FIRST_VALUE(name) OVER w AS top_earner,
    SUM(salary)   OVER dept_w AS dept_total,
    AVG(salary)   OVER dept_w AS dept_avg
FROM employees
WINDOW
    w      AS (PARTITION BY department ORDER BY salary DESC),
    dept_w AS (PARTITION BY department);
```

### 프레임 심화

```sql
-- 시계열 이동 평균 (최근 7일)
SELECT
    date,
    revenue,
    AVG(revenue) OVER (
        ORDER BY date
        RANGE BETWEEN INTERVAL '6 days' PRECEDING AND CURRENT ROW
    ) AS moving_avg_7d
FROM daily_sales;

-- 누적 합계 vs 이동 합계
SELECT
    date,
    revenue,
    -- 누적 합계 (처음부터 현재까지)
    SUM(revenue) OVER (
        ORDER BY date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative,
    -- 이동 합계 (최근 3행)
    SUM(revenue) OVER (
        ORDER BY date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS rolling_3,
    -- 전체 합계 (고정)
    SUM(revenue) OVER () AS grand_total
FROM daily_sales;

-- EXCLUDE: 특정 행 제외 (DuckDB 전용)
SELECT
    id, value,
    -- 현재 행을 제외한 전후 2행 평균 (자신 제외)
    AVG(value) OVER (
        ORDER BY id
        ROWS BETWEEN 2 PRECEDING AND 2 FOLLOWING
        EXCLUDE CURRENT ROW
    ) AS neighbor_avg
FROM data;
```

### 갭과 아일랜드 (Gaps and Islands)

연속 구간을 찾거나 끊긴 구간을 탐지하는 고전 패턴이다.

```sql
-- 아일랜드: 연속된 날짜 그룹 찾기
WITH
numbered AS (
    SELECT
        date,
        ROW_NUMBER() OVER (ORDER BY date) AS rn
    FROM active_dates
),
grouped AS (
    SELECT
        date,
        date - rn * INTERVAL '1 day' AS grp  -- 연속이면 같은 그룹
    FROM numbered
)
SELECT
    MIN(date) AS island_start,
    MAX(date) AS island_end,
    COUNT(*) AS days
FROM grouped
GROUP BY grp
ORDER BY island_start;

-- 갭: 빠진 날짜 찾기
SELECT
    curr_date + INTERVAL '1 day' AS gap_start,
    next_date - INTERVAL '1 day' AS gap_end
FROM (
    SELECT
        date AS curr_date,
        LEAD(date) OVER (ORDER BY date) AS next_date
    FROM active_dates
) t
WHERE next_date - curr_date > INTERVAL '1 day';
```

### 세션 분석 패턴

```sql
-- 30분 이상 간격이면 새 세션으로 분류
WITH
with_gap AS (
    SELECT
        user_id,
        event_time,
        LAG(event_time) OVER (
            PARTITION BY user_id ORDER BY event_time
        ) AS prev_time
    FROM events
),
with_session_flag AS (
    SELECT *,
        CASE
            WHEN prev_time IS NULL
              OR event_time - prev_time > INTERVAL '30 minutes'
            THEN 1 ELSE 0
        END AS is_new_session
    FROM with_gap
)
SELECT
    user_id,
    event_time,
    SUM(is_new_session) OVER (
        PARTITION BY user_id ORDER BY event_time
        ROWS UNBOUNDED PRECEDING
    ) AS session_id
FROM with_session_flag;
```

### 피벗과 집계 결합

```sql
-- 월별 부서별 급여 피벗
SELECT
    department,
    SUM(salary) FILTER (WHERE MONTH(hire_date) = 1)  AS jan,
    SUM(salary) FILTER (WHERE MONTH(hire_date) = 2)  AS feb,
    SUM(salary) FILTER (WHERE MONTH(hire_date) = 3)  AS mar,
    -- ...
    SUM(salary) FILTER (WHERE MONTH(hire_date) = 12) AS dec
FROM employees
GROUP BY department;

-- PIVOT 문법 (DuckDB 전용)
PIVOT employees
ON MONTH(hire_date)
USING SUM(salary)
GROUP BY department;
```

### 고급 순위 패턴

```sql
-- Top-N per group (QUALIFY 활용)
SELECT name, department, salary
FROM employees
QUALIFY RANK() OVER (PARTITION BY department ORDER BY salary DESC) <= 3;

-- 이전 대비 변화율
SELECT
    date,
    revenue,
    LAG(revenue) OVER (ORDER BY date) AS prev_revenue,
    ROUND(
        (revenue - LAG(revenue) OVER (ORDER BY date))
        / LAG(revenue) OVER (ORDER BY date) * 100, 2
    ) AS pct_change
FROM daily_sales
QUALIFY prev_revenue IS NOT NULL;  -- 첫 행 제외

-- 전년 동기 비교 (YoY)
SELECT
    date,
    revenue,
    LAG(revenue, 365) OVER (ORDER BY date) AS prev_year,
    revenue / NULLIF(LAG(revenue, 365) OVER (ORDER BY date), 0) - 1 AS yoy_growth
FROM daily_revenue;

-- 누적 비율 (Running Percent)
SELECT
    category,
    sales,
    ROUND(
        SUM(sales) OVER (ORDER BY sales DESC ROWS UNBOUNDED PRECEDING)
        / SUM(sales) OVER () * 100, 2
    ) AS cumulative_pct
FROM category_sales
ORDER BY sales DESC;
```

### 윈도우 함수 성능 팁

```sql
-- 팁 1: 동일 PARTITION/ORDER BY는 WINDOW 절로 묶기 (재계산 방지)
-- 나쁜 예
SELECT
    RANK() OVER (PARTITION BY dept ORDER BY salary DESC),
    SUM(salary) OVER (PARTITION BY dept ORDER BY salary DESC),
    AVG(salary) OVER (PARTITION BY dept ORDER BY salary DESC)
FROM employees;

-- 좋은 예
SELECT
    RANK()       OVER w,
    SUM(salary)  OVER w,
    AVG(salary)  OVER w
FROM employees
WINDOW w AS (PARTITION BY dept ORDER BY salary DESC);

-- 팁 2: QUALIFY로 서브쿼리 제거
-- 나쁜 예 (서브쿼리)
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (PARTITION BY dept ORDER BY salary DESC) AS rn
    FROM employees
) t WHERE rn = 1;

-- 좋은 예 (QUALIFY)
SELECT * FROM employees
QUALIFY ROW_NUMBER() OVER (PARTITION BY dept ORDER BY salary DESC) = 1;
```
