# 19-1. Docker로 배포하기

DuckDB는 PostgreSQL이나 MySQL처럼 별도의 서버 프로세스를 실행하는 데이터베이스가 아니다. 프로세스에 라이브러리 형태로 내장(embed)되어 함께 동작하는 임베디드 데이터베이스이기 때문에, Docker로 배포할 때도 "DuckDB 서버 컨테이너"를 따로 띄우는 것이 아니라 DuckDB를 사용하는 애플리케이션(Python 스크립트, REST API, ETL 파이프라인 등)을 컨테이너화하는 방식을 취한다. 이 차이를 이해하면 아키텍처 설계가 훨씬 명확해진다.

### 기본 Python 애플리케이션 컨테이너

```dockerfile
# Dockerfile
FROM python:3.12-slim

WORKDIR /app

# 시스템 의존성 (httpfs 확장의 SSL 지원)
RUN apt-get update && apt-get install -y --no-install-recommends \
    libssl-dev \
    ca-certificates \
    curl \
 && rm -rf /var/lib/apt/lists/*

# Python 의존성
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY . .

# 데이터 및 스필 디렉터리
RUN mkdir -p /data /tmp/duckdb_spill

ENV DUCKDB_MEMORY_LIMIT=8GB
ENV DUCKDB_THREADS=4
ENV DUCKDB_TEMP_DIR=/tmp/duckdb_spill

CMD ["python", "main.py"]
```

```
# requirements.txt
duckdb==1.1.3
pandas==2.2.0
pyarrow==15.0.0
fastapi==0.110.0
uvicorn==0.29.0
```

```python
# main.py - 환경 변수로 DuckDB 설정
import duckdb
import os

def create_connection():
    con = duckdb.connect(
        database=os.environ.get('DUCKDB_DATABASE', ':memory:'),
        config={
            'threads': int(os.environ.get('DUCKDB_THREADS', '4')),
            'memory_limit': os.environ.get('DUCKDB_MEMORY_LIMIT', '8GB'),
            'temp_directory': os.environ.get('DUCKDB_TEMP_DIR', '/tmp/duckdb'),
        }
    )
    return con
```

### FastAPI REST API 컨테이너

```python
# api.py - DuckDB 기반 REST API
from fastapi import FastAPI, HTTPException
from contextlib import asynccontextmanager
import duckdb
import os

# 연결 풀 (읽기 전용 다중 연결 지원)
_connections = {}

@asynccontextmanager
async def lifespan(app: FastAPI):
    # 시작 시: DB 연결 초기화
    _connections['main'] = duckdb.connect(
        database='/data/analytics.db',
        config={
            'threads': int(os.environ.get('DUCKDB_THREADS', '4')),
            'memory_limit': os.environ.get('DUCKDB_MEMORY_LIMIT', '8GB'),
        }
    )
    yield
    # 종료 시: 연결 닫기
    _connections['main'].close()

app = FastAPI(lifespan=lifespan)

@app.get("/query/summary")
async def get_summary(region: str = None):
    con = _connections['main']
    try:
        if region:
            result = con.execute("""
                SELECT region, SUM(amount) AS total, COUNT(*) AS cnt
                FROM sales
                WHERE region = ?
                GROUP BY region
            """, [region]).fetchdf()
        else:
            result = con.execute("""
                SELECT region, SUM(amount) AS total, COUNT(*) AS cnt
                FROM sales
                GROUP BY region
                ORDER BY total DESC
            """).fetchdf()
        return result.to_dict(orient='records')
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/health")
async def health():
    try:
        _connections['main'].execute("SELECT 1").fetchone()
        return {"status": "ok"}
    except Exception as e:
        raise HTTPException(status_code=503, detail=str(e))
```

```dockerfile
# Dockerfile.api
FROM python:3.12-slim

WORKDIR /app

RUN apt-get update && apt-get install -y --no-install-recommends \
    libssl-dev ca-certificates \
 && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY api.py .

EXPOSE 8000

CMD ["uvicorn", "api:app", "--host", "0.0.0.0", "--port", "8000", "--workers", "1"]
```

> **주의:** DuckDB는 하나의 파일에 동시에 쓸 수 있는 연결이 단 하나뿐이다. 이것이 DuckDB의 동시성 모델이다. PostgreSQL처럼 여러 클라이언트가 동시에 쓰기 연결을 맺는 방식과 달리, DuckDB는 파일 잠금(lock)으로 단일 쓰기 연결을 강제한다. 그 덕분에 별도의 잠금 관리 오버헤드 없이 고속 분석 성능을 낼 수 있지만, API 서버처럼 여러 요청을 처리하는 환경에서는 `--workers 1` 설정이 필수다. 읽기 전용 연결은 여러 개를 동시에 열 수 있으므로, ETL(쓰기)과 API(읽기 전용)를 역할에 따라 분리하면 이 제약을 자연스럽게 우회할 수 있다.

### Docker Compose 파이프라인 구성

```yaml
# docker-compose.yml
version: '3.9'

services:
  # ETL 파이프라인 (쓰기 담당)
  etl:
    build:
      context: .
      dockerfile: Dockerfile.etl
    volumes:
      - duckdb_data:/data
      - duckdb_spill:/tmp/duckdb_spill
    environment:
      - DUCKDB_DATABASE=/data/analytics.db
      - DUCKDB_MEMORY_LIMIT=16GB
      - DUCKDB_THREADS=8
      - DUCKDB_TEMP_DIR=/tmp/duckdb_spill
      - AWS_ACCESS_KEY_ID=${AWS_ACCESS_KEY_ID}
      - AWS_SECRET_ACCESS_KEY=${AWS_SECRET_ACCESS_KEY}
    deploy:
      resources:
        limits:
          memory: 20G
          cpus: '8'

  # API 서버 (읽기 전용)
  api:
    build:
      context: .
      dockerfile: Dockerfile.api
    ports:
      - "8000:8000"
    volumes:
      - duckdb_data:/data:ro   # 읽기 전용 마운트
    environment:
      - DUCKDB_DATABASE=/data/analytics.db
      - DUCKDB_MEMORY_LIMIT=8GB
      - DUCKDB_THREADS=4
    depends_on:
      - etl
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # 스케줄러 (cron 기반 ETL 트리거)
  scheduler:
    image: mcuadros/ofelia:latest
    depends_on:
      - etl
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - ./scheduler.ini:/etc/ofelia/config.ini

volumes:
  duckdb_data:
    driver: local
    driver_opts:
      type: none
      device: /mnt/fast_ssd/duckdb
      o: bind
  duckdb_spill:
    driver: local
```

```ini
# scheduler.ini (Ofelia 스케줄러)
[job-exec "daily-etl"]
schedule = 0 2 * * *
container = etl
command = python etl_daily.py
```

### 멀티스테이지 빌드로 이미지 경량화

```dockerfile
# Dockerfile.slim - 멀티스테이지 빌드
FROM python:3.12 AS builder

WORKDIR /build
COPY requirements.txt .
RUN pip install --no-cache-dir --prefix=/install -r requirements.txt

# 실행 이미지
FROM python:3.12-slim

RUN apt-get update && apt-get install -y --no-install-recommends \
    libssl3 ca-certificates \
 && rm -rf /var/lib/apt/lists/*

COPY --from=builder /install /usr/local
WORKDIR /app
COPY . .

RUN useradd -r -s /bin/false appuser && \
    mkdir -p /data /tmp/duckdb && \
    chown -R appuser:appuser /data /tmp/duckdb /app

USER appuser

CMD ["python", "main.py"]
```
