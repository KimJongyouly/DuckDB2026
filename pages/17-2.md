# 17-2. Pragmas 활용

Pragma는 DuckDB의 내부 상태를 조회하거나 특정 동작을 제어하는 명령이다. `PRAGMA key` 형식으로 값을 조회하고, `PRAGMA key = value` 또는 `SET key = value`로 설정한다.

### 데이터베이스 정보 조회

```sql
-- 연결된 데이터베이스 목록
PRAGMA database_list;
-- ┌──────┬──────────┬──────────────────────────┐
-- │  seq │   name   │         file             │
-- ├──────┼──────────┼──────────────────────────┤
-- │    0 │ memory   │                          │
-- │    1 │ mydb     │ /data/analysis.db        │
-- └──────┴──────────┴──────────────────────────┘

-- 현재 데이터베이스 버전
PRAGMA version;

-- 테이블 정보
PRAGMA table_info('employees');
-- ┌──────┬──────────┬─────────┬─────────┬───────────┐
-- │  cid │   name   │  type   │ notnull │ dflt_value│
-- ├──────┼──────────┼─────────┼─────────┼───────────┤
-- │    0 │ id       │ INTEGER │  true   │ (null)    │
-- │    1 │ name     │ VARCHAR │  false  │ (null)    │
-- └──────┴──────────┴─────────┴─────────┴───────────┘

-- 모든 테이블 목록
PRAGMA show_tables;
PRAGMA show_tables_expanded;  -- 상세 정보 포함

-- 저장 공간 사용량
PRAGMA storage_info('employees');
```

### 성능 관련 Pragma

```sql
-- 스레드 수 확인 및 설정
PRAGMA threads;          -- 현재 값 조회
PRAGMA threads = 4;      -- 값 설정

-- 메모리 한도
PRAGMA memory_limit;
PRAGMA memory_limit = '8GB';

-- 체크포인트 WAL 크기 임계값
PRAGMA wal_autocheckpoint;
PRAGMA wal_autocheckpoint = 1000;  -- 1000 페이지마다 자동 체크포인트

-- 블록 캐시 크기 (Buffer Manager)
-- 연결 시 config로만 설정 가능
-- config={'buffer_manager_block_alloc_size': '1MB'}
```

### 쿼리 프로파일링

```sql
-- 프로파일링 활성화
PRAGMA enable_profiling;
PRAGMA enable_profiling = 'json';   -- JSON 형식 출력
PRAGMA enable_profiling = 'query_tree';  -- 트리 형식

-- 프로파일 출력 파일 지정
PRAGMA profiling_output = '/tmp/profile.json';

-- 프로파일링 대상 쿼리 실행
SELECT
    region,
    SUM(amount) AS total
FROM sales
GROUP BY region
ORDER BY total DESC;

-- 프로파일링 비활성화
PRAGMA disable_profiling;

-- EXPLAIN으로 실행 계획 확인
EXPLAIN SELECT * FROM sales WHERE region = 'Asia';

-- 상세 실행 계획 (실제 실행 포함)
EXPLAIN ANALYZE SELECT * FROM sales WHERE region = 'Asia';
```

실행 계획 출력 예시:

```
┌─────────────────────────────────────────┐
│         HASH_GROUP_BY                   │
│   Aggregates: sum(amount)               │
│   Groups: region                        │
│   ~100 Rows                             │
└───────────────────┬─────────────────────┘
                    │
┌───────────────────┴─────────────────────┐
│         SEQ_SCAN (sales)                │
│   Filters: region='Asia'               │
│   ~1000000 Rows                         │
└─────────────────────────────────────────┘
```

### 통계 및 최적화 Pragma

```sql
-- 테이블 통계 갱신 (쿼리 플래너 정확도 향상)
PRAGMA analyze;
ANALYZE;                  -- 동등한 표현
ANALYZE employees;        -- 특정 테이블만

-- 체크포인트 강제 실행 (WAL → 메인 파일 병합)
PRAGMA force_checkpoint;
CHECKPOINT;               -- 동등한 표현
CHECKPOINT analysis.db;   -- 특정 DB

-- 불필요한 공간 회수
VACUUM;
VACUUM employees;

-- 최적화 힌트 비활성화 (디버깅용)
PRAGMA disable_optimizer;
PRAGMA enable_optimizer;

-- 특정 최적화 규칙 제어
SET disabled_optimizers = 'filter_pushdown,column_pruning';
SET disabled_optimizers = '';  -- 초기화
```

### 확장 및 보안 Pragma

```sql
-- 설치된 확장 목록
PRAGMA installed_extensions;

-- 외부 확장 자동 설치/로드
PRAGMA autoinstall_known_extensions = true;
PRAGMA autoload_known_extensions = true;

-- 외부 파일 접근 제한 (샌드박스 모드)
SET enable_external_access = false;

-- HTTP 접근 제한
SET lock_configuration = true;  -- 연결 후 설정 변경 차단
```

### 개발/디버깅 Pragma

```sql
-- 쿼리 에코 출력 (실행되는 SQL 표시)
PRAGMA enable_print_progress_bar;

-- 더미 데이터 생성 (테스트용)
SELECT * FROM range(10);   -- 0~9
SELECT * FROM generate_series(1, 100, 5);  -- 1,6,11,...

-- 현재 날짜/시간 설정 (테스트용 고정)
SET timezone = 'Asia/Seoul';
SET TimeZone = 'UTC';

-- NULL 처리 방식
SET default_null_order = 'NULLS_LAST';

-- 숫자 그룹 구분자 (출력용)
-- .separator ',' (CLI 전용)
```
