# 20-1. 로컬 데이터 레이크 만들기

클라우드 없이도 로컬 파일 시스템 위에 완전한 데이터 레이크를 구성할 수 있다.

### 왜 세 계층으로 나누는가?

**메달리온 아키텍처(Medallion Architecture)**는 데이터를 품질 수준에 따라 Bronze, Silver, Gold 세 계층으로 분리해 관리하는 설계 패턴이다. 계층을 나누는 핵심 이유는 두 가지다. 첫째, 원본 데이터를 손상 없이 보존해 언제든 재처리를 가능하게 한다. 둘째, 각 계층에 명확한 책임을 부여해 파이프라인의 어느 단계에서 문제가 발생했는지 빠르게 파악할 수 있다.

| 계층 | 역할 | 처리 내용 |
|------|------|-----------|
| **Bronze** | 원본 보존 | 수집한 데이터를 가공 없이 Parquet으로 저장. 메타데이터(수집 일시, 출처 파일)만 추가한다. |
| **Silver** | 정제·검증 | 타입 변환, 중복 제거, 유효성 검증을 수행한다. 분석에 바로 쓸 수 있는 신뢰할 수 있는 데이터가 만들어진다. |
| **Gold** | 집계·마트 | 비즈니스 질문에 맞는 집계 테이블을 생성한다. 대시보드나 리포트가 이 계층을 직접 읽는다. |

원본 데이터를 수집(Bronze)하고, 정제(Silver)하고, 집계(Gold)하는 메달리온 아키텍처를 DuckDB와 Parquet으로 구현한다.

### 메달리온 아키텍처 디렉터리 구조

```
/data/lake/
├── bronze/                    # 원본 데이터 (가공 없이 저장)
│   ├── orders/
│   │   ├── year=2024/
│   │   │   ├── month=01/raw_20240101.parquet
│   │   │   └── month=02/raw_20240201.parquet
│   └── customers/
│       └── customers_raw.parquet
│
├── silver/                    # 정제 데이터 (타입 변환, 중복 제거, 유효성 검증)
│   ├── orders/
│   │   └── year=2024/month=01/orders_clean.parquet
│   └── customers/
│       └── customers_clean.parquet
│
└── gold/                      # 집계 데이터 (분석 바로 사용 가능한 마트)
    ├── daily_revenue.parquet
    ├── customer_segments.parquet
    └── product_performance.parquet
```

> **주의:** 파티션 키(`year=`, `month=`)는 나중에 변경하기 어렵다. 쿼리에서 가장 자주 사용하는 필터 기준을 파티션 키로 선택하고, 파티션 수가 지나치게 많아지지 않도록 설계한다. 일(day) 단위 파티션은 데이터가 충분히 클 때만 추가한다.

### Bronze 레이어: 원본 수집

Bronze 레이어의 역할은 단순하다. 데이터를 있는 그대로 받아 Parquet으로 변환하고, 언제 어디서 수집됐는지를 나타내는 메타데이터 컬럼을 추가하는 것이다. 원본을 변형하지 않기 때문에 Silver나 Gold 처리에 문제가 생겨도 언제든 Bronze부터 재처리할 수 있다.

```python
# pipeline/bronze.py - CSV/JSON 원본 데이터를 Parquet으로 수집
import duckdb
from pathlib import Path
from datetime import date, datetime
import logging

logger = logging.getLogger(__name__)

LAKE_ROOT = Path('/data/lake')


def ingest_csv_to_bronze(
    source_path: str,
    table_name: str,
    partition_date: date = None,
):
    """CSV 파일을 Bronze 레이어 Parquet으로 적재"""
    con = duckdb.connect()
    partition_date = partition_date or date.today()

    output_dir = (
        LAKE_ROOT / 'bronze' / table_name
        / f'year={partition_date.year}'
        / f'month={partition_date.month:02d}'
    )
    output_dir.mkdir(parents=True, exist_ok=True)
    output_path = output_dir / f'{table_name}_{partition_date.strftime("%Y%m%d")}.parquet'

    # 원본 그대로 저장 + 수집 메타데이터 컬럼 추가
    row_count = con.execute(f"""
        COPY (
            SELECT
                *,
                '{source_path}' AS _source_file,
                NOW()           AS _ingested_at,
                '{partition_date}' AS _partition_date
            FROM read_csv('{source_path}', auto_detect=true, ignore_errors=true)
        )
        TO '{output_path}' (FORMAT PARQUET, CODEC 'ZSTD')
    """).fetchone()

    logger.info(f"Bronze 적재 완료: {output_path}")
    return str(output_path)


def ingest_json_to_bronze(source_path: str, table_name: str, partition_date: date = None):
    """NDJSON 파일을 Bronze 레이어로 적재"""
    con = duckdb.connect()
    partition_date = partition_date or date.today()

    output_dir = (
        LAKE_ROOT / 'bronze' / table_name
        / f'year={partition_date.year}'
        / f'month={partition_date.month:02d}'
    )
    output_dir.mkdir(parents=True, exist_ok=True)
    output_path = output_dir / f'{table_name}_{partition_date.strftime("%Y%m%d")}.parquet'

    con.execute(f"""
        COPY (
            SELECT
                *,
                NOW() AS _ingested_at
            FROM read_json('{source_path}', auto_detect=true)
        )
        TO '{output_path}' (FORMAT PARQUET, CODEC 'ZSTD')
    """)
    return str(output_path)
```

### Silver 레이어: 정제 및 검증

Bronze에서 넘어온 데이터를 신뢰할 수 있는 상태로 만드는 단계다. `TRY_CAST`로 타입 변환에 실패한 값을 NULL로 처리하고, `QUALIFY ROW_NUMBER()`로 중복을 제거하며, 비즈니스 규칙에 맞지 않는 레코드를 걸러낸다. 이 단계를 거친 데이터가 Gold 마트의 재료가 된다.

```python
# pipeline/silver.py - Bronze 데이터 정제
import duckdb
from pathlib import Path
from datetime import date
import logging

logger = logging.getLogger(__name__)
LAKE_ROOT = Path('/data/lake')


def transform_orders_to_silver(partition_date: date):
    """주문 데이터 정제: 타입 변환, 중복 제거, 유효성 검증"""
    con = duckdb.connect()

    bronze_path = (
        LAKE_ROOT / 'bronze' / 'orders'
        / f'year={partition_date.year}'
        / f'month={partition_date.month:02d}'
        / '*.parquet'
    )
    output_dir = (
        LAKE_ROOT / 'silver' / 'orders'
        / f'year={partition_date.year}'
        / f'month={partition_date.month:02d}'
    )
    output_dir.mkdir(parents=True, exist_ok=True)
    output_path = output_dir / 'orders_clean.parquet'

    result = con.execute(f"""
        COPY (
            WITH raw AS (
                SELECT *
                FROM read_parquet('{bronze_path}')
            ),
            -- 타입 변환 및 정제
            typed AS (
                SELECT
                    TRY_CAST(order_id   AS BIGINT)    AS order_id,
                    TRY_CAST(customer_id AS BIGINT)   AS customer_id,
                    TRY_CAST(amount     AS DECIMAL(12,2)) AS amount,
                    TRY_CAST(order_date AS DATE)      AS order_date,
                    UPPER(TRIM(status))               AS status,
                    UPPER(TRIM(region))               AS region,
                    _ingested_at,
                    _source_file
                FROM raw
            ),
            -- 유효성 검증 (NULL 핵심 컬럼, 음수 금액 제거)
            valid AS (
                SELECT *
                FROM typed
                WHERE order_id   IS NOT NULL
                  AND customer_id IS NOT NULL
                  AND amount      > 0
                  AND order_date  IS NOT NULL
                  AND status IN ('COMPLETED', 'PENDING', 'CANCELLED', 'REFUNDED')
            ),
            -- 중복 제거 (order_id 기준 최신 레코드 유지)
            deduped AS (
                SELECT *
                FROM valid
                QUALIFY ROW_NUMBER() OVER (
                    PARTITION BY order_id
                    ORDER BY _ingested_at DESC
                ) = 1
            )
            SELECT * FROM deduped
        )
        TO '{output_path}' (FORMAT PARQUET, CODEC 'ZSTD')
    """)

    # 품질 리포트
    stats = con.execute(f"""
        SELECT
            COUNT(*) AS total_rows,
            COUNT(DISTINCT order_id) AS unique_orders,
            MIN(order_date) AS earliest_date,
            MAX(order_date) AS latest_date,
            SUM(amount) AS total_amount,
            COUNT(*) FILTER (WHERE status = 'COMPLETED') AS completed_count
        FROM read_parquet('{output_path}')
    """).fetchone()

    logger.info(
        f"Silver 변환 완료: {output_path} | "
        f"rows={stats[0]:,} unique_orders={stats[1]:,} "
        f"total_amount={stats[4]:,.2f}"
    )
    return str(output_path)
```

### Gold 레이어: 분석 마트 생성

Silver의 정제 데이터를 비즈니스 질문 단위로 집계하는 마지막 계층이다. 일별 매출, 고객 세분화처럼 대시보드나 리포트에서 반복적으로 사용되는 집계를 미리 계산해 두어 조회 속도를 극적으로 높인다.

```python
# pipeline/gold.py - Silver 데이터로 분석 마트 생성
import duckdb
from pathlib import Path
import logging

logger = logging.getLogger(__name__)
LAKE_ROOT = Path('/data/lake')


def build_daily_revenue_mart():
    """일별 매출 집계 마트 생성"""
    con = duckdb.connect()

    silver_orders = LAKE_ROOT / 'silver' / 'orders' / '**' / '*.parquet'
    silver_customers = LAKE_ROOT / 'silver' / 'customers' / '*.parquet'
    output_path = LAKE_ROOT / 'gold' / 'daily_revenue.parquet'
    output_path.parent.mkdir(parents=True, exist_ok=True)

    con.execute(f"""
        COPY (
            SELECT
                o.order_date,
                o.region,
                c.segment                            AS customer_segment,
                COUNT(*)                             AS order_count,
                COUNT(DISTINCT o.customer_id)        AS unique_customers,
                SUM(o.amount)                        AS revenue,
                AVG(o.amount)                        AS avg_order_value,
                MEDIAN(o.amount)                     AS median_order_value,
                SUM(SUM(o.amount)) OVER (
                    PARTITION BY o.region
                    ORDER BY o.order_date
                    ROWS UNBOUNDED PRECEDING
                )                                    AS cumulative_revenue,
                NOW()                                AS _built_at
            FROM read_parquet('{silver_orders}', union_by_name=true) o
            LEFT JOIN read_parquet('{silver_customers}') c
                ON o.customer_id = c.customer_id
            WHERE o.status = 'COMPLETED'
            GROUP BY o.order_date, o.region, c.segment
            ORDER BY o.order_date, o.region
        )
        TO '{output_path}' (FORMAT PARQUET, CODEC 'ZSTD')
    """)

    logger.info(f"Gold 마트 생성 완료: {output_path}")


def build_customer_segments():
    """고객 세분화 마트 생성 (RFM 분석)"""
    con = duckdb.connect()

    silver_orders = LAKE_ROOT / 'silver' / 'orders' / '**' / '*.parquet'
    output_path = LAKE_ROOT / 'gold' / 'customer_segments.parquet'

    con.execute(f"""
        COPY (
            WITH rfm AS (
                SELECT
                    customer_id,
                    MAX(order_date)                         AS last_order_date,
                    DATE_DIFF('day', MAX(order_date), TODAY()) AS recency_days,
                    COUNT(*)                                AS frequency,
                    SUM(amount)                             AS monetary
                FROM read_parquet('{silver_orders}', union_by_name=true)
                WHERE status = 'COMPLETED'
                GROUP BY customer_id
            ),
            scored AS (
                SELECT
                    *,
                    NTILE(5) OVER (ORDER BY recency_days ASC)  AS r_score,
                    NTILE(5) OVER (ORDER BY frequency DESC)    AS f_score,
                    NTILE(5) OVER (ORDER BY monetary DESC)     AS m_score
                FROM rfm
            )
            SELECT
                customer_id,
                last_order_date,
                recency_days,
                frequency,
                ROUND(monetary, 2) AS monetary,
                r_score, f_score, m_score,
                (r_score + f_score + m_score) AS rfm_total,
                CASE
                    WHEN r_score >= 4 AND f_score >= 4 THEN 'Champions'
                    WHEN r_score >= 3 AND f_score >= 3 THEN 'Loyal Customers'
                    WHEN r_score >= 4 AND f_score <= 2 THEN 'Recent Customers'
                    WHEN r_score <= 2 AND f_score >= 3 THEN 'At Risk'
                    WHEN r_score <= 2 AND f_score <= 2 THEN 'Lost'
                    ELSE 'Potential'
                END AS segment
            FROM scored
        )
        TO '{output_path}' (FORMAT PARQUET, CODEC 'ZSTD')
    """)

    logger.info(f"고객 세분화 마트 생성 완료: {output_path}")
```

### 파이프라인 오케스트레이터

Bronze, Silver, Gold 세 단계를 순서대로 실행하는 진입점이다. 각 단계가 완료된 후에야 다음 단계가 시작되므로 중간에 실패해도 원인 파악이 쉽다.

```python
# pipeline/orchestrator.py - 전체 파이프라인 실행
from datetime import date, timedelta
from bronze import ingest_csv_to_bronze
from silver import transform_orders_to_silver
from gold import build_daily_revenue_mart, build_customer_segments
import logging

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s %(levelname)s %(message)s'
)


def run_daily_pipeline(target_date: date = None):
    target_date = target_date or date.today() - timedelta(days=1)
    print(f"\n{'='*50}")
    print(f" 파이프라인 실행: {target_date}")
    print(f"{'='*50}")

    # 1. Bronze: 원본 수집
    print("\n[1/3] Bronze 레이어 수집 중...")
    ingest_csv_to_bronze(
        f'/raw/orders_{target_date.strftime("%Y%m%d")}.csv',
        'orders',
        target_date
    )

    # 2. Silver: 정제
    print("[2/3] Silver 레이어 정제 중...")
    transform_orders_to_silver(target_date)

    # 3. Gold: 집계 마트 갱신
    print("[3/3] Gold 레이어 마트 갱신 중...")
    build_daily_revenue_mart()
    build_customer_segments()

    print(f"\n파이프라인 완료: {target_date}\n")


if __name__ == '__main__':
    import sys
    if len(sys.argv) > 1:
        run_daily_pipeline(date.fromisoformat(sys.argv[1]))
    else:
        run_daily_pipeline()
```
