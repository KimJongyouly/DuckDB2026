# 16-1. PostgreSQL 연동

PostgreSQL은 실무에서 가장 널리 쓰이는 OLTP 데이터베이스 중 하나다. DuckDB의 `postgres` 확장을 사용하면 운영 PostgreSQL에 직접 접근해 분석 쿼리를 실행하거나, 필요한 데이터만 선별해 DuckDB 로컬 저장소로 가져올 수 있다.

### 확장 설치

```sql
INSTALL postgres; LOAD postgres;
-- 또는 자동 로드
SET autoinstall_known_extensions = true;
SET autoload_known_extensions = true;
```

> **팁:** `autoinstall_known_extensions`와 `autoload_known_extensions`를 모두 활성화하면 `ATTACH` 구문만 실행해도 DuckDB가 필요한 확장을 자동으로 설치하고 로드한다. 매번 `INSTALL`/`LOAD`를 명시하지 않아도 된다.

### ATTACH로 PostgreSQL 연결

```sql
-- 기본 연결
ATTACH 'host=localhost port=5432 dbname=mydb user=postgres password=secret'
AS pg (TYPE POSTGRES);

-- 환경 변수 / 연결 URL 형식도 지원
ATTACH 'postgresql://postgres:secret@localhost:5432/mydb'
AS pg (TYPE POSTGRES);

-- 읽기 전용 연결 (OLTP DB에 부하 없이 분석)
ATTACH 'postgresql://user:pass@prod-db:5432/production'
AS pg_prod (TYPE POSTGRES, READ_ONLY);

-- 스키마 지정
ATTACH 'postgresql://user:pass@host:5432/db'
AS pg (TYPE POSTGRES, SCHEMA 'analytics');

-- 연결 확인
SHOW DATABASES;
-- ┌──────────┐
-- │   name   │
-- ├──────────┤
-- │ memory   │
-- │ pg       │
-- └──────────┘
```

### 테이블 탐색과 쿼리

```sql
-- 연결된 DB의 테이블 목록
SHOW ALL TABLES IN pg;
SHOW TABLES IN pg.public;

-- 테이블 구조 확인
DESCRIBE pg.public.employees;

-- PostgreSQL 테이블 직접 쿼리
SELECT * FROM pg.public.employees LIMIT 10;

SELECT
    department,
    COUNT(*) AS headcount,
    AVG(salary) AS avg_salary
FROM pg.public.employees
GROUP BY department
ORDER BY avg_salary DESC;

-- 스키마 생략 (기본 스키마로 연결 시)
SELECT * FROM pg.employees LIMIT 10;

-- 조인: PostgreSQL 테이블 + DuckDB 로컬 테이블
SELECT
    e.name,
    e.department,
    d.budget
FROM pg.public.employees e
JOIN local_departments d ON e.department = d.name
WHERE e.salary > 80000;
```

### 데이터 가져오기 (Push-down)

```sql
-- PostgreSQL → DuckDB 로컬 테이블로 복사
CREATE TABLE employees_local AS
SELECT * FROM pg.public.employees;

-- 조건부 가져오기 (필요한 데이터만)
CREATE TABLE recent_orders AS
SELECT *
FROM pg.public.orders
WHERE order_date >= CURRENT_DATE - INTERVAL '90 days';

-- 집계 결과만 가져오기 (네트워크 부하 최소화)
CREATE TABLE dept_summary AS
SELECT
    department,
    COUNT(*) AS headcount,
    AVG(salary) AS avg_salary,
    SUM(salary) AS total_payroll
FROM pg.public.employees
GROUP BY department;
```

### Predicate / Projection Pushdown

DuckDB는 PostgreSQL 쿼리에 WHERE와 열 선택을 가능한 한 PostgreSQL 쪽에서 처리하도록 푸시다운한다.

```sql
-- DuckDB가 아래 쿼리를 PostgreSQL에 최적화해서 전달
SELECT name, salary
FROM pg.public.employees
WHERE department = 'Engineering'
  AND hire_date >= '2022-01-01';

-- 실제로 PostgreSQL에 전달되는 쿼리 (근사):
-- SELECT name, salary FROM employees
-- WHERE department = 'Engineering'
--   AND hire_date >= '2022-01-01'
-- → 네트워크로 전송되는 데이터 최소화

-- EXPLAIN으로 플랜 확인
EXPLAIN SELECT name FROM pg.public.employees WHERE salary > 80000;
```

### PostgreSQL → DuckDB 분석 파이프라인

```python
import duckdb

con = duckdb.connect('analytics.duckdb')
con.execute("LOAD postgres")

# PostgreSQL 연결
con.execute("""
    ATTACH 'postgresql://analyst:pass@prod-db:5432/production'
    AS pg (TYPE POSTGRES, READ_ONLY)
""")

# 어제 데이터만 가져와 로컬에 적재 (증분 ETL)
con.execute("""
    INSERT INTO daily_orders
    SELECT
        order_id,
        customer_id,
        order_date,
        total_amount,
        status
    FROM pg.public.orders
    WHERE order_date = CURRENT_DATE - INTERVAL '1 day'
""")

# 로컬 데이터 고속 분석
result = con.execute("""
    SELECT
        DATE_TRUNC('week', order_date) AS week,
        COUNT(*) AS orders,
        SUM(total_amount) AS revenue,
        COUNT(DISTINCT customer_id) AS unique_customers
    FROM daily_orders
    WHERE order_date >= CURRENT_DATE - INTERVAL '90 days'
    GROUP BY week
    ORDER BY week
""").df()

# PostgreSQL 연결 해제
con.execute("DETACH pg")
```

### DuckDB → PostgreSQL 쓰기

```sql
-- DuckDB 테이블을 PostgreSQL에 쓰기
INSERT INTO pg.public.analytics_summary
SELECT * FROM local_summary;

-- 테이블 생성 후 삽입
CREATE TABLE pg.public.duckdb_results AS
SELECT * FROM local_analysis_result;

-- UPSERT (ON CONFLICT)
INSERT INTO pg.public.daily_stats
SELECT date, region, revenue FROM local_stats
ON CONFLICT (date, region) DO UPDATE
SET revenue = EXCLUDED.revenue;
```

### 멀티 데이터베이스 조인

세 개 데이터베이스의 데이터를 DuckDB에서 한 번에 조인한다.

```sql
LOAD postgres; LOAD mysql; LOAD sqlite;

-- 각 DB 연결
ATTACH 'postgresql://user:pass@pg-host/crm' AS pg_crm (TYPE POSTGRES, READ_ONLY);
ATTACH 'mysql://user:pass@mysql-host/erp'   AS mysql_erp (TYPE MYSQL, READ_ONLY);
ATTACH 'local_cache.sqlite' AS cache (TYPE SQLITE);

-- 세 DB를 한 쿼리에서 조인
SELECT
    c.customer_id,
    c.customer_name,              -- PostgreSQL CRM
    c.email,
    o.order_total,                -- MySQL ERP
    o.order_date,
    ch.last_interaction_date      -- SQLite 캐시
FROM pg_crm.customers c
JOIN mysql_erp.orders o
    ON c.customer_id = o.customer_id
LEFT JOIN cache.customer_history ch
    ON c.customer_id = ch.customer_id
WHERE o.order_date >= '2024-01-01'
  AND c.region = 'Seoul'
ORDER BY o.order_total DESC
LIMIT 100;

-- 결과를 로컬 Parquet로 저장
COPY (
    SELECT
        c.customer_id,
        c.customer_name,
        SUM(o.order_total) AS lifetime_value,
        COUNT(o.id) AS order_count
    FROM pg_crm.customers c
    JOIN mysql_erp.orders o ON c.customer_id = o.customer_id
    GROUP BY c.customer_id, c.customer_name
) TO 'customer_ltv.parquet' (FORMAT PARQUET, COMPRESSION ZSTD);
```

### 변경 데이터 캡처 (CDC) 패턴

```python
import duckdb
from datetime import datetime

con = duckdb.connect('cdc_store.duckdb')
con.execute("LOAD postgres")
con.execute("""
    ATTACH 'postgresql://cdc_user:pass@prod-db/mydb'
    AS pg (TYPE POSTGRES, READ_ONLY)
""")

# 동기화 상태 테이블
con.execute("""
    CREATE TABLE IF NOT EXISTS sync_state (
        table_name VARCHAR PRIMARY KEY,
        last_synced_at TIMESTAMP,
        last_row_count BIGINT
    )
""")

def cdc_sync(table: str, watermark_col: str):
    """워터마크 기반 변경 데이터 동기화"""
    # 마지막 동기화 시점
    state = con.execute(f"""
        SELECT last_synced_at
        FROM sync_state
        WHERE table_name = '{table}'
    """).fetchone()

    last_ts = state[0] if state else '1970-01-01 00:00:00'

    # 변경된 행 수 확인
    new_count = con.execute(f"""
        SELECT COUNT(*) FROM pg.public.{table}
        WHERE {watermark_col} > '{last_ts}'::TIMESTAMP
    """).fetchone()[0]

    if new_count == 0:
        print(f"  [{table}] 변경 없음")
        return 0

    # 로컬 DuckDB에 UPSERT
    con.execute(f"""
        CREATE TABLE IF NOT EXISTS {table}_local AS
        SELECT * FROM pg.public.{table} WHERE false;
    """)

    con.execute(f"""
        INSERT OR REPLACE INTO {table}_local
        SELECT * FROM pg.public.{table}
        WHERE {watermark_col} > '{last_ts}'::TIMESTAMP
    """)

    # 동기화 상태 업데이트
    now = datetime.now().isoformat()
    con.execute(f"""
        INSERT OR REPLACE INTO sync_state VALUES
        ('{table}', '{now}'::TIMESTAMP, {new_count})
    """)

    print(f"  [{table}] {new_count:,}행 동기화 완료")
    return new_count

# 주기적 실행
for table, col in [
    ('orders', 'updated_at'),
    ('customers', 'updated_at'),
    ('products', 'updated_at'),
]:
    cdc_sync(table, col)

con.execute("DETACH pg")
```

### 대용량 마이그레이션 (배치)

```python
import duckdb

con = duckdb.connect('migrated.duckdb')
con.execute("LOAD postgres")
con.execute("""
    ATTACH 'postgresql://user:pass@source-db/legacy'
    AS pg_legacy (TYPE POSTGRES, READ_ONLY)
""")

def migrate_table_batched(table: str, pk: str, batch_size: int = 100_000):
    """대용량 테이블 배치 마이그레이션"""
    # 전체 행 수와 PK 범위 확인
    stats = con.execute(f"""
        SELECT COUNT(*), MIN({pk}), MAX({pk})
        FROM pg_legacy.public.{table}
    """).fetchone()
    total, pk_min, pk_max = stats
    print(f"[{table}] 전체 {total:,}행 마이그레이션 시작...")

    # 대상 테이블 생성
    con.execute(f"""
        CREATE TABLE IF NOT EXISTS {table} AS
        SELECT * FROM pg_legacy.public.{table} WHERE false
    """)

    # 배치 단위로 가져오기
    offset = pk_min
    migrated = 0

    while offset <= pk_max:
        next_offset = offset + batch_size

        con.execute(f"""
            INSERT INTO {table}
            SELECT * FROM pg_legacy.public.{table}
            WHERE {pk} >= {offset} AND {pk} < {next_offset}
        """)

        batch_rows = con.execute(f"""
            SELECT COUNT(*) FROM {table}
            WHERE {pk} >= {offset} AND {pk} < {next_offset}
        """).fetchone()[0]

        migrated += batch_rows
        pct = migrated * 100 / total if total > 0 else 0
        print(f"  {migrated:,}/{total:,}행 ({pct:.1f}%)", end='\r')

        offset = next_offset

    print(f"\n[{table}] 완료: {migrated:,}행")

    # 검증
    src_count = con.execute(
        f"SELECT COUNT(*) FROM pg_legacy.public.{table}"
    ).fetchone()[0]
    dst_count = con.execute(
        f"SELECT COUNT(*) FROM {table}"
    ).fetchone()[0]

    if src_count == dst_count:
        print(f"  검증 통과: {src_count:,}행 일치")
    else:
        print(f"  경고: 원본 {src_count:,} ≠ 대상 {dst_count:,}")

# 테이블별 마이그레이션
migrate_table_batched('orders', 'order_id')
migrate_table_batched('order_items', 'item_id')
migrate_table_batched('customers', 'customer_id')

# Parquet로 백업
con.execute("""
    COPY orders TO 'backup/orders.parquet' (FORMAT PARQUET, COMPRESSION ZSTD);
    COPY customers TO 'backup/customers.parquet' (FORMAT PARQUET, COMPRESSION ZSTD);
""")

con.execute("DETACH pg_legacy")
print("\n마이그레이션 완료")
```
