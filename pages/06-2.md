# 06-2. INSERT, UPDATE, DELETE, UPSERT

DML 명령은 테이블의 행 단위 데이터를 다룬다. INSERT는 새 행을 추가하고, UPDATE는 기존 행을 수정하며, DELETE는 조건에 맞는 행을 제거한다. DuckDB는 이 세 명령 모두에 RETURNING 절을 지원하여, 변경된 행을 즉시 조회하는 왕복 쿼리를 줄일 수 있다.

### INSERT

```sql
-- 단일 행 삽입
INSERT INTO employees (id, name, salary, hire_date)
VALUES (1, 'Alice', 85000, '2022-03-01');

-- 다중 행 삽입
INSERT INTO employees (id, name, salary)
VALUES
    (2, 'Bob',   72000),
    (3, 'Carol', 95000),
    (4, 'David', 68000);

-- 열 이름 생략 (모든 열 순서대로)
INSERT INTO employees
VALUES (5, 'Eve', 'eve@company.com', 'Engineering', 91000, '2023-06-15', true, NOW());

-- SELECT로 삽입 (다른 테이블에서 복사)
INSERT INTO employees_archive
SELECT * FROM employees WHERE hire_date < '2020-01-01';

-- 파일에서 직접 삽입
INSERT INTO sales
SELECT * FROM read_csv('new_sales.csv', header=true);

-- RETURNING: 삽입된 행 반환
INSERT INTO employees (name, salary)
VALUES ('Frank', 78000)
RETURNING id, name, created_at;
-- id | name  | created_at
-- ───┼───────┼────────────────────
-- 6  | Frank | 2024-03-15 14:30:00
```

> **팁:** RETURNING 절은 DuckDB가 삽입 직후 해당 행을 다시 읽어 결과를 돌려주는 기능이다. 자동 생성된 id나 DEFAULT 값이 어떻게 채워졌는지 확인할 때 별도의 SELECT 없이 한 번에 처리할 수 있다.

### UPDATE

```sql
-- 단일 열 업데이트
UPDATE employees SET salary = 90000 WHERE id = 1;

-- 다중 열 업데이트
UPDATE employees
SET
    salary    = salary * 1.1,
    active    = true,
    updated_at = CURRENT_TIMESTAMP
WHERE department = 'Engineering'
  AND hire_date < '2022-01-01';

-- 다른 테이블 참조 (FROM 절)
UPDATE employees
SET salary = e.salary * r.multiplier
FROM salary_adjustments r
WHERE employees.department = r.department
  AND employees.hire_date < r.cutoff_date;

-- 계산식 업데이트
UPDATE products
SET price = ROUND(price * 0.9, 2)    -- 10% 할인
WHERE category = 'seasonal';

-- RETURNING: 업데이트된 행 반환
UPDATE employees
SET salary = salary * 1.05
WHERE performance_score >= 4.0
RETURNING id, name, salary AS new_salary;
```

### DELETE

```sql
-- 조건부 삭제
DELETE FROM employees WHERE active = false;

-- 다른 테이블 참조 (USING)
DELETE FROM employees
USING terminated_list t
WHERE employees.id = t.employee_id
  AND t.termination_date < CURRENT_DATE;

-- 전체 삭제 (TRUNCATE가 더 빠름)
DELETE FROM temp_table;
TRUNCATE TABLE temp_table;

-- RETURNING: 삭제된 행 반환
DELETE FROM employees
WHERE hire_date < '2010-01-01'
RETURNING id, name, hire_date;
```

### UPSERT: INSERT OR UPDATE

실시간 집계나 멱등 로드 파이프라인에서는 같은 키의 데이터가 여러 번 들어올 수 있다. 그때마다 "이미 있는지 조회하고, 없으면 삽입, 있으면 업데이트"하는 로직을 애플리케이션에서 직접 구현하면 코드가 복잡해지고 경쟁 조건이 발생할 수 있다. UPSERT는 이 패턴을 단일 SQL 문으로 원자적으로 처리한다. DuckDB에서는 INSERT ... ON CONFLICT 구문으로 UPSERT를 표현하며, 충돌이 발생한 행의 새 값은 EXCLUDED 키워드로 참조한다.

UPSERT는 행이 없으면 삽입하고, 이미 있으면 업데이트한다. ON CONFLICT 절로 처리 방식을 지정한다.

```sql
-- 기본 UPSERT: 충돌 시 업데이트
INSERT INTO employees (id, name, salary)
VALUES (1, 'Alice', 95000)
ON CONFLICT (id) DO UPDATE
SET
    name   = EXCLUDED.name,    -- EXCLUDED: 삽입하려던 새 값
    salary = EXCLUDED.salary,
    updated_at = CURRENT_TIMESTAMP;

-- 충돌 시 무시 (아무것도 안 함)
INSERT INTO employees (id, name, salary)
VALUES (1, 'Alice', 95000)
ON CONFLICT (id) DO NOTHING;

-- 복합 키 충돌
INSERT INTO daily_stats (date, region, visits)
VALUES ('2024-03-15', 'Seoul', 1500)
ON CONFLICT (date, region) DO UPDATE
SET visits = daily_stats.visits + EXCLUDED.visits;  -- 누적 합산

-- 조건부 UPSERT: 더 높은 값만 업데이트
INSERT INTO scores (user_id, score)
VALUES (42, 9500)
ON CONFLICT (user_id) DO UPDATE
SET score = GREATEST(scores.score, EXCLUDED.score)
WHERE EXCLUDED.score > scores.score;

-- RETURNING과 결합
INSERT INTO page_views (page, views)
VALUES ('home', 100)
ON CONFLICT (page) DO UPDATE
SET views = page_views.views + EXCLUDED.views
RETURNING page, views;
```

```
UPSERT 흐름
────────────────────────────────────────────────
INSERT 시도
    │
    ├─ 충돌 없음 ──→ 새 행 삽입
    │
    └─ 충돌 발생
           │
           ├─ DO NOTHING ──→ 무시, 계속 진행
           │
           └─ DO UPDATE  ──→ EXCLUDED 값으로 업데이트
```

> **팁:** EXCLUDED는 "삽입하려 했던 새 값"을 가리킨다. 예를 들어 `SET salary = EXCLUDED.salary`는 기존 행의 salary를 방금 INSERT하려던 값으로 교체한다. 이 키워드를 활용하면 조건부 업데이트(더 높은 점수만 반영, 누적 합산 등)를 간결하게 표현할 수 있다.

### COPY: 대용량 데이터 입출력

INSERT ... SELECT로도 파일에서 데이터를 가져올 수 있지만, 파일을 통째로 테이블에 적재하거나 테이블 전체를 파일로 내보낼 때는 COPY가 더 간결하다. CSV, Parquet, JSON 모두 지원하며, 내보낼 때 쿼리 결과를 바로 파일로 저장할 수도 있다.

```sql
-- CSV 파일로 내보내기
COPY employees TO 'employees.csv' (HEADER, DELIMITER ',');
COPY (SELECT * FROM employees WHERE active) TO 'active.csv';

-- Parquet로 내보내기
COPY employees TO 'employees.parquet' (FORMAT PARQUET);

-- JSON으로 내보내기
COPY employees TO 'employees.json' (FORMAT JSON);

-- 파일에서 가져오기
COPY employees FROM 'employees.csv' (HEADER, DELIMITER ',');
COPY employees FROM 'data.parquet' (FORMAT PARQUET);

-- 와일드카드로 여러 파일
COPY sales FROM 'sales_*.parquet' (FORMAT PARQUET);
```
