# 21-2. Python ML 파이프라인과 통합

DuckDB에서 피처를 만들었다면, 이제 그 데이터를 ML 모델에 전달하는 방법을 알아본다. DuckDB는 scikit-learn, XGBoost, PyTorch 등 주요 ML 프레임워크와 자연스럽게 연결되며, 훈련/검증/테스트 분할, 교차 검증, 예측 결과 저장까지 전체 파이프라인을 일관된 방식으로 구성할 수 있다.

### 데이터 분할 전략

```python
import duckdb
import numpy as np

con = duckdb.connect('ml.db')

# 시간 기반 분할 (시계열 데이터에서 미래 누수 방지)
con.execute("""
    ALTER TABLE orders ADD COLUMN IF NOT EXISTS split VARCHAR;

    UPDATE orders SET split =
        CASE
            WHEN order_date < '2024-01-01' THEN 'train'
            WHEN order_date < '2024-07-01' THEN 'validation'
            ELSE 'test'
        END;
""")

# 비율 기반 무작위 분할 (독립 동일 분포 가정 시)
con.execute("""
    UPDATE orders SET split =
        CASE
            WHEN RANDOM() < 0.7 THEN 'train'
            WHEN RANDOM() < 0.85 THEN 'validation'  -- 0.7~0.85 = 15%
            ELSE 'test'                              -- 나머지 15%
        END;
""")

# 분할 결과 확인
con.execute("""
    SELECT
        split,
        COUNT(*) AS row_count,
        ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 1) AS pct
    FROM orders
    GROUP BY split
    ORDER BY split
""").df()
#    split   row_count   pct
# 0  test       15234   15.1
# 1  train      70891   70.2
# 2  validation 14875   14.7
```

### DuckDB → 학습 데이터 로딩 (Zero-Copy)

DuckDB의 `fetch_arrow_table()`은 Apache Arrow 형식으로 데이터를 반환한다. Arrow는 메모리 레이아웃이 NumPy 배열과 호환되도록 설계되어 있어, 데이터를 복사하지 않고 ML 프레임워크에 직접 전달할 수 있다. 즉, 수백만 행의 피처 데이터를 준비할 때 Pandas를 거치는 것보다 메모리를 적게 쓰고 변환 시간도 줄어든다. 데이터 준비에 드는 시간이 줄면 그만큼 모델 실험에 집중할 수 있는 여유가 생긴다.

```python
import duckdb
import numpy as np
import pyarrow as pa
from sklearn.model_selection import train_test_split

con = duckdb.connect('ml.db', config={'threads': 8, 'memory_limit': '16GB'})

FEATURE_COLS = [
    'order_count', 'active_months', 'category_diversity',
    'total_spend', 'avg_order_value', 'std_order_value',
    'customer_lifespan_days', 'days_since_last_order',
    'completion_rate', 'refund_rate',
    'spend_last_7d', 'spend_last_30d',
    'region_asia', 'region_europe', 'region_north_america',
    'month_sin', 'month_cos', 'dow_sin', 'dow_cos',
]
TARGET_COL = 'will_churn'

feature_list = ', '.join(FEATURE_COLS)

# Arrow로 로딩 (Zero-Copy, Pandas 변환 없음)
arrow_table = con.execute(f"""
    SELECT {feature_list}, {TARGET_COL}
    FROM features_customer fc
    JOIN churn_labels cl USING (customer_id)
    WHERE split = 'train'
      AND {' AND '.join(f'{c} IS NOT NULL' for c in FEATURE_COLS)}
""").fetch_arrow_table()

# NumPy 배열로 변환 (복사 없이)
X_train = arrow_table.select(FEATURE_COLS).to_pandas().values.astype(np.float32)
y_train = arrow_table.column(TARGET_COL).to_pylist()

print(f"훈련 데이터: X={X_train.shape}, y={len(y_train)}")
# 훈련 데이터: X=(70891, 19), y=70891
```

### scikit-learn 파이프라인 통합

```python
import duckdb
import numpy as np
import pandas as pd
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.compose import ColumnTransformer
from sklearn.ensemble import GradientBoostingClassifier
from sklearn.metrics import roc_auc_score, classification_report
import joblib

con = duckdb.connect('ml.db')


def load_split(split: str, feature_cols: list, target_col: str) -> tuple:
    """DuckDB에서 학습 데이터 로드"""
    cols = ', '.join(feature_cols + [target_col])
    df = con.execute(f"""
        SELECT {cols}
        FROM features_customer fc
        JOIN churn_labels cl USING (customer_id)
        WHERE fc.split = '{split}'
    """).df()
    return df[feature_cols].values, df[target_col].values


NUMERIC_FEATURES = [
    'order_count', 'total_spend', 'avg_order_value', 'std_order_value',
    'days_since_last_order', 'customer_lifespan_days',
    'completion_rate', 'spend_last_7d', 'spend_last_30d',
]
BINARY_FEATURES = [
    'region_asia', 'region_europe', 'region_north_america',
    'is_weekend',
]
ALL_FEATURES = NUMERIC_FEATURES + BINARY_FEATURES

X_train, y_train = load_split('train',      ALL_FEATURES, 'will_churn')
X_val,   y_val   = load_split('validation', ALL_FEATURES, 'will_churn')
X_test,  y_test  = load_split('test',       ALL_FEATURES, 'will_churn')

# 전처리 + 모델 파이프라인
preprocessor = ColumnTransformer([
    ('num', StandardScaler(),
     list(range(len(NUMERIC_FEATURES)))),          # 수치형: 표준화
    ('bin', 'passthrough',
     list(range(len(NUMERIC_FEATURES),
                len(ALL_FEATURES)))),              # 이진형: 그대로
])

pipeline = Pipeline([
    ('prep',  preprocessor),
    ('model', GradientBoostingClassifier(
        n_estimators=200,
        max_depth=5,
        learning_rate=0.05,
        subsample=0.8,
        random_state=42
    ))
])

pipeline.fit(X_train, y_train)

# 검증 성능
y_val_prob = pipeline.predict_proba(X_val)[:, 1]
print(f"Validation AUC: {roc_auc_score(y_val, y_val_prob):.4f}")
print(classification_report(y_val, pipeline.predict(X_val)))

# 모델 저장
joblib.dump(pipeline, 'models/churn_model_v1.pkl')
```

### XGBoost / LightGBM 통합

```python
import duckdb
import xgboost as xgb
import lightgbm as lgb
import numpy as np

con = duckdb.connect('ml.db', config={'threads': 8})


def load_as_dmatrix(split: str) -> xgb.DMatrix:
    """DuckDB → XGBoost DMatrix (Arrow 경유)"""
    arrow_tbl = con.execute(f"""
        SELECT
            order_count, total_spend, avg_order_value,
            days_since_last_order, completion_rate,
            spend_last_7d, spend_last_30d,
            will_churn AS label
        FROM features_customer fc
        JOIN churn_labels cl USING (customer_id)
        WHERE fc.split = '{split}'
    """).fetch_arrow_table()

    df = arrow_tbl.to_pandas()
    X = df.drop('label', axis=1).values.astype(np.float32)
    y = df['label'].values
    return xgb.DMatrix(X, label=y)


dtrain = load_as_dmatrix('train')
dval   = load_as_dmatrix('validation')
dtest  = load_as_dmatrix('test')

params = {
    'objective':   'binary:logistic',
    'eval_metric': 'auc',
    'max_depth':    6,
    'eta':          0.05,
    'subsample':    0.8,
    'colsample_bytree': 0.8,
    'tree_method': 'hist',     # CPU 고속 히스토그램 알고리즘
    'nthread':      8,
}

model = xgb.train(
    params,
    dtrain,
    num_boost_round=500,
    evals=[(dtrain, 'train'), (dval, 'val')],
    early_stopping_rounds=20,
    verbose_eval=50,
)

# 피처 중요도를 DuckDB에 저장
importance = model.get_score(importance_type='gain')
con.execute("""
    CREATE OR REPLACE TABLE feature_importance AS
    SELECT * FROM (VALUES
""" + ',\n'.join(f"('{k}', {v:.4f})" for k, v in
                  sorted(importance.items(), key=lambda x: -x[1])) + """
    ) t(feature_name, importance_gain)
""")

# 확인
con.execute("""
    SELECT feature_name, ROUND(importance_gain, 2) AS gain
    FROM feature_importance
    ORDER BY gain DESC
    LIMIT 10
""").df()
```

### 예측 결과를 DuckDB에 저장

```python
import duckdb
import joblib
import numpy as np
import pandas as pd
from datetime import datetime

con = duckdb.connect('ml.db')
pipeline = joblib.load('models/churn_model_v1.pkl')

ALL_FEATURES = [
    'order_count', 'total_spend', 'avg_order_value', 'std_order_value',
    'days_since_last_order', 'customer_lifespan_days',
    'completion_rate', 'spend_last_7d', 'spend_last_30d',
    'region_asia', 'region_europe', 'region_north_america', 'is_weekend',
]

# 전체 고객에 대해 예측 (배치 스코어링)
df = con.execute(f"""
    SELECT customer_id, {', '.join(ALL_FEATURES)}
    FROM features_customer
    WHERE {' AND '.join(f'{c} IS NOT NULL' for c in ALL_FEATURES)}
""").df()

X = df[ALL_FEATURES].values
proba = pipeline.predict_proba(X)[:, 1]
pred  = (proba >= 0.5).astype(int)

# 예측 결과 DataFrame
result_df = pd.DataFrame({
    'customer_id':    df['customer_id'],
    'churn_proba':    proba.round(4),
    'churn_pred':     pred,
    'risk_tier': pd.cut(
        proba,
        bins=[0, 0.2, 0.5, 0.8, 1.0],
        labels=['Low', 'Medium', 'High', 'Critical']
    ).astype(str),
    'scored_at': datetime.now().isoformat(),
})

# DuckDB에 저장
con.execute("DROP TABLE IF EXISTS churn_predictions")
con.register('result_df', result_df)
con.execute("""
    CREATE TABLE churn_predictions AS
    SELECT * FROM result_df
""")

# 리스크 티어별 분포 확인
con.execute("""
    SELECT
        risk_tier,
        COUNT(*)                             AS customer_count,
        ROUND(AVG(churn_proba) * 100, 1)    AS avg_churn_pct,
        ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 1) AS pct_of_total
    FROM churn_predictions
    GROUP BY risk_tier
    ORDER BY avg_churn_pct DESC
""").df()
```

### K-Fold 교차 검증

```python
import duckdb
import numpy as np
from sklearn.model_selection import StratifiedKFold
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import roc_auc_score

con = duckdb.connect('ml.db')

ALL_FEATURES = [
    'order_count', 'total_spend', 'avg_order_value',
    'days_since_last_order', 'completion_rate',
    'spend_last_7d', 'spend_last_30d',
]

# 전체 데이터 로드
arrow = con.execute(f"""
    SELECT {', '.join(ALL_FEATURES)}, will_churn
    FROM features_customer fc
    JOIN churn_labels cl USING (customer_id)
    WHERE {' AND '.join(f'{c} IS NOT NULL' for c in ALL_FEATURES)}
""").fetch_arrow_table()

df  = arrow.to_pandas()
X   = df[ALL_FEATURES].values.astype(np.float32)
y   = df['will_churn'].values

# 5-Fold 교차 검증
skf    = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
scores = []

for fold, (train_idx, val_idx) in enumerate(skf.split(X, y)):
    X_tr, X_vl = X[train_idx], X[val_idx]
    y_tr, y_vl = y[train_idx], y[val_idx]

    model = RandomForestClassifier(
        n_estimators=100, max_depth=8,
        n_jobs=-1, random_state=42
    )
    model.fit(X_tr, y_tr)

    auc = roc_auc_score(y_vl, model.predict_proba(X_vl)[:, 1])
    scores.append(auc)
    print(f"Fold {fold+1}: AUC = {auc:.4f}")

print(f"\nCV AUC: {np.mean(scores):.4f} ± {np.std(scores):.4f}")

# 교차 검증 결과 저장
con.execute(f"""
    CREATE OR REPLACE TABLE cv_results AS
    SELECT
        unnest([{', '.join(str(round(s, 4)) for s in scores)}]) AS fold_auc,
        generate_subscripts(
            [{', '.join(str(round(s, 4)) for s in scores)}], 1
        ) AS fold
""")
```

### PyTorch 통합 (딥러닝 피처 서빙)

```python
import duckdb
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader
import numpy as np

con = duckdb.connect('ml.db', config={'threads': 8})

FEATURES = [
    'order_count', 'total_spend', 'avg_order_value',
    'days_since_last_order', 'completion_rate',
    'spend_last_7d', 'spend_last_30d',
    'month_sin', 'month_cos', 'dow_sin', 'dow_cos',
]


class DuckDBDataset(Dataset):
    """DuckDB 쿼리 결과를 PyTorch Dataset으로 래핑"""

    def __init__(self, split: str, feature_cols: list, target_col: str):
        cols = ', '.join(feature_cols + [target_col])
        arrow = con.execute(f"""
            SELECT {cols}
            FROM features_customer fc
            JOIN churn_labels cl USING (customer_id)
            WHERE fc.split = '{split}'
              AND {' AND '.join(f'{c} IS NOT NULL' for c in feature_cols)}
        """).fetch_arrow_table()

        df = arrow.to_pandas()
        self.X = torch.tensor(
            df[feature_cols].values, dtype=torch.float32
        )
        self.y = torch.tensor(
            df[target_col].values, dtype=torch.float32
        )

    def __len__(self):
        return len(self.y)

    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]


class ChurnMLP(nn.Module):
    def __init__(self, input_dim: int):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, 128),
            nn.BatchNorm1d(128),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(128, 64),
            nn.BatchNorm1d(64),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(64, 1),
            nn.Sigmoid()
        )

    def forward(self, x):
        return self.net(x).squeeze(1)


# 데이터셋 및 로더 생성
train_ds = DuckDBDataset('train',      FEATURES, 'will_churn')
val_ds   = DuckDBDataset('validation', FEATURES, 'will_churn')

train_loader = DataLoader(train_ds, batch_size=512, shuffle=True,  num_workers=4)
val_loader   = DataLoader(val_ds,   batch_size=1024, shuffle=False, num_workers=4)

# 모델 훈련
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model  = ChurnMLP(len(FEATURES)).to(device)
opt    = torch.optim.Adam(model.parameters(), lr=1e-3, weight_decay=1e-4)
crit   = nn.BCELoss()

for epoch in range(20):
    model.train()
    train_loss = 0.0
    for X_batch, y_batch in train_loader:
        X_batch, y_batch = X_batch.to(device), y_batch.to(device)
        opt.zero_grad()
        loss = crit(model(X_batch), y_batch)
        loss.backward()
        opt.step()
        train_loss += loss.item()

    # 검증 AUC
    model.eval()
    all_proba, all_labels = [], []
    with torch.no_grad():
        for X_b, y_b in val_loader:
            all_proba.extend(model(X_b.to(device)).cpu().numpy())
            all_labels.extend(y_b.numpy())

    from sklearn.metrics import roc_auc_score
    auc = roc_auc_score(all_labels, all_proba)
    print(f"Epoch {epoch+1:2d} | Loss: {train_loss/len(train_loader):.4f} | Val AUC: {auc:.4f}")

# 예측 결과를 DuckDB에 저장
model.eval()
test_ds = DuckDBDataset('test', FEATURES, 'will_churn')
test_loader = DataLoader(test_ds, batch_size=1024)

probas = []
with torch.no_grad():
    for X_b, _ in test_loader:
        probas.extend(model(X_b.to(device)).cpu().numpy())

import pandas as pd
pred_df = pd.DataFrame({
    'churn_proba': np.array(probas).round(4),
    'model': 'mlp_v1'
})
con.register('pred_df', pred_df)
con.execute("CREATE OR REPLACE TABLE mlp_predictions AS SELECT * FROM pred_df")
```
