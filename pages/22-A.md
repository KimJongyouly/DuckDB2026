# 22-A. SQL 빠른 참조 (Cheat Sheet)

부록 A는 DuckDB SQL 문법 전반을 한눈에 파악하고 싶을 때, 또는 특정 구문의 정확한 형태를 빠르게 확인하고 싶을 때 참고하세요.

### 연결 및 기본 설정

이 섹션은 DuckDB CLI를 처음 시작하거나 환경 설정을 조정할 때 빠르게 참고하세요.

### SELECT 핵심 문법

이 섹션은 SELECT 쿼리의 기본 구조와 DuckDB 전용 문법을 확인할 때 빠르게 참고하세요.

```sql
-- 기본 구조 (실행 순서: FROM → WHERE → GROUP BY → HAVING → SELECT → ORDER → LIMIT)
SELECT   col1, col2, agg_func(col3) AS alias
FROM     table_name
WHERE    condition
GROUP BY col1, col2          -- GROUP BY ALL (비집계 컬럼 자동 포함)
HAVING   agg_func(col3) > 0
ORDER BY alias DESC NULLS LAST
LIMIT    100 OFFSET 0;

-- FROM-first 스타일 (DuckDB 전용)
FROM orders
SELECT order_id, amount
WHERE amount > 1000;

-- 직접 파일 쿼리
SELECT * FROM 'data.csv';
SELECT * FROM 'data.parquet';
SELECT * FROM 'data.json';
SELECT * FROM read_parquet('s3://bucket/path/*.parquet', hive_partitioning=true);
```

### JOIN

이 섹션은 JOIN 종류와 구문을 빠르게 확인해야 할 때 참고하세요.

```sql
INNER JOIN  t2 ON t1.id = t2.id        -- 양쪽 일치
LEFT  JOIN  t2 ON t1.id = t2.id        -- 왼쪽 전체 + 오른쪽 일치
RIGHT JOIN  t2 ON t1.id = t2.id        -- 오른쪽 전체 + 왼쪽 일치
FULL  JOIN  t2 ON t1.id = t2.id        -- 양쪽 전체
CROSS JOIN  t2                          -- 카테시안 곱
NATURAL JOIN t2                         -- 동일 이름 컬럼 자동 JOIN
JOIN  t2 USING (id, region)            -- 명시적 공통 키

-- ASOF JOIN (시계열 가장 가까운 이전 레코드)
SELECT * FROM events ASOF JOIN prices USING (ts);
```

### CTE (Common Table Expression)

이 섹션은 CTE와 재귀 쿼리 문법을 확인할 때 빠르게 참고하세요.

```sql
-- 기본 CTE
WITH cte1 AS (SELECT ...),
     cte2 AS (SELECT ... FROM cte1)
SELECT * FROM cte2;

-- 재귀 CTE
WITH RECURSIVE hier AS (
    SELECT id, parent_id, name, 0 AS depth
    FROM categories WHERE parent_id IS NULL
    UNION ALL
    SELECT c.id, c.parent_id, c.name, h.depth + 1
    FROM categories c JOIN hier h ON c.parent_id = h.id
)
SELECT * FROM hier ORDER BY depth, id;
```

### 윈도우 함수

이 섹션은 순위 계산, 누적 집계, 이전/이후 행 참조 등 윈도우 함수 문법을 빠르게 확인할 때 참고하세요.

```sql
func() OVER (
    PARTITION BY col          -- 그룹 구분
    ORDER BY col [ASC|DESC]   -- 정렬 기준
    ROWS|RANGE|GROUPS
        BETWEEN frame_start AND frame_end
    EXCLUDE CURRENT ROW | TIES | GROUP | NO OTHERS
)

-- frame_start / frame_end 옵션
UNBOUNDED PRECEDING
N PRECEDING
CURRENT ROW
N FOLLOWING
UNBOUNDED FOLLOWING

-- 순위 함수
ROW_NUMBER()   OVER (...)    -- 고유 순번 (동점 없음)
RANK()         OVER (...)    -- 동점 시 순위 건너뜀
DENSE_RANK()   OVER (...)    -- 동점 시 순위 유지
NTILE(n)       OVER (...)    -- n 등분 버킷
PERCENT_RANK() OVER (...)    -- 0~1 상대 순위

-- 오프셋 함수
LAG(col, n, default)  OVER (...) -- n행 이전 값
LEAD(col, n, default) OVER (...) -- n행 이후 값
FIRST_VALUE(col)      OVER (...) -- 프레임 첫 번째 값
LAST_VALUE(col)       OVER (...) -- 프레임 마지막 값
NTH_VALUE(col, n)     OVER (...) -- 프레임 n번째 값

-- WINDOW 절 재사용
SELECT ..., SUM(x) OVER w, AVG(x) OVER w
FROM t
WINDOW w AS (PARTITION BY region ORDER BY dt);

-- QUALIFY (윈도우 결과로 행 필터링)
SELECT * FROM t
QUALIFY ROW_NUMBER() OVER (PARTITION BY id ORDER BY ts DESC) = 1;
```

### 서브쿼리 & 스칼라

이 섹션은 서브쿼리와 EXISTS 패턴을 확인할 때 빠르게 참고하세요.

```sql
-- 스칼라 서브쿼리
SELECT *, (SELECT MAX(amount) FROM orders) AS global_max FROM orders;

-- IN / NOT IN
SELECT * FROM orders WHERE customer_id IN (SELECT id FROM vip_customers);

-- EXISTS / NOT EXISTS
SELECT * FROM customers c
WHERE EXISTS (SELECT 1 FROM orders o WHERE o.customer_id = c.id);

-- 스칼라 서브쿼리를 FROM에서 직접 사용 (DuckDB 지원)
SELECT avg_amount FROM (SELECT AVG(amount) AS avg_amount FROM orders);
```

### GROUP BY 확장

이 섹션은 ROLLUP, CUBE 등 고급 집계 구문을 빠르게 확인할 때 참고하세요.

```sql
GROUP BY ROLLUP  (a, b)         -- a, (a,b), () 소계
GROUP BY CUBE    (a, b)         -- 모든 조합 소계
GROUP BY GROUPING SETS ((a,b), (a), ())
GROUP BY ALL                    -- 비집계 컬럼 전체 자동 포함

-- GROUPING() 함수: 소계 행 여부 판별
SELECT a, b, SUM(x), GROUPING(a) AS is_a_subtotal
FROM t GROUP BY ROLLUP(a, b);
```

### SAMPLE

이 섹션은 대용량 테이블에서 샘플링 쿼리를 작성할 때 빠르게 참고하세요.

```sql
SELECT * FROM large_table USING SAMPLE 10%;          -- BERNOULLI (기본)
SELECT * FROM large_table USING SAMPLE 1000 ROWS;    -- 고정 행 수
SELECT * FROM large_table USING SAMPLE RESERVOIR(1000 ROWS);
SELECT * FROM large_table USING SAMPLE SYSTEM(10%);  -- 블록 단위
```

### UNNEST / VALUES

이 섹션은 배열을 행으로 펼치거나 인라인 데이터를 만들 때 빠르게 참고하세요.

```sql
-- 배열 펼치기
SELECT UNNEST([1, 2, 3]);
SELECT UNNEST(tags) AS tag FROM articles;

-- WITH ORDINALITY (인덱스 포함)
SELECT tag, idx
FROM articles, UNNEST(tags) WITH ORDINALITY AS t(tag, idx);

-- 다중 배열 병렬 펼치기
SELECT UNNEST(a) AS x, UNNEST(b) AS y FROM t;

-- VALUES 절
SELECT * FROM (VALUES (1, 'Alice'), (2, 'Bob')) AS t(id, name);
```

### DDL

이 섹션은 테이블 생성, 변경, 삭제 구문을 빠르게 확인할 때 참고하세요.

```sql
-- 테이블 생성
CREATE TABLE t (
    id      BIGINT PRIMARY KEY,
    name    VARCHAR NOT NULL,
    score   DECIMAL(10,2) DEFAULT 0,
    tags    VARCHAR[],
    meta    STRUCT(k VARCHAR, v INTEGER),
    created TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE (name),
    CHECK  (score >= 0)
);

CREATE TABLE t2 AS SELECT * FROM t WHERE false;   -- 구조만 복사
CREATE TABLE t3 AS SELECT * FROM 'data.parquet';  -- 파일에서 생성

-- 변경
ALTER TABLE t ADD    COLUMN memo VARCHAR;
ALTER TABLE t DROP   COLUMN memo;
ALTER TABLE t RENAME COLUMN name TO full_name;
ALTER TABLE t ALTER  COLUMN score SET DEFAULT 1.0;
ALTER TABLE t RENAME TO new_name;

-- 삭제
DROP TABLE IF EXISTS t;
DROP TABLE IF EXISTS t CASCADE;
TRUNCATE t;                          -- 데이터만 제거, 구조 유지
```

### DML

이 섹션은 INSERT, UPDATE, DELETE, UPSERT 등 데이터 조작 구문을 빠르게 확인할 때 참고하세요.

```sql
-- 삽입
INSERT INTO t VALUES (1, 'Alice', 9.5, ['a','b'], {'k':'x','v':1}, NOW());
INSERT INTO t SELECT * FROM staging;

-- UPSERT
INSERT INTO t VALUES (...) ON CONFLICT (id) DO UPDATE SET name = EXCLUDED.name;
INSERT INTO t VALUES (...) ON CONFLICT DO NOTHING;

-- 갱신
UPDATE t SET score = score * 1.1 WHERE status = 'active';
UPDATE t SET score = s.new_score FROM scores s WHERE t.id = s.id;

-- 삭제
DELETE FROM t WHERE created < '2020-01-01';

-- RETURNING
INSERT INTO t (...) VALUES (...) RETURNING id, created;
DELETE FROM t WHERE id = 1 RETURNING *;
```

### COPY

이 섹션은 파일에서 데이터를 읽거나 파일로 내보낼 때 빠르게 참고하세요.

```sql
-- CSV 읽기
COPY t FROM 'data.csv' (DELIMITER ',', HEADER true, NULLSTR 'NULL');

-- Parquet 읽기
COPY t FROM 'data.parquet' (FORMAT PARQUET);

-- CSV 쓰기
COPY t TO 'out.csv' (DELIMITER ',', HEADER true, COMPRESSION 'gzip');

-- Parquet 쓰기
COPY (SELECT * FROM t) TO 'out.parquet' (FORMAT PARQUET, CODEC 'ZSTD');

-- 파티셔닝 쓰기
COPY t TO 'dir/' (FORMAT PARQUET, PARTITION_BY (year, month), OVERWRITE_OR_IGNORE);
```

### VIEW / MACRO / SEQUENCE

이 섹션은 뷰, 매크로, 시퀀스 생성 구문을 빠르게 확인할 때 참고하세요.

```sql
-- 뷰
CREATE OR REPLACE VIEW v_active AS SELECT * FROM t WHERE status = 'active';
DROP VIEW IF EXISTS v_active;

-- 스칼라 매크로
CREATE OR REPLACE MACRO add(a, b) AS a + b;
SELECT add(1, 2);           -- → 3

-- 테이블 매크로
CREATE OR REPLACE MACRO top_n(tbl, n) AS TABLE
    SELECT * FROM query_table(tbl) LIMIT n;
SELECT * FROM top_n('orders', 5);

-- 시퀀스
CREATE SEQUENCE seq_id START 1 INCREMENT 1;
SELECT NEXTVAL('seq_id');
```
