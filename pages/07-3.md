# 07-3. CSV 쓰기와 옵션

### COPY TO: 기본 내보내기

```sql
-- 기본 CSV 내보내기
COPY (SELECT * FROM employees) TO 'employees.csv';

-- 헤더 포함 (기본값 true)
COPY employees TO 'employees.csv' (HEADER true, DELIMITER ',');

-- 헤더 없이
COPY employees TO 'employees_noheader.csv' (HEADER false);

-- 특정 열만 내보내기
COPY (
    SELECT id, name, department, salary
    FROM employees
    WHERE active = true
    ORDER BY department, name
) TO 'active_employees.csv' (HEADER true);

-- 조건부 내보내기
COPY (
    SELECT *
    FROM sales
    WHERE order_date BETWEEN '2024-01-01' AND '2024-12-31'
) TO 'sales_2024.csv';
```

### COPY TO 옵션 전체

```sql
COPY employees TO 'output.csv' (
    FORMAT      CSV,           -- 형식 명시 (기본값)
    HEADER      true,          -- 헤더 행 포함
    DELIMITER   ',',           -- 구분자
    QUOTE       '"',           -- 인용 문자
    ESCAPE      '"',           -- 이스케이프 문자
    NULLSTR     '',            -- NULL을 표현할 문자열
    DATEFORMAT  '%Y-%m-%d',    -- 날짜 출력 형식
    TIMESTAMPFORMAT '%Y-%m-%d %H:%M:%S',
    ENCODING    'UTF-8',       -- 인코딩
    COMPRESSION 'gzip',        -- 압축 (gzip, zstd, snappy)
    OVERWRITE   true           -- 기존 파일 덮어쓰기
);
```

### 구분자와 인용 문자 커스터마이징

```sql
-- TSV 출력
COPY employees TO 'employees.tsv' (DELIMITER '\t', HEADER true);

-- 세미콜론 구분 (유럽식)
COPY employees TO 'employees_eu.csv' (DELIMITER ';', HEADER true);

-- 파이프 구분
COPY employees TO 'employees_pipe.csv' (DELIMITER '|');

-- NULL을 특정 문자열로 표현
COPY employees TO 'output.csv' (NULLSTR 'N/A');

-- 인용 문자 변경 (작은따옴표)
COPY employees TO 'output.csv' (QUOTE '''', ESCAPE '''');
```

### 압축 출력

```sql
-- gzip 압축
COPY sales TO 'sales.csv.gz' (FORMAT CSV, HEADER true, COMPRESSION GZIP);

-- zstd 압축 (더 높은 압축률, 빠른 속도)
COPY sales TO 'sales.csv.zst' (COMPRESSION ZSTD);

-- 확장자로 자동 감지
COPY sales TO 'sales.csv.gz';   -- .gz → gzip 자동 적용
COPY sales TO 'sales.csv.zst';  -- .zst → zstd 자동 적용

-- 압축된 파일 읽기도 동일하게
SELECT * FROM 'sales.csv.gz';
SELECT * FROM read_csv('data.csv.zst');
```

### 파티션 분할 출력

```sql
-- 열 값으로 파일 분할
COPY (SELECT * FROM sales) TO 'output/' (
    FORMAT           CSV,
    HEADER           true,
    PARTITION_BY     (year, region),  -- 이 열 기준으로 디렉터리 생성
    OVERWRITE_OR_IGNORE true
);
-- 결과:
-- output/year=2023/region=Seoul/data_0.csv
-- output/year=2023/region=Busan/data_0.csv
-- output/year=2024/region=Seoul/data_0.csv
-- ...

-- 파티션된 파일 다시 읽기
SELECT * FROM read_csv('output/**/*.csv',
    hive_partitioning = true  -- 디렉터리명을 열로 자동 파싱
);
-- year | region | id | amount | ...
```

### 대용량 파일 분할

```sql
-- 행 수 기준 분할 (DuckDB 직접 지원 없음, SQL로 처리)
-- 방법: 행 번호로 배치 분할
CREATE OR REPLACE MACRO export_batch(tbl, batch_size, batch_num) AS TABLE
    SELECT * FROM query_table(tbl)
    LIMIT batch_size OFFSET (batch_num * batch_size);

-- 또는 NTILE 활용
WITH batched AS (
    SELECT *, NTILE(10) OVER (ORDER BY id) AS batch
    FROM large_table
)
COPY (SELECT * EXCLUDE batch FROM batched WHERE batch = 1)
TO 'output_part1.csv';

-- Python에서 배치 처리
```

```python
import duckdb

con = duckdb.connect('data.duckdb')
batch_size = 100_000

total = con.execute("SELECT COUNT(*) FROM large_table").fetchone()[0]
num_batches = (total + batch_size - 1) // batch_size

for i in range(num_batches):
    con.execute(f"""
        COPY (
            SELECT * FROM large_table
            LIMIT {batch_size} OFFSET {i * batch_size}
        ) TO 'output/part_{i:04d}.csv' (HEADER true)
    """)
    print(f"배치 {i+1}/{num_batches} 완료")
```

### 날짜/시간 포맷 제어

```sql
-- 날짜 형식 지정
COPY (
    SELECT
        id,
        STRFTIME(order_date, '%d/%m/%Y') AS order_date,  -- 미리 포맷
        amount
    FROM orders
) TO 'orders_eu.csv';

-- 또는 COPY 옵션으로
COPY orders TO 'orders.csv' (
    HEADER       true,
    DATEFORMAT   '%Y/%m/%d',
    TIMESTAMPFORMAT '%Y-%m-%d %H:%M:%S'
);
```

### Excel 호환 CSV

```sql
-- Excel에서 열리는 CSV (BOM 포함 UTF-8, 세미콜론 구분)
-- BOM 추가는 DuckDB 직접 지원 없음 → COPY 후 후처리
COPY (SELECT * FROM employees) TO 'for_excel.csv' (
    DELIMITER ';',
    HEADER    true,
    ENCODING  'UTF-8'
);

-- Python으로 BOM 추가
```

```python
import duckdb

con = duckdb.connect()
con.execute("""
    COPY (SELECT * FROM employees)
    TO 'temp.csv' (DELIMITER ';', HEADER true)
""")

# BOM 추가
with open('temp.csv', 'rb') as f:
    content = f.read()
with open('for_excel.csv', 'wb') as f:
    f.write(b'\xef\xbb\xbf' + content)  # UTF-8 BOM
```

```

│ 탭 구분            │ delim='\t' 또는 .tsv 확장자           │
│ 여러 파일 합치기   │ 'dir/*.csv' 와일드카드                │
│ 파일 출처 추적     │ filename=true                         │
└────────────────────┴───────────────────────────────────────┘

┌────────────────────────────────────────────────────────────┐
│                   CSV 쓰기 옵션 선택                        │
├────────────────────┬───────────────────────────────────────┤
│ 상황               │ 권장 옵션                              │
├────────────────────┼───────────────────────────────────────┤
```
