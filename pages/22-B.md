# 22-B. 주요 함수 레퍼런스

부록 B는 쿼리 작성 중 특정 함수의 정확한 이름과 인수를 확인하고 싶을 때 카테고리별로 찾아보세요. 자주 쓰이는 핵심 함수는 **볼드**로 강조되어 있습니다.

### 집계 함수

```sql
**COUNT(*)**                          -- 전체 행 수
**COUNT(col)**                        -- NULL 제외 행 수
**COUNT(DISTINCT col)**               -- 고유 값 수
APPROX_COUNT_DISTINCT(col)        -- 근사 고유 값 수 (HLL)
**SUM(col)**
**AVG(col)**
**MIN(col) / MAX(col)**
**MEDIAN(col)**
QUANTILE_CONT(col, 0.25)          -- 연속형 사분위수
QUANTILE_DISC(col, [0.25, 0.75])  -- 이산형 다중 분위수
STDDEV(col)   / VARIANCE(col)
STDDEV_POP(col) / VAR_POP(col)
CORR(x, y)                        -- 상관계수
COVAR_POP(x, y)                   -- 공분산
REGR_SLOPE(y, x)                  -- 선형 회귀 기울기
REGR_INTERCEPT(y, x)              -- 선형 회귀 절편

-- 조건부 집계
**COUNT(col) FILTER (WHERE cond)**
**SUM(col)   FILTER (WHERE cond)**

-- 리스트 집계
**LIST(col)**                         -- 배열로 수집
LIST(col ORDER BY ts)             -- 순서 지정
**STRING_AGG(col, ', ')**             -- 문자열 연결
STRING_AGG(col, ', ' ORDER BY col)
ARG_MIN(val, key)                 -- key 최솟값 행의 val
ARG_MAX(val, key)                 -- key 최댓값 행의 val
```

### 문자열 함수

```sql
LENGTH(s) / LEN(s)
**UPPER(s) / LOWER(s)**
**TRIM(s) / LTRIM(s) / RTRIM(s)**
LPAD(s, n, pad) / RPAD(s, n, pad)
**SUBSTR(s, start, len)**            -- 1-based
SUBSTRING(s FROM start FOR len)
LEFT(s, n) / RIGHT(s, n)
**CONCAT(s1, s2, ...) / s1 || s2**
**REPLACE(s, from, to)**
SPLIT_PART(s, delim, n)          -- n번째 부분 (1-based)
SPLIT(s, delim)                  -- VARCHAR[] 반환
STARTS_WITH(s, prefix)
ENDS_WITH(s, suffix)
**CONTAINS(s, sub)**
POSITION(sub IN s) / STRPOS(s, sub)
REPEAT(s, n)
REVERSE(s)
PRINTF('%05d', 42)               -- 형식 문자열
FORMAT('{} + {} = {}', 1, 2, 3)

-- 정규식
**REGEXP_MATCHES(s, pattern)**       -- 매칭 여부 BOOLEAN
**REGEXP_EXTRACT(s, pattern, grp)**  -- 그룹 추출 (grp 기본 0)
REGEXP_EXTRACT_ALL(s, pattern)   -- 모든 매칭 VARCHAR[]
REGEXP_REPLACE(s, pattern, repl) -- 치환
REGEXP_SPLIT_TO_ARRAY(s, pattern)

-- 유사도
LEVENSHTEIN(s1, s2)              -- 편집 거리
JARO_SIMILARITY(s1, s2)          -- 0~1 유사도
JACCARD(s1, s2)

-- 인코딩
BASE64(blob) / UNBASE64(s)
MD5(s)
SHA256(blob)
```

### 날짜 / 시간 함수

```sql
-- 현재 시각
**NOW() / CURRENT_TIMESTAMP**
**CURRENT_DATE**
CURRENT_TIME

-- 생성
MAKE_DATE(year, month, day)
MAKE_TIMESTAMP(y, mo, d, h, mi, s)
MAKE_INTERVAL(years:=1, months:=2, days:=3)

-- 추출
**YEAR(d) / MONTH(d) / DAY(d)**
HOUR(t) / MINUTE(t) / SECOND(t)
DAYOFWEEK(d)   -- 0(일)~6(토) 또는 1~7 (설정에 따라)
DAYOFYEAR(d)
WEEKOFYEAR(d)
QUARTER(d)
EPOCH(ts)      -- UNIX 타임스탬프 (초)

-- 절단
**DATE_TRUNC('month', d)**     -- 월 첫날 00:00:00
DATE_TRUNC('week',  d)
DATE_TRUNC('hour',  ts)

-- 차이
**DATE_DIFF('day',   d1, d2)**   -- d2 - d1 (일 수)
DATE_DIFF('month', d1, d2)
AGE(d2, d1)                  -- INTERVAL 반환
DATEDIFF(d1, d2)             -- 일 수 차이

-- 가감
d + INTERVAL '7 days'
d + INTERVAL '1 month 3 days'
DATE_ADD(d, INTERVAL '1 year')

-- 형식 변환
**STRFTIME(d, '%Y-%m-%d')**          -- 날짜 → 문자열
STRPTIME('2024-01-15', '%Y-%m-%d') -- 문자열 → TIMESTAMP
TO_DATE('2024/01/15', '%Y/%m/%d')  -- 문자열 → DATE
```

### JSON 함수

```sql
-- 접근
**col -> 'key'**           -- JSON 반환
**col ->> 'key'**          -- 텍스트 반환
col -> 0               -- 배열 인덱스 (0-based)
**JSON_EXTRACT(col, '$.key.subkey')**
JSON_EXTRACT_STRING(col, '$.key')

-- 정보
JSON_TYPE(col)         -- 'object' | 'array' | 'string' | ...
JSON_KEYS(col)         -- 최상위 키 배열
JSON_STRUCTURE(col)    -- 구조 추론
JSON_VALID(col)        -- 유효성 검사
JSON_ARRAY_LENGTH(col) -- 배열 길이

-- 생성
JSON_OBJECT('k1', v1, 'k2', v2)
JSON_ARRAY(v1, v2, v3)
**JSON_GROUP_ARRAY(col)**             -- 그룹 → JSON 배열
**JSON_GROUP_OBJECT(key_col, val_col)** -- 그룹 → JSON 객체

-- 수정
JSON_MERGE_PATCH(base, patch)     -- RFC 7396 병합
JSON_SET(col, '$.key', value)
JSON_REMOVE(col, '$.key')
```

### LIST / ARRAY 함수

```sql
-- 생성
[1, 2, 3]
LIST_VALUE(1, 2, 3)
RANGE(start, stop, step)         -- [start, start+step, ...)
GENERATE_SERIES(start, stop)     -- start~stop 포함

-- 접근
**arr[1]**                           -- 1-based 인덱스
arr[-1]                          -- 마지막 요소
LIST_EXTRACT(arr, 2)

-- 정보
**LEN(arr) / ARRAY_LENGTH(arr)**
**LIST_CONTAINS(arr, val)**
LIST_POSITION(arr, val)          -- 1-based, 없으면 NULL

-- 변환
LIST_SORT(arr)
LIST_REVERSE(arr)
LIST_DISTINCT(arr)               -- 중복 제거
FLATTEN(arr)                     -- 중첩 배열 펼치기
**LIST_CONCAT(a1, a2) / a1 || a2**
LIST_PREPEND(val, arr)
LIST_APPEND(arr, val)
LIST_SLICE(arr, start, stop)     -- 슬라이싱

-- 고차 함수
**LIST_APPLY(arr, x -> x * 2)**     -- map
**LIST_FILTER(arr, x -> x > 0)**    -- filter
LIST_REDUCE(arr, (a,b) -> a+b)  -- reduce
LIST_TRANSFORM(arr, x -> UPPER(x))
LIST_AGGREGATE(arr, 'sum')       -- 내장 집계 적용

-- 집계로 배열 생성
LIST(col ORDER BY ts)
LIST(col) FILTER (WHERE cond)
```

### STRUCT / MAP 함수

```sql
-- STRUCT 생성 및 접근
{'name': 'Alice', 'age': 30}
**row_value.name**                   -- 점 표기법
row_value['name']                -- 브래킷 표기법
STRUCT_PACK(name := 'Alice', age := 30)
STRUCT_EXTRACT(s, 'name')

-- STRUCT 정보
STRUCT_KEYS(s)                   -- 키 배열

-- MAP 생성 및 접근
MAP(['k1','k2'], [v1, v2])
MAP_FROM_ENTRIES([{'key':k,'value':v}, ...])
**m['key']**                         -- 값 접근
MAP_KEYS(m) / MAP_VALUES(m)
MAP_ENTRIES(m)                   -- STRUCT 배열
CARDINALITY(m)                   -- 항목 수
ELEMENT_AT(m, 'key')
MAP_CONCAT(m1, m2)               -- 두 MAP 병합
```

### 조건 표현식

```sql
CASE WHEN cond1 THEN r1
     WHEN cond2 THEN r2
     ELSE r_default END

CASE expr
     WHEN v1 THEN r1
     WHEN v2 THEN r2
     ELSE r_default END

IIF(cond, true_val, false_val)
**IFNULL(val, default)**             -- val이 NULL이면 default
**NULLIF(a, b)**                     -- a = b이면 NULL, 아니면 a
**COALESCE(v1, v2, v3, ...)**       -- 첫 번째 non-NULL

-- NULL 안전 비교
val IS NULL
val IS NOT NULL
val IS NOT DISTINCT FROM val2    -- NULL = NULL → TRUE
```
