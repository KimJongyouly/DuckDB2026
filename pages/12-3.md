# 12-3. 관계형 API (Relational API)

관계형 API는 SQL 문자열 대신 Python 메서드 체이닝으로 쿼리를 구성하는 방식이다. 자동완성과 타입 힌트를 활용할 수 있고, 쿼리를 동적으로 조합하기 쉽다.

### Relation 객체 생성

```python
import duckdb

con = duckdb.connect()
con.execute("""
    CREATE TABLE employees AS
    SELECT * FROM (VALUES
        (1, 'Alice', 'Engineering', 95000, true),
        (2, 'Bob',   'Sales',       72000, true),
        (3, 'Carol', 'Engineering', 105000, true),
        (4, 'David', 'Marketing',   68000, false),
        (5, 'Eve',   'Engineering', 88000, true)
    ) t(id, name, dept, salary, active)
""")

# Relation 생성
rel = con.table('employees')          # 테이블에서
rel = con.sql("SELECT * FROM employees")  # SQL에서
rel = con.read_csv('data.csv')        # CSV 파일에서
rel = con.read_parquet('data.parquet')  # Parquet에서
rel = con.read_json('data.json')      # JSON에서

print(type(rel))  # <class 'duckdb.DuckDBPyRelation'>
```

### Relation 메서드 체이닝

```python
rel = con.table('employees')

# filter (WHERE)
active = rel.filter("active = true")
high_salary = rel.filter("salary > 80000 AND dept = 'Engineering'")

# select (SELECT)
names = rel.select("name, dept, salary")
with_bonus = rel.select("*, salary * 0.1 AS bonus")

# 체이닝
result = (
    con.table('employees')
    .filter("active = true")
    .select("dept, name, salary")
    .filter("salary > 70000")
)

# order (ORDER BY)
sorted_rel = rel.order("salary DESC")
sorted_rel = rel.order("dept ASC, salary DESC")

# limit / offset
top5 = rel.order("salary DESC").limit(5)
page2 = rel.order("id").limit(10, offset=10)

# 결과 출력
result.show()       # 터미널에 출력
df = result.df()    # Pandas DataFrame
pl = result.pl()    # Polars DataFrame
arr = result.arrow()  # PyArrow Table
rows = result.fetchall()  # 리스트 of 튜플
```

### 집계와 그룹화

```python
rel = con.table('employees')

# aggregate (GROUP BY + 집계)
dept_stats = rel.aggregate("""
    dept,
    COUNT(*)    AS headcount,
    AVG(salary) AS avg_salary,
    MAX(salary) AS max_salary
""", group_expr="dept")

dept_stats.show()
# dept         | headcount | avg_salary | max_salary
# ─────────────┼───────────┼────────────┼───────────
# Engineering  | 3         | 96000.0    | 105000
# Marketing    | 1         | 68000.0    | 68000
# Sales        | 1         | 72000.0    | 72000

# 집계 단축 메서드
rel.count("*")                     # 행 수
rel.sum("salary")                  # 합계
rel.mean("salary")                 # 평균
rel.median("salary")               # 중앙값
rel.min("salary")                  # 최솟값
rel.max("salary")                  # 최댓값
rel.std("salary")                  # 표준편차
rel.var("salary")                  # 분산
rel.unique("dept")                 # DISTINCT
rel.value_counts("dept")           # 값별 카운트
```

### JOIN

```python
employees = con.table('employees')
con.execute("""
    CREATE TABLE projects AS VALUES
        (1, 'DuckDB 분석',  1),
        (2, 'ML 파이프라인', 3),
        (3, '대시보드',      1),
        (4, '데이터 정제',   5)
    AS t(project_id, project_name, lead_id)
""")
projects = con.table('projects')

# INNER JOIN
joined = employees.join(
    projects,
    condition="employees.id = projects.lead_id"
)
joined.select("name, dept, project_name").show()

# join_type 지정
left_joined = employees.join(
    projects,
    condition="employees.id = projects.lead_id",
    how='left'   # 'inner', 'left', 'right', 'outer', 'cross', 'semi', 'anti'
)
```

### 동적 쿼리 조합

관계형 API의 가장 큰 장점 — 조건에 따라 쿼리를 조립할 수 있다.

```python
import duckdb

con = duckdb.connect('analytics.duckdb')

def query_employees(
    dept=None,
    min_salary=None,
    active_only=True,
    order_by='salary',
    limit=100
):
    rel = con.table('employees')

    if active_only:
        rel = rel.filter("active = true")
    if dept:
        rel = rel.filter(f"dept = '{dept}'")
    if min_salary:
        rel = rel.filter(f"salary >= {min_salary}")

    rel = rel.select("id, name, dept, salary")
    rel = rel.order(f"{order_by} DESC")
    rel = rel.limit(limit)

    return rel.df()

# 동적으로 조건 조합
df1 = query_employees(dept='Engineering', min_salary=90000)
df2 = query_employees(active_only=False, order_by='id')
df3 = query_employees(limit=10)
```

### Relation을 SQL에서 활용

```python
# Relation을 SQL 쿼리 안에서 뷰처럼 사용
filtered = con.table('employees').filter("active = true")
filtered.create_view('active_employees')  # 임시 뷰 생성

# 이후 SQL에서 뷰 이름으로 참조
result = con.execute("""
    SELECT dept, AVG(salary) AS avg_sal
    FROM active_employees
    GROUP BY dept
""").df()

# 테이블로 저장
filtered.create("active_employees_table")  # 실제 테이블 생성
filtered.insert_into("archive_employees")  # 기존 테이블에 삽입

# Relation → SQL 쿼리 문자열 확인 (디버깅)
print(rel.filter("salary > 80000").select("name, salary").query("r", "SELECT"))
```
