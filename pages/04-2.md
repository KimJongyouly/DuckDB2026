# 04-2. LIST, STRUCT, MAP, UNION

DuckDB는 중첩 타입을 기본으로 지원한다. JSON이나 반정형 데이터를 별도 파싱 없이 SQL에서 직접 다룰 수 있다. 이 타입들을 사용하면 정규화 비용 없이 복잡한 데이터 구조를 하나의 열에 자연스럽게 표현할 수 있다.

### LIST: 순서 있는 동일 타입 배열

하나의 레코드에 여러 값을 묶어서 저장해야 할 때 LIST를 사용하면 별도의 조인 테이블 없이 데이터를 한 행에 담을 수 있다. 게시물의 태그 목록, 사용자의 방문 이력처럼 "항목 수가 유동적인 반복 데이터"에 적합하다.

```sql
-- LIST 생성
SELECT [1, 2, 3, 4, 5];                    -- INTEGER[]
SELECT ['apple', 'banana', 'cherry'];       -- VARCHAR[]
SELECT [DATE '2024-01-01', DATE '2024-06-01'];  -- DATE[]

-- 테이블 정의
CREATE TABLE articles (
    id      INTEGER,
    title   VARCHAR,
    tags    VARCHAR[],    -- VARCHAR 리스트
    scores  DOUBLE[]
);

INSERT INTO articles VALUES
    (1, 'DuckDB 입문', ['database', 'sql', 'analytics'], [9.2, 8.8, 9.5]),
    (2, 'Python 데이터 분석', ['python', 'pandas'], [8.0, 9.1]);

-- 원소 접근 (1-based index)
SELECT tags[1] AS first_tag FROM articles;     -- 'database', 'python'

-- 리스트 함수
SELECT
    len(tags) AS tag_count,                     -- 원소 수
    array_length(tags) AS also_count,           -- 동일
    list_contains(tags, 'sql') AS has_sql,      -- 포함 여부
    list_position(tags, 'pandas'),              -- 위치 (없으면 NULL)
    list_slice(tags, 1, 2) AS first_two,        -- 슬라이스
    list_sort(tags) AS sorted_tags,             -- 정렬
    list_distinct(tags) AS unique_tags,         -- 중복 제거
    list_reverse(tags) AS reversed,             -- 역순
    list_append(tags, 'new') AS appended,       -- 원소 추가
    list_prepend('first', tags) AS prepended,   -- 앞에 추가
    list_concat(tags, scores::VARCHAR[]) AS all -- 리스트 합치기
FROM articles;

-- 리스트 집계
SELECT list(name ORDER BY salary DESC) AS ranked_names
FROM employees;

-- UNNEST로 행으로 변환
SELECT id, UNNEST(tags) AS tag
FROM articles;

-- 리스트 생성 집계
SELECT department, list(name) AS members
FROM employees
GROUP BY department;
```

> **팁:** LIST의 인덱스는 1부터 시작한다(0-based가 아님). `tags[0]`은 NULL을 반환하고 `tags[1]`이 첫 번째 원소다. Python 등에서 넘어온 데이터를 처리할 때 인덱스 기준이 다르다는 점을 주의해야 한다.

> **주의:** LIST는 모든 원소가 동일한 타입이어야 한다. 서로 다른 타입을 하나의 LIST에 담으려 하면 DuckDB가 공통 타입으로 자동 변환을 시도하며, 적절한 공통 타입이 없으면 오류가 발생한다. 서로 다른 타입을 담아야 한다면 STRUCT나 UNION 타입을 고려한다.

### STRUCT: 여러 타입 필드의 복합 구조

STRUCT는 스키마가 고정된 복합 객체가 필요할 때 사용한다. JSON 오브젝트처럼 필드 이름과 각기 다른 타입을 가진 값을 하나로 묶어, DuckDB가 열 단위로 압축하고 조회할 수 있도록 한다.

```sql
-- STRUCT 생성
SELECT {'name': 'Alice', 'age': 30, 'city': 'Seoul'};

-- 타입 정의
CREATE TABLE employees (
    id      INTEGER,
    info    STRUCT(name VARCHAR, age INTEGER, dept VARCHAR),
    address STRUCT(city VARCHAR, zip VARCHAR, street VARCHAR)
);

INSERT INTO employees VALUES (
    1,
    {'name': 'Alice', 'age': 30, 'dept': 'Engineering'},
    {'city': 'Seoul', 'zip': '04523', 'street': '테헤란로 123'}
);

-- 필드 접근: . 표기법
SELECT
    info.name,
    info.age,
    address.city
FROM employees;

-- 또는 [] 표기법
SELECT info['name'], address['city'] FROM employees;

-- STRUCT 함수
SELECT
    struct_extract(info, 'name') AS name,   -- 필드 추출
    struct_keys(info) AS keys,              -- 필드 이름 목록
    struct_insert(info, 'salary': 75000),   -- 필드 추가
    row_to_json(info) AS json_str           -- JSON 변환
FROM employees;

-- 집계로 STRUCT 생성
SELECT STRUCT_PACK(
    avg_salary := AVG(salary),
    max_salary := MAX(salary),
    count := COUNT(*)
) AS dept_stats
FROM employees
GROUP BY department;
```

### MAP: 키-값 쌍의 동적 구조

어떤 키가 올지 사전에 알 수 없는 경우, 예를 들어 제품마다 서로 다른 속성 집합을 가지는 데이터라면 MAP이 적합하다. MAP은 STRUCT와 달리 키가 동적이고, 모든 키-값 쌍이 동일한 타입이어야 한다.

```sql
-- MAP 생성
SELECT MAP(['a', 'b', 'c'], [1, 2, 3]);    -- 키 리스트, 값 리스트
SELECT MAP {
    'apple': 1.5,
    'banana': 0.8,
    'cherry': 3.2
};

-- 테이블 정의
CREATE TABLE product_attributes (
    product_id  INTEGER,
    attributes  MAP(VARCHAR, VARCHAR)
);

INSERT INTO product_attributes VALUES
    (1, MAP {'color': 'red', 'size': 'L', 'material': 'cotton'}),
    (2, MAP {'color': 'blue', 'weight': '1.5kg'});

-- MAP 접근 및 함수
SELECT
    attributes['color'] AS color,              -- 키로 접근
    map_keys(attributes) AS keys,             -- 키 목록
    map_values(attributes) AS values,         -- 값 목록
    map_contains(attributes, 'size') AS has_size,  -- 키 존재 여부
    cardinality(attributes) AS pair_count     -- 쌍의 수
FROM product_attributes;

-- MAP 변환
SELECT UNNEST(map_entries(attributes)) AS entry
FROM product_attributes;
-- entry.key, entry.value로 접근 가능
```

### STRUCT vs MAP 비교

```
┌──────────────────────────────────────────────────────────┐
│               STRUCT vs MAP 비교                          │
├────────────────────┬──────────────────────────────────── ┤
│ STRUCT             │ MAP                                  │
├────────────────────┼──────────────────────────────────── ┤
│ 고정 키 (컴파일 시) │ 동적 키 (런타임 시)                   │
│ 필드마다 다른 타입  │ 모든 값이 동일 타입                   │
│ . 또는 [] 접근     │ [] 접근                              │
│ JSON 객체 매핑     │ 딕셔너리/해시맵 매핑                  │
│ 스키마가 명확할 때  │ 키 집합이 가변적일 때                 │
└────────────────────┴──────────────────────────────────── ┘
```

### UNION 타입: 여러 타입 중 하나

SQL의 UNION 연산과 다른 개념으로, 하나의 열에 여러 타입 중 하나를 저장할 수 있다.

```sql
-- UNION 타입 정의
CREATE TABLE events (
    id      INTEGER,
    payload UNION(num INTEGER, str VARCHAR, ts TIMESTAMP)
);

INSERT INTO events VALUES
    (1, 42::INTEGER),
    (2, 'hello'::VARCHAR),
    (3, TIMESTAMP '2024-03-15 10:00:00');

-- UNION 타입 접근
SELECT
    id,
    union_tag(payload) AS type_tag,    -- 현재 저장된 타입 이름
    union_extract(payload, 'num') AS num_val,
    union_extract(payload, 'str') AS str_val
FROM events;

-- 타입에 따른 처리
SELECT id,
    CASE union_tag(payload)
        WHEN 'num' THEN union_extract(payload, 'num')::VARCHAR
        WHEN 'str' THEN union_extract(payload, 'str')
        WHEN 'ts'  THEN union_extract(payload, 'ts')::VARCHAR
    END AS value_str
FROM events;
```
